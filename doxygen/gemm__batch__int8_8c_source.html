1/**
2 * @file gemm_batch_int8.c
3 * @brief Batch GEMM kernels for quantized weights with INT8 activations
4 *
5 * Implements batch matrix multiplication where:
6 * - Activations (A): Q8_0 quantized (INT8 + scale)
7 * - Weights (B): Q5_0 or Q8_0 quantized
8 * - Output (C): FP32
9 *
10 * Operation: C[M,N] = A[M,K] @ B[N,K]^T (B is transposed/row-major weights)
11 *
12 * Instruction Set Implementations:
13 * - Scalar: Reference implementation for correctness verification
14 * - AVX: 256-bit SIMD (8 floats, or 32 int8s)
15 * - AVX-512: 512-bit SIMD (16 floats, or 64 int8s)
16 * - AMX: Intel Advanced Matrix Extensions (tile-based, requires Sapphire Rapids+)
17 *
18 * Design Philosophy:
19 * - Every kernel MUST produce bit-identical results to scalar reference
20 * - Comprehensive testing against llama.cpp ensures correctness
21 * - Performance optimizations never compromise accuracy
22 *
23 * @author C-Kernel-Engine Team
24 * @date 2024
25 */

58/* ============================================================================
59 * SECTION 1: GEMM Q8_0 x Q8_0 -> FP32
60 *
61 * Both weights and activations are Q8_0 quantized.
62 * This is the simplest case - direct INT8 x INT8 -> INT32 accumulation.

65/**
66 * @brief Scalar reference: gemm_nt_q8_0_q8_0
67 *
68 * C[m,n] = sum_k( dequant(A[m,k]) * dequant(B[n,k]) )
69 * = sum_blocks( d_a * d_b * sum_j(a_qs[j] * b_qs[j]) )
70 *
71 * @param A Input activations [M, K] in Q8_0 format
72 * @param B Weight matrix [N, K] in Q8_0 format (row-major, each row is one output)
73 * @param C Output matrix [M, N] in FP32
74 * @param M Number of tokens (batch size)
75 * @param N Number of output features (rows in B)
76 * @param K Number of input features (must be multiple of 32)
77 */
78void gemm_nt_q8_0_q8_0_ref(
79 const void *A,
80 const void *B,
81 float *C,
82 int M, int N, int K)
83{
84 const int nb = K / QK8_0; /* Number of blocks per row */
85 const block_q8_0 *a_blocks = (const block_q8_0 *)A;
86 const block_q8_0 *b_blocks = (const block_q8_0 *)B;
87
88 for (int m = 0; m < M; m++) {
89 const block_q8_0 *a_row = a_blocks + (size_t)m * nb;
90
91 for (int n = 0; n < N; n++) {
92 const block_q8_0 *b_row = b_blocks + (size_t)n * nb;
93 float sum = 0.0f;
94
95 for (int ib = 0; ib < nb; ib++) {
96 const float d_a = CK_FP16_TO_FP32(a_row[ib].d);
97 const float d_b = CK_FP16_TO_FP32(b_row[ib].d);
98 const float d = d_a * d_b;
99
100 int32_t sumi = 0;
101 for (int j = 0; j < QK8_0; j++) {
102 sumi += (int32_t)a_row[ib].qs[j] * (int32_t)b_row[ib].qs[j];
103 }
104
105 sum += d * (float)sumi;
106 }
107
108 C[(size_t)m * N + n] = sum;
109 }
110 }
111}

113#if defined(__AVX2__)
114/**
115 * @brief AVX2 implementation: gemm_nt_q8_0_q8_0
116 *
117 * Uses 256-bit vectors to process 32 int8 values at once.
118 * Requires AVX2 for _mm256_cvtepi8_epi16, _mm256_madd_epi16, etc.
119 * Accumulates in INT32, then scales by d_a * d_b.
120 */
121void gemm_nt_q8_0_q8_0_avx2(
122 const void *A,
123 const void *B,
124 float *C,
125 int M, int N, int K)
126{
127 const int nb = K / QK8_0;
128 const block_q8_0 *a_blocks = (const block_q8_0 *)A;
129 const block_q8_0 *b_blocks = (const block_q8_0 *)B;
130
131 for (int m = 0; m < M; m++) {
132 const block_q8_0 *a_row = a_blocks + (size_t)m * nb;
133
134 for (int n = 0; n < N; n++) {
135 const block_q8_0 *b_row = b_blocks + (size_t)n * nb;
136 float sum = 0.0f;
137
138 for (int ib = 0; ib < nb; ib++) {
139 const float d_a = CK_FP16_TO_FP32(a_row[ib].d);
140 const float d_b = CK_FP16_TO_FP32(b_row[ib].d);
141 const float d = d_a * d_b;
142
143 /* Load 32 int8 values from A and B */
144 __m256i va = _mm256_loadu_si256((const __m256i *)a_row[ib].qs);
145 __m256i vb = _mm256_loadu_si256((const __m256i *)b_row[ib].qs);
146
147 /* Split into 16-bit for multiplication without overflow
148 * Process low 16 bytes and high 16 bytes separately */
149 __m128i va_lo = _mm256_castsi256_si128(va);
150 __m128i va_hi = _mm256_extracti128_si256(va, 1);
151 __m128i vb_lo = _mm256_castsi256_si128(vb);
152 __m128i vb_hi = _mm256_extracti128_si256(vb, 1);
153
154 /* Extend to 16-bit and multiply */
155 __m256i va_lo_16 = _mm256_cvtepi8_epi16(va_lo);
156 __m256i vb_lo_16 = _mm256_cvtepi8_epi16(vb_lo);
157 __m256i va_hi_16 = _mm256_cvtepi8_epi16(va_hi);
158 __m256i vb_hi_16 = _mm256_cvtepi8_epi16(vb_hi);
159
160 __m256i prod_lo = _mm256_mullo_epi16(va_lo_16, vb_lo_16);
161 __m256i prod_hi = _mm256_mullo_epi16(va_hi_16, vb_hi_16);
162
163 /* Horizontal sum: extend to 32-bit and add */
164 __m256i sum_lo = _mm256_madd_epi16(prod_lo, _mm256_set1_epi16(1));
165 __m256i sum_hi = _mm256_madd_epi16(prod_hi, _mm256_set1_epi16(1));
166 __m256i sum_32 = _mm256_add_epi32(sum_lo, sum_hi);
167
168 /* Reduce 8 x int32 to single int32 */
169 __m128i sum_128 = _mm_add_epi32(
170 _mm256_castsi256_si128(sum_32),
171 _mm256_extracti128_si256(sum_32, 1)
172 );
173 sum_128 = _mm_add_epi32(sum_128, _mm_srli_si128(sum_128, 8));
174 sum_128 = _mm_add_epi32(sum_128, _mm_srli_si128(sum_128, 4));
175 int32_t sumi = _mm_cvtsi128_si32(sum_128);
176
177 sum += d * (float)sumi;
178 }
179
180 C[(size_t)m * N + n] = sum;
181 }
182 }
183}
184#endif /* __AVX2__ */

187/**
188 * @brief AVX-512 implementation: gemm_nt_q8_0_q8_0
189 *
190 * Uses 512-bit vectors to process 64 int8 values at once.
191 * With VNNI, can use _mm512_dpbusd for even faster int8 dot products.
192 */
193void gemm_nt_q8_0_q8_0_avx512(
194 const void *A,
195 const void *B,
196 float *C,
197 int M, int N, int K)
198{
199 const int nb = K / QK8_0;
200 const block_q8_0 *a_blocks = (const block_q8_0 *)A;
201 const block_q8_0 *b_blocks = (const block_q8_0 *)B;
202
203 for (int m = 0; m < M; m++) {
204 const block_q8_0 *a_row = a_blocks + (size_t)m * nb;
205
206 for (int n = 0; n < N; n++) {
207 const block_q8_0 *b_row = b_blocks + (size_t)n * nb;
208 float sum = 0.0f;
209
210 for (int ib = 0; ib < nb; ib++) {
211 const float d_a = CK_FP16_TO_FP32(a_row[ib].d);
212 const float d_b = CK_FP16_TO_FP32(b_row[ib].d);
213 const float d = d_a * d_b;
214
215 /* Load 32 int8 values - use 256-bit load, extend to 512-bit for processing */
216 __m256i va_256 = _mm256_loadu_si256((const __m256i *)a_row[ib].qs);
217 __m256i vb_256 = _mm256_loadu_si256((const __m256i *)b_row[ib].qs);
218
219 /* Extend int8 to int16 for multiplication */
220 __m512i va_16 = _mm512_cvtepi8_epi16(va_256);
221 __m512i vb_16 = _mm512_cvtepi8_epi16(vb_256);
222
223 /* Multiply 32 pairs of int16 -> int16 (no overflow for int8*int8) */
224 __m512i prod = _mm512_mullo_epi16(va_16, vb_16);
225
226 /* Sum adjacent pairs to int32: madd adds pairs of int16 products */
227 __m512i sum_32 = _mm512_madd_epi16(prod, _mm512_set1_epi16(1));
228
229 /* Reduce all 16 int32 lanes to single int32 */
230 int32_t sumi = _mm512_reduce_add_epi32(sum_32);
231
232 sum += d * (float)sumi;
233 }
234
235 C[(size_t)m * N + n] = sum;
236 }
237 }

302/* ============================================================================
303 * SECTION 2: GEMM Q5_0 x Q8_0 -> FP32
304 *
305 * Weights are Q5_0 (5-bit), activations are Q8_0 (8-bit).
306 * Q5_0 requires unpacking: 4 bits from qs[] + 1 bit from qh[].
307 * ============================================================================ */
308
309/**
310 * @brief Scalar reference: gemm_nt_q5_0_q8_0
311 *
312 * Q5_0 weight reconstruction:
313 * weight[j] = d * ((qs_nibble | (qh_bit << 4)) - 16)
314 *
315 * For j in 0..15: use low nibble + qh bit j
316 * For j in 16..31: use high nibble + qh bit (j+16) -> actually bit (j) for j=16..31
317 *
318 * @param A Input activations [M, K] in Q8_0 format
319 * @param B Weight matrix [N, K] in Q5_0 format
320 * @param C Output matrix [M, N] in FP32
321 * @param M Number of tokens (batch size)
322 * @param N Number of output features
323 * @param K Number of input features (must be multiple of 32)
324 */
325void gemm_nt_q5_0_q8_0_ref(
326 const void *A,
327 const void *B,
328 float *C,
329 int M, int N, int K)
330{
331 const int nb = K / QK5_0;
332 const block_q8_0 *a_blocks = (const block_q8_0 *)A;
333 const block_q5_0 *b_blocks = (const block_q5_0 *)B;
334
335 for (int m = 0; m < M; m++) {
336 const block_q8_0 *a_row = a_blocks + (size_t)m * nb;
337
338 for (int n = 0; n < N; n++) {
339 const block_q5_0 *b_row = b_blocks + (size_t)n * nb;
340 float sum = 0.0f;
341
342 for (int ib = 0; ib < nb; ib++) {
343 const float d_a = CK_FP16_TO_FP32(a_row[ib].d);
344 const float d_b = CK_FP16_TO_FP32(b_row[ib].d);
345 const float d = d_a * d_b;
346
347 /* Load high bits as 32-bit value */
348 uint32_t qh;
349 memcpy(&qh, b_row[ib].qh, sizeof(qh));
350
351 int32_t sumi = 0;
352
353 /* Process 32 weights: j=0..15 uses low nibble, j=16..31 uses high nibble */
354 for (int j = 0; j < 16; j++) {
355 /* First 16 weights: low nibble + qh bit j */
356 const uint8_t xh_0 = ((qh >> j) & 1) << 4;
357 const int8_t w0 = (int8_t)(((b_row[ib].qs[j] & 0x0F) | xh_0) - 16);
358
359 /* Second 16 weights: high nibble + qh bit (j+16) */
360 const uint8_t xh_1 = ((qh >> (j + 16)) & 1) << 4;
361 const int8_t w1 = (int8_t)(((b_row[ib].qs[j] >> 4) | xh_1) - 16);
362
363 /* Accumulate with activation values */
364 sumi += (int32_t)w0 * (int32_t)a_row[ib].qs[j];
365 sumi += (int32_t)w1 * (int32_t)a_row[ib].qs[j + 16];
366 }
367
368 sum += d * (float)sumi;
369 }
370
371 C[(size_t)m * N + n] = sum;
372 }
373 }
374}

376/* ============================================================================
377 * SECTION 3: AMX Implementation (Intel Advanced Matrix Extensions)
378 *
379 * AMX uses tile registers (TMM0-TMM7) for matrix operations.
380 * Each tile can hold up to 16 rows x 64 bytes (1KB).
381 *
382 * Key operations:
383 * _tile_loadd: Load tile from memory
384 * _tile_dpbssd: Signed int8 dot product accumulate (A signed, B signed)
385 * _tile_stored: Store tile to memory
386 *
387 * Requirements:
388 * - Sapphire Rapids or later CPU
389 * - __AMX_INT8__ defined
390 * - OS support (XSAVE/XRSTOR for tiles)
391 * ============================================================================ */
392
393#if HAS_AMX
394
395/* AMX tile configuration */
396typedef struct {
397 uint8_t palette_id;
398 uint8_t start_row;
399 uint8_t reserved[14];
400 uint16_t colsb[8];
401 uint8_t rows[8];
402} tile_config_t;
403
404static void amx_tile_config_init(void)
405{
406 static __thread int initialized = 0;
407 if (initialized) return;
408
409 tile_config_t tc = {0};
410 tc.palette_id = 1;
411
412 /* Configure tiles for our GEMM pattern:
413 * TMM0: accumulator C (16 rows x 16 cols of int32)
414 * TMM1: A tile (16 rows x 64 bytes = 64 int8 per row)
415 * TMM2: B tile (16 rows x 64 bytes)
416 */
417 tc.rows[0] = 16; tc.colsb[0] = 64; /* TMM0: 16x16 int32 */
418 tc.rows[1] = 16; tc.colsb[1] = 64; /* TMM1: A */
419 tc.rows[2] = 16; tc.colsb[2] = 64; /* TMM2: B */
420 tc.rows[3] = 16; tc.colsb[3] = 64; /* TMM3: spare */
421 tc.rows[4] = 16; tc.colsb[4] = 64; /* TMM4: spare */
422 tc.rows[5] = 16; tc.colsb[5] = 64; /* TMM5: spare */
423 tc.rows[6] = 16; tc.colsb[6] = 64; /* TMM6: spare */
424 tc.rows[7] = 16; tc.colsb[7] = 64; /* TMM7: spare */
425
426 _tile_loadconfig(&tc);
427 initialized = 1;
428}
