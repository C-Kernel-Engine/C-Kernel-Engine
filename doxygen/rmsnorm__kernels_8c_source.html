8// AVX1 horizontal sum helper (no _mm256_reduce_add_ps in AVX1)
9#if defined(__AVX__) && !defined(__AVX512F__)
10static inline float hsum256_ps_rmsnorm(__m256 v) {
11 // Sum upper and lower 128-bit lanes
12 __m128 hi = _mm256_extractf128_ps(v, 1);
13 __m128 lo = _mm256_castps256_ps128(v);
14 __m128 sum128 = _mm_add_ps(lo, hi);
15 // Horizontal add within 128-bit lane
16 sum128 = _mm_hadd_ps(sum128, sum128);
17 sum128 = _mm_hadd_ps(sum128, sum128);
18 return _mm_cvtss_f32(sum128);

22// RMSNorm forward:
23// For each token t:
24// r = sqrt( (1/D) * sum_i x_i^2 + eps )
25// rstd = 1 / r
26// x_hat_i = x_i * rstd
27// y_i = gamma_i * x_hat_i
28//
29// We cache rstd per token for use in backward.
30void rmsnorm_forward(const float *input,
31 const float *gamma,
32 float *output,
33 float *rstd_cache,
34 int tokens,
35 int d_model,
36 int aligned_embed_dim,
37 float eps)
38{
39 int T = tokens;
40 int D = d_model;
41 int aligned = aligned_embed_dim;
42
43 for (int t = 0; t < T; ++t) {
44 const float *x = input + (size_t)t * aligned;
45 float *y = output + (size_t)t * aligned;
46
47#if defined(__AVX512F__)
48 // AVX-512: Process 16 floats at a time
49 __m512 sum_sq_vec = _mm512_setzero_ps();
50 int d = 0;
51
52 // Vectorized sum of squares
53 for (; d + 16 <= D; d += 16) {
54 __m512 xv = _mm512_loadu_ps(&x[d]);
55 sum_sq_vec = _mm512_fmadd_ps(xv, xv, sum_sq_vec);
56 }
57 float sum_sq = _mm512_reduce_add_ps(sum_sq_vec);
58
59 // Handle remaining elements
60 for (; d < D; ++d) {
61 sum_sq += x[d] * x[d];
62 }
63
64 float mean_sq = sum_sq / (float)D;
65 float rstd = 1.0f / sqrtf(mean_sq + eps);
66 if (rstd_cache) {
67 rstd_cache[t] = rstd;
68 }
69
70 // Apply normalization and scale (vectorized)
71 __m512 rstd_vec = _mm512_set1_ps(rstd);
72 d = 0;
73 for (; d + 16 <= D; d += 16) {
74 __m512 xv = _mm512_loadu_ps(&x[d]);
75 __m512 gv = _mm512_loadu_ps(&gamma[d]);
76 __m512 x_hat = _mm512_mul_ps(xv, rstd_vec);
77 __m512 yv = _mm512_mul_ps(x_hat, gv);
78 _mm512_storeu_ps(&y[d], yv);
79 }
80 // Handle remaining elements
81 for (; d < D; ++d) {
82 y[d] = x[d] * rstd * gamma[d];
83 }
84
85#elif defined(__AVX__)
86 // AVX: Process 8 floats at a time
87 __m256 sum_sq_vec = _mm256_setzero_ps();
88 int d = 0;
89
90 // Vectorized sum of squares (no FMA in AVX1, use mul + add)
91 for (; d + 8 <= D; d += 8) {
92 __m256 xv = _mm256_loadu_ps(&x[d]);
93 __m256 xv_sq = _mm256_mul_ps(xv, xv);
94 sum_sq_vec = _mm256_add_ps(sum_sq_vec, xv_sq);
95 }
96 float sum_sq = hsum256_ps_rmsnorm(sum_sq_vec);
97
98 // Handle remaining elements
99 for (; d < D; ++d) {
100 sum_sq += x[d] * x[d];
101 }
102
103 float mean_sq = sum_sq / (float)D;
104 float rstd = 1.0f / sqrtf(mean_sq + eps);
105 if (rstd_cache) {
106 rstd_cache[t] = rstd;
107 }
108
109 // Apply normalization and scale (vectorized)
110 __m256 rstd_vec = _mm256_set1_ps(rstd);
111 d = 0;
112 for (; d + 8 <= D; d += 8) {
113 __m256 xv = _mm256_loadu_ps(&x[d]);
114 __m256 gv = _mm256_loadu_ps(&gamma[d]);
115 __m256 x_hat = _mm256_mul_ps(xv, rstd_vec);
116 __m256 yv = _mm256_mul_ps(x_hat, gv);
117 _mm256_storeu_ps(&y[d], yv);
118 }
119 // Handle remaining elements
120 for (; d < D; ++d) {
121 y[d] = x[d] * rstd * gamma[d];
122 }
123
124#else
125 // Scalar fallback
126 double sum_sq = 0.0;
127 for (int d = 0; d < D; ++d) {
128 double v = (double)x[d];
129 sum_sq += v * v;
130 }
131 double mean_sq = sum_sq / (double)D;
132 double r = sqrt(mean_sq + (double)eps);
133 float rstd = (float)(1.0 / r);
134 if (rstd_cache) {
135 rstd_cache[t] = rstd;
136 }
137
138 // Apply normalization and scale
139 for (int d = 0; d < D; ++d) {
140 float x_hat = x[d] * rstd;
141 y[d] = x_hat * gamma[d];
142 }
143#endif
144
145 // Zero padding (if any)
146 for (int d = D; d < aligned; ++d) {
147 y[d] = 0.0f;
148 }
149 }
150}

152// RMSNorm backward:
153// Given dY, X, gamma, and cached rstd per token, compute:
154// dX and dGamma.
155//
156// From derivation:
157// x_hat_i = x_i * rstd_t
158// m = (1/D) * sum_j (dY_j * gamma_j * x_hat_j)
159// dX_i = rstd_t * (dY_i * gamma_i - x_hat_i * m)
160// dGamma_i = sum_t (dY_i * x_hat_i)
161//
162// We do not include a beta parameter for RMSNorm here.
163void rmsnorm_backward(const float *d_output,
164 const float *input,
165 const float *gamma,
166 const float *rstd_cache,
167 float *d_input,
168 float *d_gamma,
169 int tokens,
170 int d_model,
171 int aligned_embed_dim)
172{
173 int T = tokens;
174 int D = d_model;
175 int aligned = aligned_embed_dim;
176
177 // Zero parameter gradients
178#if defined(__AVX512F__)
179 {
180 int d = 0;
181 for (; d + 16 <= D; d += 16) {
182 _mm512_storeu_ps(&d_gamma[d], _mm512_setzero_ps());
183 }
184 for (; d < D; ++d) {
185 d_gamma[d] = 0.0f;
186 }
187 }
188#elif defined(__AVX__)
189 {
190 int d = 0;
191 for (; d + 8 <= D; d += 8) {
192 _mm256_storeu_ps(&d_gamma[d], _mm256_setzero_ps());
193 }
194 for (; d < D; ++d) {
195 d_gamma[d] = 0.0f;
196 }
197 }
198#else
199 for (int d = 0; d < D; ++d) {
200 d_gamma[d] = 0.0f;
201 }
202#endif
203
204 for (int t = 0; t < T; ++t) {
205 const float *x = input + (size_t)t * aligned;
206 const float *dY = d_output + (size_t)t * aligned;
207 float *dX = d_input + (size_t)t * aligned;
208
209 float rstd = rstd_cache[t];
210
211#if defined(__AVX512F__)
212 // Compute m = (1/D) * sum_j (dY_j * gamma_j * x_hat_j)
213 __m512 rstd_vec = _mm512_set1_ps(rstd);
214 __m512 sum_vec = _mm512_setzero_ps();
215 int d = 0;
216
217 for (; d + 16 <= D; d += 16) {
218 __m512 xv = _mm512_loadu_ps(&x[d]);
219 __m512 dyv = _mm512_loadu_ps(&dY[d]);
220 __m512 gv = _mm512_loadu_ps(&gamma[d]);
221 __m512 x_hat = _mm512_mul_ps(xv, rstd_vec);
222 // sum += dY * gamma * x_hat
223 __m512 prod = _mm512_mul_ps(dyv, gv);
224 sum_vec = _mm512_fmadd_ps(prod, x_hat, sum_vec);
225 }
226 float sum_dY_g_xhat = _mm512_reduce_add_ps(sum_vec);
227
228 // Handle remaining elements
229 for (; d < D; ++d) {
230 float x_hat = x[d] * rstd;
231 sum_dY_g_xhat += dY[d] * gamma[d] * x_hat;
232 }
233 float m = sum_dY_g_xhat / (float)D;
234
235 // Compute dX and accumulate dGamma (vectorized)
236 __m512 m_vec = _mm512_set1_ps(m);
237 d = 0;
238 for (; d + 16 <= D; d += 16) {
239 __m512 xv = _mm512_loadu_ps(&x[d]);
240 __m512 dyv = _mm512_loadu_ps(&dY[d]);
241 __m512 gv = _mm512_loadu_ps(&gamma[d]);
242 __m512 dgv = _mm512_loadu_ps(&d_gamma[d]);
243
244 __m512 x_hat = _mm512_mul_ps(xv, rstd_vec);
245
246 // dX = rstd * (dY * gamma - x_hat * m)
247 __m512 dy_g = _mm512_mul_ps(dyv, gv);
248 __m512 xhat_m = _mm512_mul_ps(x_hat, m_vec);
249 __m512 diff = _mm512_sub_ps(dy_g, xhat_m);
250 __m512 dxv = _mm512_mul_ps(rstd_vec, diff);
251 _mm512_storeu_ps(&dX[d], dxv);
252
253 // d_gamma += dY * x_hat
254 dgv = _mm512_fmadd_ps(dyv, x_hat, dgv);
255 _mm512_storeu_ps(&d_gamma[d], dgv);
256 }
257 // Handle remaining elements
258 for (; d < D; ++d) {
259 float x_hat = x[d] * rstd;
260 float dy = dY[d];
261 dX[d] = rstd * (dy * gamma[d] - x_hat * m);
262 d_gamma[d] += dy * x_hat;
263 }
264
265#elif defined(__AVX__)
266 // Compute m = (1/D) * sum_j (dY_j * gamma_j * x_hat_j)
267 __m256 rstd_vec = _mm256_set1_ps(rstd);
268 __m256 sum_vec = _mm256_setzero_ps();
269 int d = 0;
270
271 for (; d + 8 <= D; d += 8) {
272 __m256 xv = _mm256_loadu_ps(&x[d]);
273 __m256 dyv = _mm256_loadu_ps(&dY[d]);
274 __m256 gv = _mm256_loadu_ps(&gamma[d]);
275 __m256 x_hat = _mm256_mul_ps(xv, rstd_vec);
276 // sum += dY * gamma * x_hat (no FMA, use mul + mul + add)
277 __m256 prod = _mm256_mul_ps(dyv, gv);
278 __m256 prod2 = _mm256_mul_ps(prod, x_hat);
279 sum_vec = _mm256_add_ps(sum_vec, prod2);
280 }

282

342 dX[d] = 0.0f;
343 }
