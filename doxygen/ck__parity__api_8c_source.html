444 void ck_test_rope(float *q, float *k,
445  int n_tokens, int n_heads, int n_heads_kv, int head_dim,
446  int pos_offset, float theta)
447 {
448  /* Precompute cos/sin cache */
449  int half_dim = head_dim / 2;
450  int max_seq = pos_offset + n_tokens;
451 
452  float *cos_cache = (float *)malloc(max_seq * half_dim * sizeof(float));
453  float *sin_cache = (float *)malloc(max_seq * half_dim * sizeof(float));
454  if (!cos_cache || !sin_cache) {
455  free(cos_cache);
456  free(sin_cache);
457  return;
458  }
459 
460  rope_precompute_cache(cos_cache, sin_cache, max_seq, head_dim, theta);
461 
462  /* CK RoPE expects layout [num_heads, num_tokens, head_dim]
463  * Reshape from [n_tokens, n_heads * head_dim] to [n_heads, n_tokens, head_dim]
464  */
465  float *q_reorder = (float *)malloc(n_heads * n_tokens * head_dim * sizeof(float));
466  float *k_reorder = (float *)malloc(n_heads_kv * n_tokens * head_dim * sizeof(float));
467 
468  if (q_reorder && k_reorder) {
469  /* Reorder Q: [T, H*D] -> [H, T, D] */
470  for (int t = 0; t < n_tokens; t++) {
471  for (int h = 0; h < n_heads; h++) {
472  for (int d = 0; d < head_dim; d++) {
473  q_reorder[h * n_tokens * head_dim + t * head_dim + d] =
474  q[t * n_heads * head_dim + h * head_dim + d];
475  }
476  }
477  }
478 
479  /* Reorder K: [T, H_kv*D] -> [H_kv, T, D] */
480  for (int t = 0; t < n_tokens; t++) {
481  for (int h = 0; h < n_heads_kv; h++) {
482  for (int d = 0; d < head_dim; d++) {
483  k_reorder[h * n_tokens * head_dim + t * head_dim + d] =
484  k[t * n_heads_kv * head_dim + h * head_dim + d];
485  }
486  }
487  }
488 
489  /* Apply RoPE */
490  rope_forward_qk(q_reorder, k_reorder,
491  cos_cache, sin_cache,
492  n_heads, n_heads_kv, n_tokens,
493  head_dim, head_dim, pos_offset);
494 
495  /* Reorder back: [H, T, D] -> [T, H*D] */
496  for (int t = 0; t < n_tokens; t++) {
497  for (int h = 0; h < n_heads; h++) {
498  for (int d = 0; d < head_dim; d++) {
499  q[t * n_heads * head_dim + h * head_dim + d] =
500  q_reorder[h * n_tokens * head_dim + t * head_dim + d];
501  }
502  }
503  }
504 
505  for (int t = 0; t < n_tokens; t++) {
506  for (int h = 0; h < n_heads_kv; h++) {
507  for (int d = 0; d < head_dim; d++) {
508  k[t * n_heads_kv * head_dim + h * head_dim + d] =
509  k_reorder[h * n_tokens * head_dim + t * head_dim + d];
510  }
511  }
512  }
513  }
514 
515  free(q_reorder);
516  free(k_reorder);
517  free(cos_cache);
518  free(sin_cache);
519 }

521 void ck_test_rope_interleaved(float *q, float *k,
522  int n_tokens, int n_heads, int n_heads_kv, int head_dim,
523  int pos_offset, float theta)
524 {
525  /* Interleaved RoPE format (matches llama.cpp):
526  * (x0, x1) -> (x0*cos - x1*sin, x0*sin + x1*cos)
527  * Applied to consecutive pairs of elements
528  */
529 
530  /* Precompute inverse frequencies */
531  float *inv_freq = (float *)malloc((head_dim / 2) * sizeof(float));
532  if (!inv_freq) return;
533 
534  for (int i = 0; i < head_dim / 2; i++) {
535  inv_freq[i] = 1.0f / powf(theta, (float)(2 * i) / head_dim);
536  }
537 
538  /* Apply RoPE to Q */
539  for (int t = 0; t < n_tokens; t++) {
540  int pos = pos_offset + t;

573  }
