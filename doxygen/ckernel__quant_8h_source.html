28/* ============================================================================
29 * Q4_0: Simple 4-bit Quantization
30 * - 32 weights per block
31 * - 1 FP16 scale per block
32 * - 18 bytes per 32 weights = 4.5 bits/weight
33 * ============================================================================ */
34
35#define QK4_0 32
36
37typedef struct {
38 ck_half d; /* 2 bytes: scale (delta) */
39 uint8_t qs[QK4_0 / 2]; /* 16 bytes: 32 x 4-bit weights (2 per byte) */
40} block_q4_0;
41/* Total: 18 bytes per 32 weights */
42
43/* ============================================================================
44 * Q4_1: Simple 4-bit Quantization with Min
45 * - 32 weights per block
46 * - 2 FP16 values: scale (d) and min (m)
47 * - 20 bytes per 32 weights = 5.0 bits/weight
48 * ============================================================================ */
49
50#define QK4_1 32
51
52typedef struct {
53 ck_half d; /* 2 bytes: scale (delta) */
54 ck_half m; /* 2 bytes: minimum */
55 uint8_t qs[QK4_1 / 2]; /* 16 bytes: 32 x 4-bit weights (2 per byte) */
56} block_q4_1;
57/* Total: 20 bytes per 32 weights */
58
59/* ============================================================================
60 * Q5_0: Simple 5-bit Quantization
61 * - 32 weights per block
62 * - 1 FP16 scale per block
63 * - Low 4 bits stored like Q4_0, high 1 bit packed separately
64 * - 22 bytes per 32 weights = 5.5 bits/weight
65 * ============================================================================ */
66
67#define QK5_0 32
68
69typedef struct {
70 ck_half d; /* 2 bytes: scale (delta) */
71 uint8_t qh[4]; /* 4 bytes: high 1-bit of each weight (32 bits total) */
72 uint8_t qs[QK5_0 / 2]; /* 16 bytes: low 4-bits of 32 weights (2 per byte) */
73} block_q5_0;
74/* Total: 22 bytes per 32 weights */
75
76/* ============================================================================
77 * Q5_1: Simple 5-bit Quantization with Min
78 * - 32 weights per block
79 * - 2 FP16 values: scale (d) and min (m)
80 * - Low 4 bits stored like Q4_1, high 1 bit packed separately
81 * - 24 bytes per 32 weights = 6.0 bits/weight
82 * ============================================================================ */
83
84#define QK5_1 32
85
86typedef struct {
87 ck_half d; /* 2 bytes: scale (delta) */
88 ck_half m; /* 2 bytes: minimum */
89 uint8_t qh[4]; /* 4 bytes: high 1-bit of each weight (32 bits total) */
90 uint8_t qs[QK5_1 / 2]; /* 16 bytes: low 4-bits of 32 weights (2 per byte) */
91} block_q5_1;
92/* Total: 24 bytes per 32 weights */
93
94/* ============================================================================
95 * Q8_0: Simple 8-bit Quantization
96 * - 32 weights per block
97 * - 1 FP16 scale per block
98 * - 34 bytes per 32 weights = 8.5 bits/weight
99 * ============================================================================ */
100
101#define QK8_0 32
102
103typedef struct {
104 ck_half d; /* 2 bytes: scale */
105 int8_t qs[QK8_0]; /* 32 bytes: 32 x 8-bit signed weights */
106} block_q8_0;
107/* Total: 34 bytes per 32 weights */
108
109/* ============================================================================
110 * Q4_K: K-Quant 4-bit with Nested Scales (Primary Target)
111 * - 256 weights per super-block
112 * - 8 sub-blocks of 32 weights each
113 * - Two-level scaling: super-block FP16 + sub-block 6-bit
114 * - 144 bytes per 256 weights = 4.5 bits/weight
115 *
116 * This is the format used by Q4_K_M, Q4_K_S, Q4_K_L variants.
117 * The M/S/L suffix indicates quantization aggressiveness, not structure.
118 * ============================================================================ */
119
120#define QK_K 256
121#define K_SCALE_SIZE 12
122
123typedef struct {
124 ck_half d; /* 2 bytes: super-block scale */
125 ck_half dmin; /* 2 bytes: super-block minimum */
126 uint8_t scales[K_SCALE_SIZE]; /* 12 bytes: 8 sub-block scales + 8 sub-block mins (6-bit packed) */
127 uint8_t qs[QK_K / 2]; /* 128 bytes: 256 x 4-bit weights */
128} block_q4_K;
129/* Total: 144 bytes per 256 weights */
130
131/* ============================================================================
132 * Q6_K: K-Quant 6-bit (per-16 scales)
133 * - 256 weights per block
134 * - 16 sub-blocks of 16 weights each
135 * - Stored as low 4 bits (ql) + high 2 bits (qh) + int8 scales
136 * ============================================================================ */
137
138typedef struct {
139 uint8_t ql[QK_K / 2]; /* 128 bytes: low 4 bits */
140 uint8_t qh[QK_K / 4]; /* 64 bytes: high 2 bits */
141 int8_t scales[QK_K / 16]; /* 16 bytes: 16 sub-block scales */
142 ck_half d; /* 2 bytes: super-block scale */
143} block_q6_K;
144/* Total: 210 bytes per 256 weights */
145
146/* ============================================================================
147 * Q8_K: K-Quant 8-bit (used for activations in some ops)
148 * - 256 weights per super-block
149 * - 1 FP32 scale per block (not FP16 like others!)
150 * ============================================================================ */
151
152typedef struct {
153 float d; /* 4 bytes: scale */
154 int8_t qs[QK_K]; /* 256 bytes: 256 x 8-bit signed weights */
155 int16_t bsums[QK_K / 16]; /* 32 bytes: block sums for optimization */
156} block_q8_K;
157/* Total: 292 bytes per 256 weights */
158

163/**
164 * @brief Get the block size (number of weights per block) for a quant type
165 */
166static inline size_t ck_quant_block_size(int type) {
167 switch (type) {
168 case 0: return QK4_0; /* Q4_0 */
169 case 1: return QK8_0; /* Q8_0 */
170 case 2: return QK_K; /* Q4_K */
171 case 3: return QK_K; /* Q8_K */
172 case CK_DT_Q4_1: return QK4_1;
173 case CK_DT_Q5_0: return QK5_0;
174 case CK_DT_Q5_1: return QK5_1;
175 case CK_DT_Q6_K: return QK_K;
176 default: return 1;
177 }
178}
179
180/**
181 * @brief Get the byte size per block for a quant type
182 */
183static inline size_t ck_quant_type_size(int type) {
184 switch (type) {
185 case 0: return sizeof(block_q4_0);
186 case 1: return sizeof(block_q8_0);
187 case 2: return sizeof(block_q4_K);
188 case 3: return sizeof(block_q8_K);
189 case CK_DT_Q4_1: return sizeof(block_q4_1);
190 case CK_DT_Q5_0: return sizeof(block_q5_0);
191 case CK_DT_Q5_1: return sizeof(block_q5_1);
192 case CK_DT_Q6_K: return sizeof(block_q6_K);
193 default: return 4; /* FP32 */
194 }
195}

197/**
198 * @brief Calculate total bytes needed for n_elements with given quant type
199 */
200static inline size_t ck_quant_row_size(int type, int64_t n_elements) {
201 size_t block_size = ck_quant_block_size(type);
202 size_t type_size = ck_quant_type_size(type);
203 return (n_elements / block_size) * type_size;
204}

206/* ============================================================================
207 * Q4_K Scale Unpacking Utilities
208 *
209 * The scales[12] array packs 8 scales and 8 mins in 6-bit format.
210 * Unpacking is non-trivial due to the bit packing.
211 * ============================================================================ */
212
213/**
214 * @brief Unpack Q4_K sub-block scales and mins
215 *
216 * @param scales The packed scales[12] array from block_q4_K
217 * @param sc Output: 8 unpacked scale values (multiply by super-block d)
218 * @param m Output: 8 unpacked min values (multiply by super-block dmin)
219 *
220 * This matches llama.cpp's get_scale_min_k4() function exactly.
221 * The 12-byte scales array layout:
222 * - bytes 0-3: 6-bit scales[0-3] (high 2 bits used for scales[4-7])
223 * - bytes 4-7: 6-bit mins[0-3] (high 2 bits used for mins[4-7])
224 * - bytes 8-11: low 4 bits for scales[4-7], high 4 bits for mins[4-7]
225 */
226static inline void unpack_q4_k_scales(const uint8_t *scales,
227 uint8_t *sc, uint8_t *m) {
228 /* Direct 6-bit values for indices 0-3 */
229 sc[0] = scales[0] & 0x3F;
230 sc[1] = scales[1] & 0x3F;
231 sc[2] = scales[2] & 0x3F;
232 sc[3] = scales[3] & 0x3F;
233
234 m[0] = scales[4] & 0x3F;
235 m[1] = scales[5] & 0x3F;
236 m[2] = scales[6] & 0x3F;
237 m[3] = scales[7] & 0x3F;
238
239 /* 6-bit values for indices 4-7: low 4 bits from bytes 8-11,
240 * high 2 bits from upper bits of bytes 0-3 (scales) and 4-7 (mins) */
241 sc[4] = (scales[8] & 0x0F) | ((scales[0] >> 6) << 4);
242 sc[5] = (scales[9] & 0x0F) | ((scales[1] >> 6) << 4);
243 sc[6] = (scales[10] & 0x0F) | ((scales[2] >> 6) << 4);
244 sc[7] = (scales[11] & 0x0F) | ((scales[3] >> 6) << 4);
245
246 m[4] = (scales[8] >> 4) | ((scales[4] >> 6) << 4);
247 m[5] = (scales[9] >> 4) | ((scales[5] >> 6) << 4);
248 m[6] = (scales[10] >> 4) | ((scales[6] >> 6) << 4);
249 m[7] = (scales[11] >> 4) | ((scales[7] >> 6) << 4);
250}

252/* ============================================================================
253 * FP16 Conversion Utilities
254 *
255 * Three variants:
256 * _soft - Pure C bit manipulation (always available, portable)
257 * _simd - F16C hardware instruction (vcvtph2ps/vcvtps2ph, Ivy Bridge+)
258 * (default) - Auto-dispatches to best available at compile time
259 * ============================================================================ */
260
261/**
262 * @brief Convert FP16 (ck_half) to FP32 — software implementation
263 */
264static inline float ck_fp16_to_fp32_soft(ck_half h) {
265 uint32_t sign = (h & 0x8000) << 16;
266 uint32_t exp = (h >> 10) & 0x1F;
267 uint32_t mant = h & 0x3FF;
268
269 uint32_t result;
270
271 if (exp == 0) {
272 if (mant == 0) {
273 result = sign;
274 } else {
275 /* Denormalized - convert to normalized FP32 */
276 exp = 1;
277 while ((mant & 0x400) == 0) {
278 mant <<= 1;
279 exp--;
280 }
281 mant &= 0x3FF;
282 result = sign | ((exp + 127 - 15) << 23) | (mant << 13);
283 }
284 } else if (exp == 31) {
285 result = sign | 0x7F800000 | (mant << 13);
286 } else {
287 result = sign | ((exp + 127 - 15) << 23) | (mant << 13);
288 }
289
290 union { uint32_t u; float f; } u;
291 u.u = result;
292 return u.f;
293}
294
295/**
296 * @brief Convert FP32 to FP16 (ck_half) — software implementation
297 */
298static inline ck_half ck_fp32_to_fp16_soft(float f) {
299 union { uint32_t u; float f; } u;
300 u.f = f;
301
302 uint32_t sign = (u.u >> 16) & 0x8000;
303 int32_t exp = ((u.u >> 23) & 0xFF) - 127 + 15;
304 uint32_t mant = (u.u >> 13) & 0x3FF;
305
306 if (exp <= 0) {
307 if (exp < -10) {
308 return sign;
309 }
310 mant = (mant | 0x400) >> (1 - exp);
311 return sign | mant;
312 } else if (exp >= 31) {
313 return sign | 0x7C00;
314 }
315
316 return sign | (exp << 10) | mant;
317}

341/* --------------------------------------------------------------------------
342 * Default dispatch: selects hardware SIMD when available, else software
343 * -------------------------------------------------------------------------- */
344static inline float ck_fp16_to_fp32(ck_half h) {
345#if defined(__F16C__)
346 return ck_fp16_to_fp32_simd(h);
347#else
348 return ck_fp16_to_fp32_soft(h);
