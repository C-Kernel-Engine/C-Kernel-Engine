30 static float dot_q6_k_ref(const block_q6_K *w,
31  const float *x,
32  int K)
33 {
34  const int blocks_per_row = K / QK_K;
35  float sum = 0.0f;
36 
37  for (int b = 0; b < blocks_per_row; ++b) {
38  const block_q6_K *block = &w[b];
39  const float d = GGML_FP16_TO_FP32(block->d);
40 
41  const uint8_t *ql = block->ql;
42  const uint8_t *qh = block->qh;
43  const int8_t *sc = block->scales;
44  const float *xp = x + (size_t)b * QK_K;
45 
46  for (int n = 0; n < QK_K; n += 128) {
47  for (int l = 0; l < 32; ++l) {
48  const int is = l / 16;
49  const int8_t q1 = (int8_t)((ql[l + 0] & 0xF) | (((qh[l] >> 0) & 3) << 4)) - 32;
50  const int8_t q2 = (int8_t)((ql[l + 32] & 0xF) | (((qh[l] >> 2) & 3) << 4)) - 32;
51  const int8_t q3 = (int8_t)((ql[l + 0] >> 4) | (((qh[l] >> 4) & 3) << 4)) - 32;
52  const int8_t q4 = (int8_t)((ql[l + 32] >> 4) | (((qh[l] >> 6) & 3) << 4)) - 32;
53 
54  sum += (d * (float)sc[is + 0] * (float)q1) * xp[l + 0];
55  sum += (d * (float)sc[is + 2] * (float)q2) * xp[l + 32];
56  sum += (d * (float)sc[is + 4] * (float)q3) * xp[l + 64];
57  sum += (d * (float)sc[is + 6] * (float)q4) * xp[l + 96];
58  }
59  xp += 128;
60  ql += 64;
61  qh += 32;
62  sc += 8;
63  }
64  }
65 
66  return sum;
67 }

69 /* ============================================================================
70  * AVX Implementation (256-bit)
71  *
72  * Q6_K is complex: 6-bit weights packed as 4+2 bits, with 16 sub-block scales.
73  * Uses AVX for the final accumulation while keeping scalar dequantization
74  * for simplicity and correctness.
75  * ============================================================================ */

77 #if defined(__AVX__) && !defined(__AVX512F__)
78 static float dot_q6_k_avx(const block_q6_K *w,
79  const float *x,
80  int K)
81 {
82  const int blocks_per_row = K / QK_K;
83 
84  /* Use 4 accumulators for better ILP */
85  __m256 acc0 = _mm256_setzero_ps();
86  __m256 acc1 = _mm256_setzero_ps();
87  __m256 acc2 = _mm256_setzero_ps();
88  __m256 acc3 = _mm256_setzero_ps();
89 
90  for (int b = 0; b < blocks_per_row; ++b) {
91  const block_q6_K *block = &w[b];
92  const float d = GGML_FP16_TO_FP32(block->d);
93 
94  const uint8_t *ql = block->ql;
95  const uint8_t *qh = block->qh;
96  const int8_t *sc = block->scales;
97  const float *xp = x + (size_t)b * QK_K;
98 
99  /* Process 256 weights in 2 iterations of 128 */
100  for (int n = 0; n < QK_K; n += 128) {
101  /* Process 32 elements at a time in groups of 8 for AVX */
102  for (int l = 0; l < 32; l += 8) {
103  const int is = l / 16;
104 
105  /* Dequantize 8 weights for each of the 4 streams */
106  float dq0[8], dq1[8], dq2[8], dq3[8];
107 
108  for (int i = 0; i < 8; i++) {
109  int idx = l + i;
110  const int8_t q1 = (int8_t)((ql[idx + 0] & 0xF) | (((qh[idx] >> 0) & 3) << 4)) - 32;
111  const int8_t q2 = (int8_t)((ql[idx + 32] & 0xF) | (((qh[idx] >> 2) & 3) << 4)) - 32;
112  const int8_t q3 = (int8_t)((ql[idx + 0] >> 4) | (((qh[idx] >> 4) & 3) << 4)) - 32;
113  const int8_t q4 = (int8_t)((ql[idx + 32] >> 4) | (((qh[idx] >> 6) & 3) << 4)) - 32;
114 
115  const int is_i = (l + i) / 16;
116  dq0[i] = d * (float)sc[is_i + 0] * (float)q1;
117  dq1[i] = d * (float)sc[is_i + 2] * (float)q2;
118  dq2[i] = d * (float)sc[is_i + 4] * (float)q3;
119  dq3[i] = d * (float)sc[is_i + 6] * (float)q4;
120  }
121 
122  __m256 vw0 = _mm256_loadu_ps(dq0);
123  __m256 vw1 = _mm256_loadu_ps(dq1);
124  __m256 vw2 = _mm256_loadu_ps(dq2);
125  __m256 vw3 = _mm256_loadu_ps(dq3);
126 
127  __m256 vx0 = _mm256_loadu_ps(&xp[l + 0]);
128  __m256 vx1 = _mm256_loadu_ps(&xp[l + 32]);
129  __m256 vx2 = _mm256_loadu_ps(&xp[l + 64]);
130  __m256 vx3 = _mm256_loadu_ps(&xp[l + 96]);
131 
132  acc0 = _mm256_add_ps(acc0, _mm256_mul_ps(vw0, vx0));
133  acc1 = _mm256_add_ps(acc1, _mm256_mul_ps(vw1, vx1));
134  acc2 = _mm256_add_ps(acc2, _mm256_mul_ps(vw2, vx2));
135  acc3 = _mm256_add_ps(acc3, _mm256_mul_ps(vw3, vx3));
136  }
137  xp += 128;
138  ql += 64;
139  qh += 32;
140  sc += 8;
141  }
142  }
143 
144  /* Combine accumulators */
145  __m256 sum01 = _mm256_add_ps(acc0, acc1);
146  __m256 sum23 = _mm256_add_ps(acc2, acc3);
147  __m256 sum = _mm256_add_ps(sum01, sum23);
148 
149  /* Horizontal sum of 8 floats */
150  __m128 hi = _mm256_extractf128_ps(sum, 1);
151  __m128 lo = _mm256_castps256_ps128(sum);
152  __m128 sum128 = _mm_add_ps(hi, lo);
153  sum128 = _mm_hadd_ps(sum128, sum128);
154  sum128 = _mm_hadd_ps(sum128, sum128);
155 
156  return _mm_cvtss_f32(sum128);
157 }

160 void gemv_q6_k(float *y,
161  const void *W,
162  const float *x,
163  int M, int K)
164 {
165  if (!y || !W || !x) {
166  return;
167  }
168  if (M <= 0 || K <= 0) {
169  return;
170  }
171 
172  const block_q6_K *blocks = (const block_q6_K *)W;
173  const int blocks_per_row = K / QK_K;
174 
175  for (int row = 0; row < M; ++row) {
176  const block_q6_K *w_row = blocks + (size_t)row * (size_t)blocks_per_row;
177 #if defined(__AVX__) && !defined(__AVX512F__)
178  y[row] = dot_q6_k_avx(w_row, x, K);
179 #else
180  y[row] = dot_q6_k_ref(w_row, x, K);
181 #endif
182  }
183 }
