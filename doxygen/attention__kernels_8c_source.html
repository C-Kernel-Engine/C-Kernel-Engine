76static CK_NOINLINE CK_OPTNONE float ck_vec_dot_f32_strict(const float *x,
77 const float *y,
78 int n)
79{
80 float sumf = 0.0f;
81 for (int i = 0; i < n; ++i) {
82 volatile float prod = x[i] * y[i];
83 volatile float next = sumf + prod;
84 sumf = next;
85 }
86 return sumf;
87}

186static int ck_attention_vec_dump_should_emit(int layer_id, int head_id, int query_id)
187{
188 if (!ck_attention_vec_dump_enabled()) {
189 return 0;
190 }
191 int want_layer = -1;
192 int want_head = -1;
193 int want_query = -1;
194 if (!ck_attention_vec_dump_parse_env_int("CK_STRICT_ATTN_DUMP_LAYER", &want_layer)) {
195 return 0;
196 }
197 const int have_head = ck_attention_vec_dump_parse_env_int("CK_STRICT_ATTN_DUMP_HEAD", &want_head);
198 const int have_query = ck_attention_vec_dump_parse_env_int("CK_STRICT_ATTN_DUMP_QUERY", &want_query);
199 const int trace_target = layer_id == want_layer &&
200 (!have_head || head_id == want_head) &&
201 (!have_query || query_id == want_query);
202 if (layer_id != want_layer) {
203 return 0;
204 }
205 if (have_head && head_id != want_head) {
206 return 0;
207 }
208 if (have_query && query_id != want_query) {
209 return 0;
210 }
211 if (trace_target) {
212 ck_attention_trace_query("vec_dump_should_emit", layer_id, head_id, query_id, 1);
213 }
214 return 1;
215}

224static void ck_attention_vec_dump_tensor(const char *name,
225 int layer_id,
226 int query_id,
227 const float *data,
228 size_t elem_count)
229{
230 const char *dir = getenv("CK_PARITY_DIR");
231 if (!dir || !dir[0] || !name || !name[0] || !data || elem_count == 0) {
232 return;
233 }
234
235 char path[4096];
236 snprintf(path, sizeof(path), "%s/%s", dir, "strict_internal.bin");
237 FILE *f = fopen(path, "ab");
238 if (!f) {
239 return;
240 }
241
242 ck_attention_vec_dump_header_t h;
243 memset(&h, 0, sizeof(h));
244 memcpy(h.magic, ck_attention_vec_dump_magic, sizeof(ck_attention_vec_dump_magic));
245 h.version = ck_attention_vec_dump_version;
246 h.layer_id = layer_id;
247 strncpy(h.op_name, name, sizeof(h.op_name) - 1);
248 h.dtype = 0u;
249 h.rank = 1u;
250 h.shape[0] = (int64_t) elem_count;
251 h.elem_count = (uint32_t) elem_count;
252 h.token_id = query_id;
253
254 fwrite(&h, sizeof(h), 1, f);
255 fwrite(data, sizeof(float), elem_count, f);
256 fclose(f);
257}

372{
373 static int tried = 0;
374 static void *handle = NULL;
375 if (!tried) {
376 tried = 1;
377 const char *env_path = getenv("CK_GGML_CPU_SO");
378 const char *candidates[] = {
379 env_path && env_path[0] ? env_path : NULL,
380 "libggml-cpu.so",
381 "libggml-cpu.so.0",
382 "./llama.cpp/build/bin/libggml-cpu.so",
383 "./llama.cpp/build/bin/libggml-cpu.so.0",
384 "llama.cpp/build/bin/libggml-cpu.so",
385 "llama.cpp/build/bin/libggml-cpu.so.0",
386 NULL,
387 };
388 for (int i = 0; candidates[i] != NULL; ++i) {
389 handle = dlopen(candidates[i], RTLD_NOW | RTLD_LOCAL);
390 if (handle) {
391 break;
392 }
393 }
394 }
395 return handle;
396}

687static CK_NOINLINE CK_OPTNONE float ck_ggml_vec_dot_f32_contig(const float *x,
688 const float *y,
689 int n)
690{
691 // Keep this a literal port of ggml_vec_dot_f32 so strict parity can match
692 // llama.cpp's CPU attention path instead of merely approximating it.
693#if defined(__AVX__)
694 float sumf = 0.0f;
695 const int np = (n & ~31);
696 __m256 sum[4] = {
697 _mm256_setzero_ps(),
698 _mm256_setzero_ps(),
699 _mm256_setzero_ps(),
700 _mm256_setzero_ps(),
701 };
702
703 for (int i = 0; i < np; i += 32) {
704 for (int j = 0; j < 4; ++j) {
705 const __m256 ax = _mm256_loadu_ps(x + i + j * 8);
706 const __m256 ay = _mm256_loadu_ps(y + i + j * 8);
707#if defined(__FMA__)
708 sum[j] = _mm256_fmadd_ps(ax, ay, sum[j]);
709#else
710 sum[j] = _mm256_add_ps(_mm256_mul_ps(ax, ay), sum[j]);
711#endif
712 }
713 }
714
715 sum[0] = _mm256_add_ps(sum[0], sum[2]);
716 sum[1] = _mm256_add_ps(sum[1], sum[3]);
717 sum[0] = _mm256_add_ps(sum[0], sum[1]);
718 const __m128 t0 = _mm_add_ps(_mm256_castps256_ps128(sum[0]),
719 _mm256_extractf128_ps(sum[0], 1));
720 const __m128 t1 = _mm_hadd_ps(t0, t0);
721 sumf = _mm_cvtss_f32(_mm_hadd_ps(t1, t1));
722
723 for (int i = np; i < n; ++i) {
724 sumf += x[i] * y[i];
725 }
726 return sumf;
727#elif defined(__SSE2__)
728 float sumf = 0.0f;
729 const int np = (n & ~15);
730 __m128 sum[4] = {
731 _mm_setzero_ps(),
732 _mm_setzero_ps(),
733 _mm_setzero_ps(),
734 _mm_setzero_ps(),
735 };
736
737 for (int i = 0; i < np; i += 16) {
738 for (int j = 0; j < 4; ++j) {
739 const __m128 ax = _mm_loadu_ps(x + i + j * 4);
740 const __m128 ay = _mm_loadu_ps(y + i + j * 4);
741#if defined(__FMA__)
742 sum[j] = _mm_fmadd_ps(ax, ay, sum[j]);
743#else
744 sum[j] = _mm_add_ps(_mm_mul_ps(ax, ay), sum[j]);
745#endif
746 }
747 }

749 sum[0] = _mm_add_ps(sum[0], sum[2]);
750 sum[1] = _mm_add_ps(sum[1], sum[3]);
751 sum[0] = _mm_add_ps(sum[0], sum[1]);
752#if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512F__)
753 sum[0] = _mm_add_ps(sum[0], _mm_movehl_ps(sum[0], sum[0]));
754 sum[0] = _mm_add_ss(sum[0], _mm_movehdup_ps(sum[0]));
755#else
756 __m128 tmp = _mm_shuffle_ps(sum[0], sum[0], _MM_SHUFFLE(2, 3, 0, 1));
757 sum[0] = _mm_add_ps(sum[0], tmp);
758 tmp = _mm_movehl_ps(tmp, sum[0]);
759 sum[0] = _mm_add_ss(sum[0], tmp);
760#endif
761 sumf = _mm_cvtss_f32(sum[0]);
762
763 for (int i = np; i < n; ++i) {
764 sumf += x[i] * y[i];
765 }
766 return sumf;
767#else
768 double sumf = 0.0;
769 for (int i = 0; i < n; ++i) {
770 sumf += (double) (x[i] * y[i]);
771 }
772 return (float) sumf;
773#endif
774}

776static CK_NOINLINE CK_OPTNONE float ck_attention_strict_scale_f32(int head_dim)
777{
778 // Keep strict parity on the precise libm sqrtf path. icx -O3 on AVX2 was
779 // lowering 1/sqrtf(d) to a slightly smaller effective scale, which is
780 // enough to move layer-0 vision softmax by ~2e-7 and snowball later.
781 volatile float hd = (float) head_dim;
782 float (*sqrtf_fn)(float) = sqrtf;
783 volatile float root = sqrtf_fn(hd);
784 volatile float one = 1.0f;
785 volatile float scale = one / root;
786 return scale;
787}

789static CK_NOINLINE CK_OPTNONE double ck_ggml_vec_soft_max_row(int n,
790 float *y,
791 const float *x,
792 float max)
793{
794 int i = 0;
795 double sum = 0.0;
796
797#if defined(__AVX2__) && defined(__FMA__)
798 for (; i + 7 < n; i += 8) {
799 const __m256 val = ck_ggml_v_expf256(
800 _mm256_sub_ps(_mm256_loadu_ps(x + i), _mm256_set1_ps(max)));
801 _mm256_storeu_ps(y + i, val);
802 __m128 val2 = _mm_add_ps(_mm256_extractf128_ps(val, 1),
803 _mm256_castps256_ps128(val));
804 val2 = _mm_add_ps(val2, _mm_movehl_ps(val2, val2));
805 val2 = _mm_add_ss(val2, _mm_movehdup_ps(val2));
806 sum += (double) _mm_cvtss_f32(val2);
807 }
808#elif defined(__SSE2__)
809 for (; i + 3 < n; i += 4) {
810 const __m128 val = ck_ggml_v_expf128(
811 _mm_sub_ps(_mm_loadu_ps(x + i), _mm_set1_ps(max)));
812 _mm_storeu_ps(y + i, val);
813#if defined(__AVX__) || defined(__AVX2__) || defined(__AVX512F__)
814 __m128 acc = _mm_add_ps(val, _mm_movehl_ps(val, val));
815 acc = _mm_add_ss(acc, _mm_movehdup_ps(acc));
816#else
817 __m128 tmp = _mm_shuffle_ps(val, val, _MM_SHUFFLE(2, 3, 0, 1));
818 __m128 acc = _mm_add_ps(val, tmp);
819 tmp = _mm_movehl_ps(tmp, acc);
820 acc = _mm_add_ss(acc, tmp);
821#endif
822 sum += (double) _mm_cvtss_f32(acc);
823 }
824#endif
825
826 for (; i < n; ++i) {
827 const float val = expf(x[i] - max);
828 y[i] = val;
829 sum += (double) val;
830 }
831
832 return sum;
833}

847/**
848 * Causal attention forward (score-matrix version)
849 * @test test_attention.py::TestAttentionForward::test_causal_forward
850 * @test test_attention.py::TestAttentionForward::test_gqa_broadcast
851 * @test test_attention.py::TestAttentionForward::test_exact_vs_fast
852 * @test test_parity.py::test_attention_parity
853 *
854 * Computes softmax(Q @ K^T / sqrt(d)) @ V with causal masking.
855 * Uses O(N^2) memory for scores matrix.
856 *
857 * After changes: make test && make llamacpp-parity-full
858 */
859void attention_forward_causal_head_major(const float *q,
860 const float *k,
861 const float *v,
862 float *scores,
863 float *output,
864 int num_heads,
865 int num_tokens,
866 int head_dim,
867 int aligned_head_dim,
868 int aligned_context_window)
869{
870 const float scale = 1.0f / sqrtf((float)head_dim);
871
872 // Phase 1: compute scaled dot-product scores Q·K^T / sqrt(d_k),
873 // lower triangle only (j <= i).
874 for (int h = 0; h < num_heads; ++h) {
875 for (int i = 0; i < num_tokens; ++i) {
876 for (int j = 0; j <= i; ++j) {
877 float dot = 0.0f;
878 size_t base_q = qkv_index(h, i, 0, num_tokens, aligned_head_dim);
879 size_t base_k = qkv_index(h, j, 0, num_tokens, aligned_head_dim);
880
881 for (int d = 0; d < head_dim; ++d) {
882 dot += q[base_q + d] * k[base_k + d];
883 }
884
885 scores[score_index(h, i, j, aligned_context_window)] = dot * scale;
886 }
887
888 // Ensure upper triangle is zeroed so there are no stale values
889 // before the softmax kernel runs.
890 for (int j = i + 1; j < num_tokens; ++j) {
891 scores[score_index(h, i, j, aligned_context_window)] = 0.0f;
892 }
893 }
894 }
895
896 // Phase 2: apply causal row-wise softmax in-place over j <= i.
897 causal_softmax_head_major(scores,
898 num_heads,
899 num_tokens,
900 aligned_context_window);
901
902 // Phase 3: attention weights · V.
903 for (int h = 0; h < num_heads; ++h) {
904 for (int i = 0; i < num_tokens; ++i) {
905 size_t out_base = qkv_index(h, i, 0, num_tokens, aligned_head_dim);
906
907 // Zero the full aligned head slice so padded dims stay clean.
908 for (int d = 0; d < aligned_head_dim; ++d) {
909 output[out_base + d] = 0.0f;
910 }
911
912 // Weighted sum over causal positions.
913 for (int j = 0; j <= i; ++j) {
914 float w = scores[score_index(h, i, j, aligned_context_window)];
915 size_t v_base = qkv_index(h, j, 0, num_tokens, aligned_head_dim);
916
917 for (int d = 0; d < head_dim; ++d) {
918 output[out_base + d] += w * v[v_base + d];
919 }
920 }
921 }
922 }
923}

1228 const float *v_vec = v_head + (size_t)j * (size_t)aligned_head_dim;

1281 }

1289 _mm512_storeu_ps(&out_vec[d], _mm512_mul_ps(out_v, inv_s_vec));

1292 out_vec[d] *= inv_s;
1293 }
1294
1295 // Zero padding
1296 for (d = head_dim; d < aligned_head_dim; ++d) {
1297 out_vec[d] = 0.0f;
1298 }
1299}

3490 int num_kv_heads,
3491 int num_tokens,
3492 int head_dim,
3493 int aligned_head_dim,
3494 int aligned_context_window)

3498 int H = num_heads;
3499 int H_kv = num_kv_heads;
3500 int hd = head_dim;
3501 int ad = aligned_head_dim;

3505 const size_t kv_elems = (size_t)H_kv * (size_t)T * (size_t)ad;

3508 d_q[idx] = 0.0f;

3510 for (size_t idx = 0; idx < kv_elems; ++idx) {

3516 for (int h = 0; h < H; ++h) {
3517 // Which KV head does this query head use?

3519
3520 // ----------------------------------------------------------------
3521 // Step 1: d_weights = d_output @ V^T and d_v += weights^T @ d_output
3522 // ----------------------------------------------------------------
3523 // For each query position i, compute d_weights[i, j] for j <= i
3524 // and accumulate d_v[j] contributions
3525
3526 for (int i = 0; i < T; ++i) {
3527 size_t d_out_base = qkv_index(h, i, 0, T, ad);
3528
3529 for (int j = 0; j <= i; ++j) {
3530 size_t v_base = qkv_index(kv_h, j, 0, T, ad);

3535 float dot = 0.0f;
3536 for (int dd = 0; dd < hd; ++dd) {
3537 dot += d_output[d_out_base + dd] * v[v_base + dd];
3538 }
3539 d_scores[w_idx] = dot;
3540
3541 // d_v[kv_h, j, :] += weights[h, i, j] * d_output[h, i, :]
3542 for (int dd = 0; dd < hd; ++dd) {
3543 d_v[v_base + dd] += w * d_output[d_out_base + dd];
3544 }
3545 }
3546
3547 // Zero out upper triangle of d_scores
3548 for (int j = i + 1; j < T; ++j) {
3549 d_scores[score_index(h, i, j, aw)] = 0.0f;
3550 }
3551 /* Scores scratch uses aligned_context_window, zero the padded columns. */
3552 for (int j = T; j < aw; ++j) {
3553 d_scores[score_index(h, i, j, aw)] = 0.0f;
3554 }
3555 }
3556
3557 // ----------------------------------------------------------------
3558 // Step 2: Backward through softmax (in-place on d_scores for this head)
3559 // ----------------------------------------------------------------
3560 // d_scores = softmax_backward(d_scores, attn_weights)
3561 // Formula: d_score[i,j] = w[i,j] * (d_w[i,j] - sum_k(w[i,k] * d_w[i,k]))
3562
3563 for (int i = 0; i < T; ++i) {
3564 int base = h * aw * aw + i * aw;
3565
3566 // Compute dot product: sum_j w[i,j] * d_w[i,j]
3567 float dot_product = 0.0f;
3568 for (int j = 0; j <= i; ++j) {
3569 float wt = attn_weights[base + j];
3570 float dw = d_scores[base + j];
3571 dot_product += wt * dw;
3572 }
3573
3574 // Apply softmax backward formula
3575 for (int j = 0; j <= i; ++j) {
3576 float wt = attn_weights[base + j];
3577 float dw = d_scores[base + j];
3578 d_scores[base + j] = wt * (dw - dot_product);
3579 }
3580 }
3581
3582 // ----------------------------------------------------------------
3583 // Step 3: d_q = d_scores @ K * scale
3584 // d_k += d_scores^T @ Q * scale
3585 // ----------------------------------------------------------------
3586
3587 for (int i = 0; i < T; ++i) {
3588 size_t d_q_base = qkv_index(h, i, 0, T, ad);
3589 size_t q_base = qkv_index(h, i, 0, T, ad);
3590
3591 // d_q[h, i, :] = sum_j d_scores[h, i, j] * k[kv_h, j, :] * scale
3592 // d_k[kv_h, j, :] += d_scores[h, i, j] * q[h, i, :] * scale
3593 for (int j = 0; j <= i; ++j) {
3594 size_t k_base = qkv_index(kv_h, j, 0, T, ad);
3595 size_t d_k_base = qkv_index(kv_h, j, 0, T, ad);
3596 float ds = d_scores[score_index(h, i, j, aw)] * scale;
3597
3598 for (int dd = 0; dd < hd; ++dd) {
3599 d_q[d_q_base + dd] += ds * k[k_base + dd];
3600 d_k[d_k_base + dd] += ds * q[q_base + dd];
3601 }
3602 }
3603 }
3604 }
