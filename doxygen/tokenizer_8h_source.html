1/*
2 * C-Kernel-Engine Tokenizer
3 *
4 * High-performance tokenizer supporting:
5 * - BPE (Byte-Pair Encoding): GPT-2, LLaMA, Qwen
6 * - WordPiece: BERT, RoBERTa
7 * - SentencePiece (unigram): LLaMA, T5
8 *
9 * Features:
10 * - MurmurHash3 hashing
11 * - AVX-512 optimized string comparison
12 * - Greedy longest-match encoding
13 * - Full UTF-8 support
14 * - GGUF vocab loading
15 *
16 * By Anthony Shivakumar
17 */
18

52/* Tokenizer model type */
53typedef enum {
54 CK_TOKENIZER_BPE = 0, /* Byte-Pair Encoding (GPT-2, LLaMA, Qwen) */
55 CK_TOKENIZER_WORDPIECE = 1, /* WordPiece (BERT, RoBERTa) */
56 CK_TOKENIZER_SPM = 2 /* SentencePiece (unigram) */
57} CKTokenizerType;
58
59/* Space prefix style for BPE tokenizers */
60typedef enum {
61 CK_SPACE_PREFIX_AUTO = 0, /* Auto-detect from vocabulary */
62 CK_SPACE_PREFIX_GPT2 = 1, /* GPT-2 style: Ġ (U+0120, bytes 0xC4 0xA0) */
63 CK_SPACE_PREFIX_SPM = 2 /* SentencePiece style: ▁ (U+2581, bytes 0xE2 0x96 0x81) */
64} CKSpacePrefixStyle;
65
66/* Tokenizer configuration */
67typedef struct {
68 CKTokenizerType type; /* Tokenization algorithm */
69 bool add_bos; /* Add beginning-of-sequence token */
70 bool add_eos; /* Add end-of-sequence token */
71 bool lowercase; /* Convert text to lowercase before tokenizing */
72 bool treat_whitespace_as_suffix; /* For SentencePiece */
73 float unk_score; /* Unknown token score (for SPM) */
74 bool use_trie; /* Use trie for lookups (faster), false = use hash table */
75 CKSpacePrefixStyle space_prefix_style; /* Space prefix style (GPT-2 Ġ vs SentencePiece ▁) */
76 bool space_prefix_detected; /* True if auto-detection has run */
77} CKTokenizerConfig;
78
79/* Vocabulary entry */
80typedef struct {
81 int32_t id; /* Token ID */
82 float score; /* Score (for SPM) */
83 bool is_special; /* Is special token */
84} CKTokenizerToken;

86/* Main tokenizer structure */
87typedef struct {
88 /* Configuration */
89 CKTokenizerConfig config;
90
91 /* Vocabulary: token string -> token info */
92 CKTokenizerHashTable *vocab;
93
94 /* Trie for fast longest-match lookups (O(k) instead of O(n*k)) */
95 CKTrie *vocab_trie;
96
97 /* Reverse vocabulary: ID -> token string */
98 char **id_to_token;
99 size_t vocab_size;
100 size_t vocab_capacity;
101
102 /* Special token IDs */
103 int32_t unk_id;
104 int32_t bos_id;
105 int32_t eos_id;
106 int32_t pad_id;
107 int32_t mask_id;
108
109 /* Memory pool for allocations */
110 CKTokenizerMemPool pool;
111
112 /* For BPE: merge rules */
113 int32_t *merge_pairs; /* left_id * vocab_size + right_id -> merge priority */
114 size_t merge_pairs_size;
115 int32_t *merge_result; /* merge priority -> merged token ID */
116 size_t merge_result_size;
117 int32_t num_merges;
118
119 /* Cache for encoding */
120 char *encode_buffer;
121 size_t encode_buffer_size;
122} CKTokenizer;
