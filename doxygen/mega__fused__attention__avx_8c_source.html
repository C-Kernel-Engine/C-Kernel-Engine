2 * @file mega_fused_attention_avx.c
3 * @brief Mega-Fused Attention for AVX (256-bit) and AVX-512 (512-bit)
4 *
5 * Holy grail fusion: RMSNorm → QKV → RoPE → Flash Attention → OutProj + Residual
6 *
7 * AVX approach: Keep intermediates in L1 cache (not registers)
8 * AVX-512 approach: Keep intermediates in registers
9 *
10 * Both achieve the same goal: Eliminate DRAM traffic for intermediates.

81/*============================================================================
82 * Phase 1: RMSNorm + QKV Fusion (AVX version)
83 *
84 * Keep ln1_row in stack buffer, not DRAM.
85 * Q/K/V go directly to next operation.
86 *============================================================================*/
87
88/**
89 * @brief Fused RMSNorm + QKV for decode (single token)
90 *
91 * AVX-optimized version. Intermediates stay in L1/L2.
92 *
93 * @param q_out Output Q [num_heads * head_dim] - to RoPE
94 * @param k_out Output K [num_kv_heads * head_dim] - to RoPE
95 * @param v_out Output V [num_kv_heads * head_dim] - to KV cache
96 * @param input Input token [hidden] - from DRAM
97 * @param gamma RMSNorm gamma [hidden]
98 * @param W_qkv QKV weights [3*hidden, hidden]
99 * @param b_qkv QKV bias [3*hidden] or NULL
100 * @param hidden Model hidden dimension
101 * @param num_heads Number of attention heads
102 * @param num_kv_heads Number of KV heads
103 * @param head_dim Head dimension
104 * @param eps RMSNorm epsilon
105 */
106void mega_fuse_rmsnorm_qkv_avx(
107 float *q_out,
108 float *k_out,
109 float *v_out,
110 const float *input,
111 const float *gamma,
112 const float *W_qkv,
113 const float *b_qkv,
114 int hidden,
115 int num_heads,
116 int num_kv_heads,
117 int head_dim,
118 float eps)
119{
120 const int H = num_heads;
121 const int H_kv = num_kv_heads;
122 const int hd = head_dim;
123 const int ad = (head_dim + MEGA_VLEN - 1) / MEGA_VLEN * MEGA_VLEN; /* aligned */
124
125 /* Stack buffer for RMSNorm output - stays in L1! */
126 float ln1_row[hidden];
127 float rstd;
128 float sum_sq = 0.0f;
129
130 /* RMSNorm: compute in registers, store to L1 stack buffer */
131#if defined(__AVX512F__)
132 __m512 sum = _mm512_setzero_ps();
133 for (int i = 0; i < hidden; i += 16) {
134 __m512 xv = _mm512_loadu_ps(input + i);
135 sum = _mm512_add_ps(sum, _mm512_mul_ps(xv, xv));
136 }
137 float rowsum = _mm512_reduce_add_ps(sum);
138 rstd = 1.0f / sqrtf(rowsum / hidden + eps);
139#elif defined(__AVX__)
140 __m256 sum = _mm256_setzero_ps();
141 for (int i = 0; i < hidden; i += 8) {
142 __m256 xv = _mm256_loadu_ps(input + i);
143 sum = _mm256_add_ps(sum, _mm256_mul_ps(xv, xv));
144 }
145 /* Horizontal sum of 256-bit vector */
146 __m128 sum128 = _mm256_extractf128_ps(sum, 0);
147 __m128 hi = _mm256_extractf128_ps(sum, 1);
148 sum128 = _mm_add_ps(sum128, hi);
149 float rowsum = _mm_cvtss_f32(_mm_hadd_ps(sum128, sum128));
150 rstd = 1.0f / sqrtf(rowsum / hidden + eps);
151#endif
152
153 /* Normalize and project - no DRAM write of ln1_row! */
154 size_t q_elems = (size_t)H * ad;
155 size_t kv_elems = (size_t)H_kv * ad;
156
157 /* Q projection: ln1_row is still hot in L1 */
158 for (int j = 0; j < q_elems; j += 4) {
159 float dot = 0.0f;
160 for (int i = 0; i < hidden; i++) {
161 dot += ln1_row[i] * W_qkv[j * (size_t)hidden + i];
162 }
163 q_out[j] = dot + (b_qkv ? b_qkv[j] : 0.0f);
164 }
165
166 /* K projection - same L1 buffer */
167 const float *W_k = W_qkv + (size_t)H * ad * hidden;
168 for (int j = 0; j < kv_elems; j += 4) {
169 float dot = 0.0f;
170 for (int i = 0; i < hidden; i++) {
171 dot += ln1_row[i] * W_k[j * (size_t)hidden + i];
172 }
173 k_out[j] = dot + (b_qkv ? b_qkv[H * ad + j] : 0.0f);
174 }
175
176 /* V projection - same L1 buffer */
177 const float *W_v = W_qkv + (size_t)(H + H_kv) * ad * hidden;
178 for (int j = 0; j < kv_elems; j += 4) {
179 float dot = 0.0f;
180 for (int i = 0; i < hidden; i++) {
181 dot += ln1_row[i] * W_v[j * (size_t)hidden + i];
182 }
183 v_out[j] = dot + (b_qkv ? b_qkv[(H + H_kv) * ad + j] : 0.0f);
184 }
185
186 /* q_out, k_out, v_out are in L1 - ready for RoPE! */
187}

189/*============================================================================
190 * Phase 2: RoPE In-Place (Q/K still hot in L1/L2)
191 *============================================================================*/
192
193/**
194 * @brief Apply RoPE to Q and K (in-place, from L1)
195 *
196 * Q and K are already in L1 from QKV projection.
197 * Just apply rotation in-place.
198 */
199void mega_fuse_rope_inplace_avx(
200 float *q, /* [num_heads * head_dim] - in L1 */
201 float *k, /* [num_kv_heads * head_dim] - in L1 */
202 const float *rope_cos,
203 const float *rope_sin,
204 int pos,
205 int num_heads,
206 int num_kv_heads,
207 int head_dim)
208{
209 int half = head_dim / 2;
210
211 for (int h = 0; h < num_heads; h++) {
212 float *q_h = q + h * head_dim;
213 float *k_h = k + (h % num_kv_heads) * head_dim;
214
215 const float *cos_ptr = rope_cos + pos * half;
216 const float *sin_ptr = rope_sin + pos * half;
217
218 for (int i = 0; i < half; i += MEGA_VLEN) {
219#if defined(__AVX512F__)
220 __m512 q0 = _mm512_loadu_ps(q_h + i);
221 __m512 q1 = _mm512_loadu_ps(q_h + i + half);
222 __m512 k0 = _mm512_loadu_ps(k_h + i);
223 __m512 k1 = _mm512_loadu_ps(k_h + i + half);
224 __m512 cos = _mm512_loadu_ps(cos_ptr + i);
225 __m512 sin = _mm512_loadu_ps(sin_ptr + i);
226
227 __m512 q_rot0 = _mm512_fmsub_ps(q0, cos, _mm512_mul_ps(q1, sin));
228 __m512 q_rot1 = _mm512_fmadd_ps(q0, sin, _mm512_mul_ps(q1, cos));
229 __m512 k_rot0 = _mm512_fmsub_ps(k0, cos, _mm512_mul_ps(k1, sin));
230 __m512 k_rot1 = _mm512_fmadd_ps(k0, sin, _mm512_mul_ps(k1, cos));
231
232 _mm512_storeu_ps(q_h + i, q_rot0);
233 _mm512_storeu_ps(q_h + i + half, q_rot1);
234 _mm512_storeu_ps(k_h + i, k_rot0);
235 _mm512_storeu_ps(k_h + i + half, k_rot1);
236#elif defined(__AVX__)
237 __m256 q0 = _mm256_loadu_ps(q_h + i);
238 __m256 q1 = _mm256_loadu_ps(q_h + i + half);
239 __m256 k0 = _mm256_loadu_ps(k_h + i);
240 __m256 k1 = _mm256_loadu_ps(k_h + i + half);
241 __m256 cos = _mm256_loadu_ps(cos_ptr + i);
242 __m256 sin = _mm256_loadu_ps(sin_ptr + i);
243
244 __m256 q_rot0 = _mm256_fmsub_ps(q0, cos, _mm256_mul_ps(q1, sin));
245 __m256 q_rot1 = _mm256_fmadd_ps(q0, sin, _mm256_mul_ps(q1, cos));
246 __m256 k_rot0 = _mm256_fmsub_ps(k0, cos, _mm256_mul_ps(k1, sin));
247 __m256 k_rot1 = _mm256_fmadd_ps(k0, sin, _mm256_mul_ps(k1, cos));
248
249 _mm256_storeu_ps(q_h + i, q_rot0);
250 _mm256_storeu_ps(q_h + i + half, q_rot1);
251 _mm256_storeu_ps(k_h + i, k_rot0);
252 _mm256_storeu_ps(k_h + i + half, k_rot1);
253#endif
254 }
255 }
256}

258/*============================================================================
259 * Phase 3: Flash Attention with Online Softmax
260 *
261 * O, m, l stay in registers across all KV tiles.
262 * K/V stream from KV cache (in L2).
263 *============================================================================*/
264
265/**
266 * @brief Flash attention with online softmax (AVX version)
267 *
268 * Key insight: O, m, l stay in registers throughout!
269 * K/V tiles stream from L2 cache.
270 *
271 * @param o_out Output [num_heads * head_dim] - in registers/L1
272 * @param q Q tensor [num_heads * head_dim] - from L1
273 * @param kv_cache_k KV cache K [seq, num_kv_heads * head_dim] - in L2
274 * @param kv_cache_v KV cache V [seq, num_kv_heads * head_dim] - in L2
275 * @param num_heads Number of heads
276 * @param num_kv_heads Number of KV heads
277 * @param seq_len Current sequence length
278 * @param head_dim Head dimension
279 */
280void mega_fuse_flash_attention_avx(
281 float *o_out,
282 const float *q,
283 const float *kv_cache_k,
284 const float *kv_cache_v,
285 int num_heads,
286 int num_kv_heads,
287 int seq_len,
288 int head_dim)
289{
290 const int hd = head_dim;
291 const float scale = 1.0f / sqrtf((float)hd);
292
293 for (int h = 0; h < num_heads; h++) {
294 const float *q_h = q + h * hd;
295 const int kv_idx = h % num_kv_heads;
296 const float *k_cache = kv_cache_k + kv_idx * hd;
297 const float *v_cache = kv_cache_v + kv_idx * hd;
298
299 /* O, m, l in registers for this head */
300 float o_h[hd]; /* in L1 */
301 float m = -INFINITY; /* running max */
302 float l = 0.0f; /* running sum */
303
304 /* Initialize O to zeros */
305 memset(o_h, 0, hd * sizeof(float));
306
307 /* Iterate over KV cache tiles */
308 for (int t = 0; t < seq_len; t += MEGA_KV_TILE) {
309 int tile_end = t + MEGA_KV_TILE;
310 if (tile_end > seq_len) tile_end = seq_len;
311 int tile_size = tile_end - t;
312
313 /* Load K tile from L2 cache */
314 float k_tile[MEGA_KV_TILE * hd];
315 for (int i = 0; i < tile_size; i++) {
316 memcpy(k_tile + i * hd, k_cache + (t + i) * hd, hd * sizeof(float));
317 }
318
319 /* S_ij = Q @ K_tile.T / sqrt(d) - in registers */
320 float s_row[MEGA_KV_TILE];
321 for (int j = 0; j < tile_size; j++) {
322 s_row[j] = 0.0f;
323 for (int i = 0; i < hd; i++) {
324 s_row[j] += q_h[i] * k_tile[j * hd + i];
325 }
326 s_row[j] *= scale;
327 }
328
329 /* Online softmax update */
330 float m_new = m;
331 for (int j = 0; j < tile_size; j++) {
332 if (s_row[j] > m_new) m_new = s_row[j];
333 }
334
335 float l_new = 0.0f;
336 for (int j = 0; j < tile_size; j++) {
337 float p = expf(s_row[j] - m_new);
338 s_row[j] = p;
339 l_new += p;
340 }
341
342 /* Scale O by exp(m - m_new) and add P @ V_tile */
343 float exp_m_diff = expf(m - m_new);
344 for (int i = 0; i < hd; i++) {
345 o_h[i] *= exp_m_diff;
346 }
347
348 /* Load V tile and accumulate */
349 for (int j = 0; j < tile_size; j++) {
350 float p = s_row[j];
351 for (int i = 0; i < hd; i++) {
352 o_h[i] += p * v_cache[(t + j) * hd + i];
353 }
354 }
355
356 l = l_new * exp_m_diff + l;
357 m = m_new;
358 }
359
360 /* Normalize by l */
361 for (int i = 0; i < hd; i++) {
362 o_h[i] /= l;
363 }
364
365 /* Store O - still in L1, goes to output projection */
366 memcpy(o_out + h * hd, o_h, hd * sizeof(float));
367 }
368}

370/*============================================================================
371 * Phase 4: Full Mega-Fused Attention Decode
372 *
373 * RMSNorm → QKV → RoPE → Flash Attn → OutProj + Residual
374 * All intermediates in L1/L2, single DRAM round-trip.
375 *============================================================================*/
376
377/**
378 * @brief Full mega-fused attention for decode
379 *
380 * THE HOLY GRAIL: All operations fused, no intermediates to DRAM.
381 *
382 * Reads: input[hidden], W_qkv, W_o, KV cache, RoPE tables
383 * Writes: output[hidden]
384 *
385 * Total DRAM traffic: 4KB input + 4KB output = 8KB
386 * (vs ~800KB with unfused attention)
387 */
388void mega_fused_attention_decode_avx(
389 float *output, /* [hidden] - single DRAM write */
390 const float *input, /* [hidden] - single DRAM read */
391 const float *residual, /* [hidden] - for residual add */
392 const float *W_qkv, /* QKV weights */
393 const float *b_qkv, /* QKV bias or NULL */
394 const float *W_o, /* Output projection */
395 const float *b_o, /* Output bias or NULL */
396 float *kv_cache_k, /* KV cache K - in L2 */
397 float *kv_cache_v, /* KV cache V - in L2 */
398 const float *rope_cos,
399 const float *rope_sin,
400 int pos,
401 int hidden,
402 int num_heads,
403 int num_kv_heads,
404 int head_dim,
405 int max_seq,
406 float eps)
407{
408 /* Stack-allocated buffers - stay in L1 (no malloc!) */
409 /* Max: 64 heads * 128 dim = 8192 floats = 32KB per buffer */
410 size_t q_elems = (size_t)num_heads * head_dim;
411 size_t kv_elems = (size_t)num_kv_heads * head_dim;
412
413 if (q_elems > 8192 || kv_elems > 8192) return;
414
415 float q[8192];
416 float k[8192];
417 float v[8192];
418 float o[8192];
419
420 /* Phase 1: RMSNorm + QKV (ln1_row in L1 stack) */
421 mega_fuse_rmsnorm_qkv_avx(q, k, v, input, NULL, W_qkv, b_qkv,
422 hidden, num_heads, num_kv_heads, head_dim, eps);
423
424 /* Phase 2: RoPE in-place (q, k still hot in L1) */
425 mega_fuse_rope_inplace_avx(q, k, rope_cos, rope_sin, pos,
426 num_heads, num_kv_heads, head_dim);
427
428 /* Write K/V to KV cache (L2) */
429 for (int h = 0; h < num_kv_heads; h++) {
430 memcpy(kv_cache_k + pos * kv_elems + h * head_dim,
431 k + h * head_dim, head_dim * sizeof(float));
432 memcpy(kv_cache_v + pos * kv_elems + h * head_dim,
433 v + h * head_dim, head_dim * sizeof(float));
434 }
435
436 /* Phase 3: Flash Attention (o in L1) */
437 mega_fuse_flash_attention_avx(o, q, kv_cache_k, kv_cache_v,
438 num_heads, num_kv_heads, pos + 1, head_dim);
439
440 /* Phase 4: Output Projection + Residual (in final store) */
441 for (int j = 0; j < hidden; j++) {
442 float dot = 0.0f;
443 for (int h = 0; h < num_heads; h++) {
444 const float *o_h = o + h * head_dim;
445 const float *W_o_h = W_o + h * head_dim * hidden + j * head_dim;
446 for (int i = 0; i < head_dim; i++) {
447 dot += o_h[i] * W_o_h[i];
448 }
449 }
450 float proj = dot + (b_o ? b_o[j] : 0.0f);
451 output[j] = proj + residual[j]; /* Residual add in store! */
452 }
453
454 /* No free needed - stack buffers auto-deallocate */
455 /* Total DRAM traffic: 4KB input + 4KB output = 8KB */
456}
