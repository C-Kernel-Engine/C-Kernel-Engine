1// KV-cache helper kernels (head-major layout).
2//
3// These are small, explicit helpers used by the runtime/orchestrator to maintain
4// per-layer KV caches during autoregressive decoding.
5//
6// Layout:
7// k_cache[ kv_head, token, aligned_head_dim ]
8// v_cache[ kv_head, token, aligned_head_dim ]
9// with contiguous row-major storage and stride aligned_head_dim.
10

16void kv_cache_repack_head_major_inplace(float *buf,
17 int num_heads,
18 int tokens,
19 int cache_capacity,
20 int aligned_head_dim)
21{
22 if (!buf) {
23 return;
24 }
25 if (num_heads <= 0 || tokens <= 0 || cache_capacity <= 0 || aligned_head_dim <= 0) {
26 return;
27 }
28 if (tokens > cache_capacity) {
29 tokens = cache_capacity;
30 }
31 if (tokens == cache_capacity) {
32 return;
33 }
34
35 const size_t old_head_stride = (size_t)tokens * (size_t)aligned_head_dim;
36 const size_t new_head_stride = (size_t)cache_capacity * (size_t)aligned_head_dim;
37 const size_t bytes = (size_t)tokens * (size_t)aligned_head_dim * sizeof(float);
38
39 // Move head blocks from high to low to avoid overwriting source data
40 // for heads that have not yet been moved.
41 for (int h = num_heads - 1; h >= 0; --h) {
42 float *src = buf + (size_t)h * old_head_stride;
43 float *dst = buf + (size_t)h * new_head_stride;
44 memmove(dst, src, bytes);
45 }
46}

48void kv_cache_write_head_major(const float *__restrict k_token,
49 const float *__restrict v_token,
50 float *__restrict k_cache,
51 float *__restrict v_cache,
52 int num_kv_heads,
53 int token_index,
54 int cache_capacity,
55 int head_dim,
56 int aligned_head_dim)
57{
58 if (!k_token || !v_token || !k_cache || !v_cache) {
59 return;
60 }
61 if (num_kv_heads <= 0 || token_index < 0 || cache_capacity <= 0) {
62 return;
63 }
64 if (token_index >= cache_capacity || head_dim <= 0 || aligned_head_dim <= 0) {
65 return;
66 }
67
68 const size_t head_stride = (size_t)cache_capacity * (size_t)aligned_head_dim;
69 const size_t token_stride = (size_t)aligned_head_dim;
70
71 for (int h = 0; h < num_kv_heads; ++h) {
72 const float *k_src = k_token + (size_t)h * token_stride;
73 const float *v_src = v_token + (size_t)h * token_stride;
74
75 float *k_dst = k_cache + (size_t)h * head_stride + (size_t)token_index * token_stride;
76 float *v_dst = v_cache + (size_t)h * head_stride + (size_t)token_index * token_stride;
77
78 for (int d = 0; d < head_dim; ++d) {
79 k_dst[d] = k_src[d];
80 v_dst[d] = v_src[d];
81 }
82 for (int d = head_dim; d < aligned_head_dim; ++d) {
83 k_dst[d] = 0.0f;
84 v_dst[d] = 0.0f;
85 }
86 }
87}
