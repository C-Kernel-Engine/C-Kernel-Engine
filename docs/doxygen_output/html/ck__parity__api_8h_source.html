142 /**
143  * @brief RoPE (Rotary Position Embedding)
144  *
145  * Applies rotary position embeddings to Q and K tensors.
146  *
147  * NOTE: CK uses rotate-half format (split first/second halves)
148  * while some implementations use interleaved format.
149  * The test harness should account for this.
150  *
151  * @param q Query tensor [n_tokens, n_heads * head_dim], modified in-place
152  * @param k Key tensor [n_tokens, n_heads_kv * head_dim], modified in-place
153  * @param n_tokens Number of tokens
154  * @param n_heads Number of query heads
155  * @param n_heads_kv Number of key/value heads
156  * @param head_dim Dimension per head
157  * @param pos_offset Starting position for RoPE
158  * @param theta RoPE base frequency (typically 10000.0)
159  */
160 void ck_test_rope(float *q, float *k,
161  int n_tokens, int n_heads, int n_heads_kv, int head_dim,
162  int pos_offset, float theta);
