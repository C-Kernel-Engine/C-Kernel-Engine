63 /* Create a new tokenizer */
64 CKTokenizer *ck_tokenizer_create(CKTokenizerType type) {
65  CKTokenizer *tok = (CKTokenizer *)malloc(sizeof(CKTokenizer));
66  if (!tok) {
67  return NULL;
68  }
69 
70  memset(tok, 0, sizeof(*tok));
71 
72  /* Create hash table for vocabulary */
73  tok->vocab = ck_tokenizer_hash_table_create(CK_TOKENIZER_HT_BUCKETS_LARGE);
74  if (!tok->vocab) {
75  free(tok);
76  return NULL;
77  }
78 
79  /* Create trie for fast lookups (1M nodes for ~50k vocab) */
80  tok->vocab_trie = ck_trie_create(1000000);
81  if (!tok->vocab_trie) {
82  ck_tokenizer_hash_table_free(tok->vocab, true);
83  free(tok);

86 
87  /* Initialize reverse vocab */

92  free(tok);
93  return NULL;
94  }
95 
96  /* Set default special tokens */
97  tok->unk_id = 0;
98  tok->bos_id = 1;
99  tok->eos_id = 2;
100  tok->pad_id = -1;
101  tok->mask_id = -1;
102 
103  /* Set config */

163 /* Add a token to vocabulary */
164 int ck_tokenizer_add_token(CKTokenizer *tok, const char *token, int32_t id, float score) {
165  if (!tok || !token) {
166  return -1;
167  }
168 
169  /* Ensure we have space in reverse vocab */
170  if (id >= (int32_t)tok->vocab_capacity) {
171  size_t new_cap = tok->vocab_capacity * 2;
172  while (new_cap <= (size_t)id) {
173  new_cap *= 2;
174  }
175  char **new_array = (char **)realloc(tok->id_to_token, new_cap * sizeof(char *));
176  if (!new_array) {
177  return -1;
178  }
179  memset(new_array + tok->vocab_capacity, 0, (new_cap - tok->vocab_capacity) * sizeof(char *));
180  tok->id_to_token = new_array;
181  tok->vocab_capacity = new_cap;
182  }
183 
184  /* Check if token already exists */
185  TokenInfo *existing = (TokenInfo *)ck_tokenizer_hash_table_lookup(tok->vocab, token);
186  if (existing) {
187  existing->id = id;
188  existing->score = score;
189  if (id >= (int32_t)tok->vocab_size) tok->vocab_size = id + 1;
190  if (tok->id_to_token[id]) free(tok->id_to_token[id]);
191  tok->id_to_token[id] = strdup(token);
192  return 0;
193  }
194 
195  /* Create new token info */
196  TokenInfo *info = (TokenInfo *)malloc(sizeof(TokenInfo));
197  if (!info) return -1;
198  info->id = id;
199  info->score = score;
200  info->is_special = false;
201 
202  if (ck_tokenizer_hash_table_insert(tok->vocab, token, info) != 0) {
203  free(info);
204  return -1;
205  }
206 
207  /* Also add to trie for fast longest-match lookups */
208  if (tok->vocab_trie) {
209  ck_trie_insert(tok->vocab_trie, token, id, false, 0);
210  }
211 
212  if (id >= (int32_t)tok->vocab_size) tok->vocab_size = id + 1;
213  if (tok->id_to_token[id]) free(tok->id_to_token[id]);
214  tok->id_to_token[id] = strdup(token);
215 
216  return 0;
217 }

219 /* Add special token */
220 int ck_tokenizer_add_special_token(CKTokenizer *tok, const char *name, int32_t id) {
221  if (!tok || !name) return -1;
222  if (ck_tokenizer_add_token(tok, name, id, -1e10f) != 0) return -1;
223 
224  TokenInfo *info = (TokenInfo *)ck_tokenizer_hash_table_lookup(tok->vocab, name);
225  if (info) info->is_special = true;
226 
227  /* Also add to trie as special */
228  if (tok->vocab_trie) {
229  ck_trie_insert(tok->vocab_trie, name, id, true, 0);
230  }
231 
232  if (strcmp(name, "<unk>") == 0 || strcmp(name, "[UNK]") == 0) tok->unk_id = id;
233  else if (strcmp(name, "<s>") == 0 || strcmp(name, "<bos>") == 0 || strcmp(name, "[BOS]") == 0) tok->bos_id = id;
234  else if (strcmp(name, "</s>") == 0 || strcmp(name, "<eos>") == 0 || strcmp(name, "[EOS]") == 0) tok->eos_id = id;
235  else if (strcmp(name, "<pad>") == 0 || strcmp(name, "[PAD]") == 0) tok->pad_id = id;
236 
237  return 0;
238 }

265 /* Auto-detect space prefix style from vocabulary.
266  * Checks for presence of tokens starting with Ġ (GPT-2) vs ▁ (SentencePiece). */
267 CKSpacePrefixStyle ck_tokenizer_detect_space_prefix_style(CKTokenizer *tok) {
268  if (!tok) return CK_SPACE_PREFIX_GPT2;
269 
270  /* Already detected? */
271  if (tok->config.space_prefix_detected && tok->config.space_prefix_style != CK_SPACE_PREFIX_AUTO) {
272  return tok->config.space_prefix_style;
273  }
274 
275  /* Count tokens starting with each style:
276  * Ġ (U+0120) = bytes 0xC4 0xA0
277  * ▁ (U+2581) = bytes 0xE2 0x96 0x81
278  */
279  int gpt2_count = 0;
280  int spm_count = 0;
281 
282  for (size_t i = 0; i < tok->vocab_size && i < 10000; i++) { /* Sample first 10k tokens */
283  const char *token = tok->id_to_token[i];
284  if (!token) continue;
285 
286  unsigned char c0 = (unsigned char)token[0];
287  unsigned char c1 = (unsigned char)token[1];
288 
289  /* Check for Ġ (0xC4 0xA0) */
290  if (c0 == 0xC4 && c1 == 0xA0) {
291  gpt2_count++;
292  }
293  /* Check for ▁ (0xE2 0x96 0x81) */
294  else if (c0 == 0xE2 && c1 == 0x96 && (unsigned char)token[2] == 0x81) {
295  spm_count++;
296  }
297  }
298 
299  /* Determine style based on counts */
300  CKSpacePrefixStyle detected;
301  if (spm_count > gpt2_count * 2) {
302  detected = CK_SPACE_PREFIX_SPM;
303  } else {
304  detected = CK_SPACE_PREFIX_GPT2; /* Default to GPT-2 if similar counts */
305  }
306 
307  tok->config.space_prefix_style = detected;
308  tok->config.space_prefix_detected = true;
309 
310  return detected;
311 }

327 static int32_t find_longest_match_trie(const CKTokenizer *tok, const char *text, size_t text_len, size_t pos, size_t *match_len) {
328  if (!tok || !tok->vocab_trie || !text || pos >= text_len) {
329  *match_len = 0;
330  return tok ? tok->unk_id : -1;
331  }
332 
333  int32_t token_id = ck_trie_find_longest(tok->vocab_trie, text, text_len, pos, match_len);
334  return token_id >= 0 ? token_id : tok->unk_id;

337 /* Find longest matching token at position using hash table (O(n*k) worst case) */
338 static int32_t find_longest_match_hash(const CKTokenizer *tok, const char *text, size_t text_len, size_t pos, size_t *match_len) {
339  if (!tok || !text || pos >= text_len) {
340  *match_len = 0;
341  return tok ? tok->unk_id : -1;
342  }
343 
344  size_t max_len = 64;
345  if (pos + max_len > text_len) max_len = text_len - pos;
346 
347  int32_t best_id = tok->unk_id;
348  size_t best_len = 0;
349 
350  for (size_t len = max_len; len >= 1; len--) {
351  char tmp[65];
352  memcpy(tmp, text + pos, len);
353  tmp[len] = '\0';
354 
355  TokenInfo *info = (TokenInfo *)ck_tokenizer_hash_table_lookup(tok->vocab, tmp);
356  if (info) {
357  best_id = info->id;
358  best_len = len;
359  break;
360  }
361  }
362 
363  *match_len = best_len;
364  return best_id;
365 }

367 /* Find longest matching token at position - dispatches to trie or hash table */

376 /* Convert ASCII spaces to space prefix marker.
377  * GPT-2/Qwen use Ġ (U+0120, bytes 0xC4 0xA0) - replaces spaces only
378  * LLaMA/SentencePiece use ▁ (U+2581, bytes 0xE2 0x96 0x81) - adds prefix at start AND replaces spaces
379  * Returns new length, or -1 if buffer too small. */
380 static int preprocess_bpe_spaces(const char *text, int text_len, char *out, int out_max, CKSpacePrefixStyle style) {
381  int out_len = 0;
382 
383  /* For SentencePiece, add ▁ at the start of text (unless text starts with space) */
384  if (style == CK_SPACE_PREFIX_SPM && text_len > 0 && text[0] != ' ') {
385  if (out_len + 3 > out_max) return -1;
386  out[out_len++] = (char)0xE2;
387  out[out_len++] = (char)0x96;
388  out[out_len++] = (char)0x81;
389  }
390 
391  for (int i = 0; i < text_len; i++) {
392  if (text[i] == ' ') {
393  if (style == CK_SPACE_PREFIX_SPM) {
394  /* SentencePiece style: ▁ (3 bytes: 0xE2 0x96 0x81) */
395  if (out_len + 3 > out_max) return -1;
396  out[out_len++] = (char)0xE2;
397  out[out_len++] = (char)0x96;
398  out[out_len++] = (char)0x81;
399  } else {
400  /* GPT-2 style: Ġ (2 bytes: 0xC4 0xA0) */
401  if (out_len + 2 > out_max) return -1;
402  out[out_len++] = (char)0xC4;
403  out[out_len++] = (char)0xA0;
404  }
405  } else {
406  if (out_len + 1 > out_max) return -1;
407  out[out_len++] = text[i];
408  }
409  }
410  return out_len;
411 }

413 /* Encode text to token IDs using greedy longest-match */
414 int ck_tokenizer_encode(const CKTokenizer *tok, const char *text, int text_len, int32_t *ids, int max_ids) {
415  if (!tok || !text || !ids || max_ids <= 0) return 0;
416  if (text_len < 0) text_len = (int)strlen(text);
417 
418  /* For BPE tokenizers, convert spaces to appropriate prefix marker.
419  * Auto-detect style from vocabulary if not already set. */
420  char preprocessed[8192];
421  const char *input = text;
422  int input_len = text_len;
423 
424  if (tok->config.type == CK_TOKENIZER_BPE) {
425  /* Get or detect space prefix style */
426  CKSpacePrefixStyle style = ((CKTokenizer *)tok)->config.space_prefix_style;
427  if (!((CKTokenizer *)tok)->config.space_prefix_detected) {
428  style = ck_tokenizer_detect_space_prefix_style((CKTokenizer *)tok);
429  }
430 
431  int pp_len = preprocess_bpe_spaces(text, text_len, preprocessed, sizeof(preprocessed) - 1, style);
432  if (pp_len > 0) {
433  preprocessed[pp_len] = '\0';
434  input = preprocessed;
435  input_len = pp_len;
436  }
437  }
438 
439  int out_idx = 0;
440  if (tok->config.add_bos && tok->bos_id >= 0 && out_idx < max_ids) {
441  ids[out_idx++] = tok->bos_id;
442  }
443 
444  size_t pos = 0;
445  while (pos < (size_t)input_len && out_idx < max_ids) {
446  size_t match_len = 0;
447  int32_t id = find_longest_match(tok, input, input_len, pos, &match_len);
448 
449  if (match_len == 0) {
450  /* Emit UNK for unknown characters */
451  if (tok->unk_id >= 0) ids[out_idx++] = tok->unk_id;
452  pos++;
453  } else {
454  ids[out_idx++] = id;
455  pos += match_len;
456  }
457  }
458 
459  if (tok->config.add_eos && tok->eos_id >= 0 && out_idx < max_ids) {
460  ids[out_idx++] = tok->eos_id;
461  }
462 
463  return out_idx;
464 }

466 /* Decode token IDs to text */
467 int ck_tokenizer_decode(const CKTokenizer *tok, const int32_t *ids, int num_ids, char *text, int max_len) {
468  if (!tok || !ids || !text || max_len <= 0) return 0;
469  int len = 0;
470  for (int i = 0; i < num_ids; i++) {
471  int32_t id = ids[i];
472  if (id < 0) continue;
473  const char *token = ck_tokenizer_id_to_token(tok, id);
474  if (!token) continue;
475  int token_len = (int)strlen(token);
476 
477  /* Check for space prefix markers and convert to ASCII space */
478  unsigned char c0 = (unsigned char)token[0];
479  unsigned char c1 = (unsigned char)token[1];
480 
481  if (c0 == 0xC4 && c1 == 0xA0) {
482  /* Ġ (U+0120) is 2 bytes - convert to space */
483  if (len < max_len - 1) text[len++] = ' ';
484  token += 2; token_len -= 2;
485  } else if (c0 == 0xE2 && c1 == 0x96 && (unsigned char)token[2] == 0x81) {
486  /* ▁ (U+2581) is 3 bytes - convert to space */
487  if (len < max_len - 1) text[len++] = ' ';
488  token += 3; token_len -= 3;
489  }
490 
491  for (int j = 0; j < token_len && len < max_len - 1; j++) text[len++] = token[j];
492  }
493  text[len] = '\0';
494  return len;
495 }
