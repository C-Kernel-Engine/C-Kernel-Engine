45static int ck_parse_env_int(const char *name)
46{
47 const char *val = getenv(name);
48 if (!val || !val[0]) {
49 return 0;
50 }
51
52 errno = 0;
53 char *end = NULL;
54 long n = strtol(val, &end, 10);
55 if (errno != 0 || end == val || n <= 0 || n > (1L << 20)) {
56 return 0;
57 }
58 return (int)n;
59}

61// Detect physical CPU cores (not hyperthreads) when possible.
62int ck_get_physical_cores(void)
63{
64 int physical_cores = 0;
65 int logical_cores = (int)sysconf(_SC_NPROCESSORS_ONLN);
66 if (logical_cores <= 0) {
67 logical_cores = 1;
68 }
69
70 int cpu_cores_hint = 0;
71 int siblings_hint = 0;
72
73 // Read from /proc/cpuinfo (Linux) and count unique (physical id, core id) pairs.
74 FILE *f = fopen("/proc/cpuinfo", "r");
75 if (f) {
76 char line[256];
77 int physical_id = -1;
78 int core_id = -1;
79
80 struct {
81 int physical_id;
82 int core_id;
83 } seen[8192];
84 int seen_count = 0;
85
86 const int seen_cap = (int)(sizeof(seen) / sizeof(seen[0]));
87
88 // Helper: add (pid,cid) to set if not present.
89 #define CK_ADD_PAIR(pid, cid) \
90 do { \
91 if ((pid) >= 0 && (cid) >= 0) { \
92 int exists = 0; \
93 for (int ii = 0; ii < seen_count; ++ii) { \
94 if (seen[ii].physical_id == (pid) && \
95 seen[ii].core_id == (cid)) { \
96 exists = 1; \
97 break; \
98 } \
99 } \
100 if (!exists && seen_count < seen_cap) { \
101 seen[seen_count].physical_id = (pid); \
102 seen[seen_count].core_id = (cid); \
103 ++seen_count; \
104 } \
105 } \
106 } while (0)
107
108 while (fgets(line, sizeof(line), f)) {
109 int val;
110
111 // Blank line separates processor blocks.
112 if (line[0] == '\n' || line[0] == '\0') {
113 CK_ADD_PAIR(physical_id, core_id);
114 physical_id = -1;
115 core_id = -1;
116 continue;
117 }
118
119 if (sscanf(line, "physical id : %d", &val) == 1) {
120 physical_id = val;
121 continue;
122 }
123 if (sscanf(line, "core id : %d", &val) == 1) {
124 core_id = val;
125 continue;
126 }
127 if (sscanf(line, "cpu cores : %d", &val) == 1) {
128 if (val > cpu_cores_hint) cpu_cores_hint = val;
129 continue;
130 }
131 if (sscanf(line, "siblings : %d", &val) == 1) {
132 if (val > siblings_hint) siblings_hint = val;
133 continue;
134 }
135 }
136 fclose(f);
137
138 // Handle file without trailing blank line.
139 CK_ADD_PAIR(physical_id, core_id);
140
141 #undef CK_ADD_PAIR
142
143 physical_cores = seen_count;
144 }
145
146 // Fallback: infer threads-per-core from siblings/cpu cores when pair data
147 // is missing (common in containers/VMs).
148 if (physical_cores <= 1 && logical_cores > 1) {
149 int threads_per_core = 0;
150 if (siblings_hint > 0 && cpu_cores_hint > 0 && siblings_hint >= cpu_cores_hint) {
151 threads_per_core = siblings_hint / cpu_cores_hint;
152 }
153 if (threads_per_core > 1) {
154 int inferred_physical = logical_cores / threads_per_core;
155 if (inferred_physical > 1) {
156 return inferred_physical;
157 }
158 }
159 if (cpu_cores_hint > 1 && cpu_cores_hint <= logical_cores) {
160 return cpu_cores_hint;
161 }
162 return logical_cores;
163 }
164
165 if (physical_cores > 1) {
166 return physical_cores;
167 }
168
169 return logical_cores;
170}

172void ck_set_num_threads(int num_threads)
173{
174 // 0 = auto-detect
175 if (num_threads <= 0) {
176 // Prefer explicit env controls when present:
177 // - CK_NUM_THREADS: engine-level override
178 // - OMP_NUM_THREADS: standard OpenMP control (set by `ck run --threads`)
179 int env_threads = ck_parse_env_int("CK_NUM_THREADS");
180 if (env_threads <= 0) {
181 env_threads = ck_parse_env_int("OMP_NUM_THREADS");
182 }
183 num_threads = env_threads > 0 ? env_threads : ck_get_physical_cores();
184 }
185
186 g_num_threads = num_threads;
187 g_threads_initialized = 1;
188
189#ifdef _OPENMP
190 omp_set_dynamic(0); // Disable dynamic adjustment
191 omp_set_num_threads(num_threads);
192#endif
193
194#if defined(USE_MKL)
195 mkl_set_num_threads(num_threads);
196#endif
197
198 fprintf(stderr, "[CK] Set %d threads (auto=%d)\n",
199 num_threads, ck_get_physical_cores());
200}
