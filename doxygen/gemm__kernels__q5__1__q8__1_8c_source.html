14/* Quantize one FP32 row to Q8_1 blocks (ggml-compatible scalar path). */
15static void quantize_row_q8_1_scalar(const float *x, block_q8_1 *y, int k) {
16 const int nb = k / QK8_1;
17 for (int b = 0; b < nb; ++b) {
18 const float *xb = x + (size_t)b * QK8_1;
19 float amax = 0.0f;
20 for (int j = 0; j < QK8_1; ++j) {
21 float av = xb[j] >= 0.0f ? xb[j] : -xb[j];
22 if (av > amax) amax = av;
23 }
24
25 const float d = amax / 127.0f;
26 const float id = (d != 0.0f) ? (1.0f / d) : 0.0f;
27 y[b].d = CK_FP32_TO_FP16(d);
28
29 int sum = 0;
30 for (int j = 0; j < QK8_1; ++j) {
31 int q = (int)roundf(xb[j] * id);
32 y[b].qs[j] = (int8_t)q;
33 sum += q;
34 }
35 y[b].s = CK_FP32_TO_FP16((float)sum * d);
36 }
37}

39/* One 32-element block dot: Q5_1(weights) x Q8_1(activations). */
40static float dot_q5_1_q8_1_block(const block_q5_1 *w, const block_q8_1 *x) {
41 uint32_t qh;
42 memcpy(&qh, w->qh, sizeof(qh));
43
44 int sumi0 = 0;
45 int sumi1 = 0;
46 for (int j = 0; j < QK5_1 / 2; ++j) {
47 const uint8_t xh0 = (uint8_t)(((qh >> (j + 0)) << 4) & 0x10);
48 const uint8_t xh1 = (uint8_t)(((qh >> (j + 12)) ) & 0x10);
49 const int32_t q0 = (int32_t)((w->qs[j] & 0x0F) | xh0);
50 const int32_t q1 = (int32_t)((w->qs[j] >> 4) | xh1);
51 sumi0 += q0 * (int32_t)x->qs[j];
52 sumi1 += q1 * (int32_t)x->qs[j + QK5_1 / 2];
53 }
54
55 const float wd = CK_FP16_TO_FP32(w->d);
56 const float wm = CK_FP16_TO_FP32(w->m);
57 const float xd = CK_FP16_TO_FP32(x->d);
58 const float xs = CK_FP16_TO_FP32(x->s);
59 return (wd * xd) * (float)(sumi0 + sumi1) + wm * xs;
60}

62void gemv_q5_1_q8_1_ref(float *y,
63 const void *W,
64 const void *x_q8,
65 int M, int K)
66{
67 if (!y || !W || !x_q8 || M <= 0 || K <= 0 || (K % QK5_1) != 0) {
68 return;
69 }
70
71 const block_q5_1 *blocks = (const block_q5_1 *)W;
72 const block_q8_1 *x = (const block_q8_1 *)x_q8;
73 const int blocks_per_row = K / QK5_1;
74
75 for (int row = 0; row < M; ++row) {
76 const block_q5_1 *w_row = &blocks[row * blocks_per_row];
77 float sum = 0.0f;
78 for (int b = 0; b < blocks_per_row; ++b) {
79 sum += dot_q5_1_q8_1_block(&w_row[b], &x[b]);
80 }
81 y[row] = sum;
82 }
83}

85void gemm_nt_q5_1_q8_1_ref(const void *A_q8,
86 const void *B,
87 const float *bias,
88 float *C,

90{
91 if (!A_q8 || !B || !C || M <= 0 || N <= 0 || K <= 0 || (K % QK5_1) != 0) {
92 return;
93 }
94
95 const block_q8_1 *A = (const block_q8_1 *)A_q8;
96 const block_q5_1 *W = (const block_q5_1 *)B;
97 const int blocks_per_row = K / QK5_1;
98
99 for (int m = 0; m < M; ++m) {
100 const block_q8_1 *a_row = &A[m * blocks_per_row];
101 for (int n = 0; n < N; ++n) {
102 const block_q5_1 *w_row = &W[n * blocks_per_row];
103 float sum = 0.0f;
104 for (int b = 0; b < blocks_per_row; ++b) {
105 sum += dot_q5_1_q8_1_block(&w_row[b], &a_row[b]);
106 }
107 C[m * N + n] = sum + (bias ? bias[n] : 0.0f);
108 }
109 }
