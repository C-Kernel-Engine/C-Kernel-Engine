10static inline int32_t hsum_epi32_sse(__m128i v) {
11 __m128i shuf = _mm_shuffle_epi32(v, _MM_SHUFFLE(1, 0, 3, 2));
12 __m128i sums = _mm_add_epi32(v, shuf);
13 shuf = _mm_shuffle_epi32(sums, _MM_SHUFFLE(2, 3, 0, 1));
14 sums = _mm_add_epi32(sums, shuf);
15 return _mm_cvtsi128_si32(sums);
16}

93 __m128i prod_hi_H = _mm_madd_epi16(q4_hi_16_H, q8_hi_16_H);

101 int32_t bsum_lo = (int32_t)b8->bsums[j / 16] +
102 (int32_t)b8->bsums[j / 16 + 1];
103 int32_t bsum_hi = (int32_t)b8->bsums[(j + 32) / 16] +
104 (int32_t)b8->bsums[(j + 32) / 16 + 1];
105
106 sumf += d * (float)sc[is] * (float)sum_q4q8_lo;
107 sumf -= dmin * (float)m_val[is] * (float)bsum_lo;
108 sumf += d * (float)sc[is + 1] * (float)sum_q4q8_hi;
109 sumf -= dmin * (float)m_val[is + 1] * (float)bsum_hi;
110
111 q_offset += 32;
112 is += 2;
113 }
