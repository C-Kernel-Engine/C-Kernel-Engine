2  * @file parallel_orchestration.c
3  * @brief Parallel decode orchestration for CK-Engine v6.6
4  *
5  * Demonstrates the llama.cpp-style parallelization pattern:
6  * - OpenMP parallel region at orchestration level
7  * - Each kernel receives (ith, nth) and processes its slice
8  * - Barriers between dependent operations
9  *
10  * Key insight: OpenMP overhead is amortized over the entire decode,
11  * not per-kernel. Thread pool stays active for entire forward pass.
12  */

29  * Single-token decode with parallel SIMD kernels.
30  *
31  * This is the main decode function that processes one token through all layers.
32  * OpenMP parallel region is created ONCE at the top, and all kernels
33  * receive (ith, nth) to split their work.
34  *
35  * Pattern:
36  * #pragma omp parallel
37  * {
38  * int ith = omp_get_thread_num();
39  * int nth = omp_get_num_threads();
40  *
41  * // Each kernel processes only its slice
42  * gemv_q4_k_q8_k_parallel_simd(..., ith, nth);
43  * #pragma omp barrier
44  *
45  * rmsnorm_parallel(..., ith, nth); // (not implemented yet)
46  * #pragma omp barrier
47  * ...
48  * }
49  */

51 /* Parallel residual add: out = a + b, split across threads */
52 static void residual_add_parallel(const float *a, const float *b,
53  float *out, int n,
54  int ith, int nth)
55 {
56  const int dr = (n + nth - 1) / nth;
57  const int r0 = dr * ith;
58  const int r1 = (r0 + dr < n) ? (r0 + dr) : n;
59 
60  if (r0 >= n) return;
61 
62  for (int i = r0; i < r1; i++) {
63  out[i] = a[i] + b[i];
64  }
65 }

83 static void vec_zero_parallel(float *y, int n, int ith, int nth)
84 {
85  const int dr = (n + nth - 1) / nth;
86  const int r0 = dr * ith;
87  const int r1 = (r0 + dr < n) ? (r0 + dr) : n;
88 
89  if (r0 >= n) return;
90 
91  memset(y + r0, 0, (r1 - r0) * sizeof(float));

102  * Parallel Q/K/V projection for single token decode.

108  * @param q_out Output: Q vectors [H, head_dim]
109  * @param k_out Output: K vector [H_kv, head_dim]
110  * @param v_out Output: V vector [H_kv, head_dim]
111  * @param H Number of query heads
112  * @param H_kv Number of KV heads (GQA)
113  * @param head_dim Head dimension
114  * @param embed_dim Embedding dimension
115  * @param num_threads Number of threads to use (0 = auto)
116  */
117 void qkv_projection_parallel(const void *ln1_q8,
118  const void *WQ,
119  const void *WK,
120  const void *WV,
121  float *q_out,
122  float *k_out,
123  float *v_out,
124  int H, int H_kv,
125  int head_dim, int embed_dim,
126  int num_threads)
127 {
128  const int q_dim = H * head_dim;
129  const int kv_dim = H_kv * head_dim;
130 
131  /* Align to QK_K for quantized matmul */
132  const int aligned_embed = ((embed_dim + 255) / 256) * 256;
133 
134  if (num_threads <= 0) {
135  num_threads = omp_get_max_threads();
136  }
137 
138  /* Single OpenMP region for all three projections */
139  #pragma omp parallel num_threads(num_threads)
140  {
141  const int ith = omp_get_thread_num();
142  const int nth = omp_get_num_threads();
143 
144  /* Q projection: largest, benefits most from parallelism */
145  gemv_q4_k_q8_k_parallel_simd(q_out, WQ, ln1_q8, q_dim, aligned_embed, ith, nth);
146 
147  /* K projection: smaller, but still benefits */
148  gemv_q4_k_q8_k_parallel_simd(k_out, WK, ln1_q8, kv_dim, aligned_embed, ith, nth);
149 
150  /* V projection */
151  gemv_q4_k_q8_k_parallel_simd(v_out, WV, ln1_q8, kv_dim, aligned_embed, ith, nth);
152 
153  /* Implicit barrier at end of parallel region */
154  }
155 }

157 /**
158  * Parallel MLP (gate/up + SwiGLU + down projection).
159  *
160  * @param ln2_q8 Input: RMSNorm output quantized to Q8_K
161  * @param W_gate Gate weights [intermediate, embed] in Q4_K
162  * @param W_up Up weights [intermediate, embed] in Q4_K
163  * @param W_down Down weights [embed, intermediate] in Q4_K
164  * @param gate_buf Scratch: gate output [intermediate]
165  * @param up_buf Scratch: up output [intermediate]
166  * @param swiglu_buf Scratch: SwiGLU output [intermediate]
167  * @param down_q8 Scratch: down input quantized [intermediate Q8_K blocks]
168  * @param mlp_out Output: MLP output [embed]
169  * @param intermediate Intermediate dimension
170  * @param embed_dim Embedding dimension
171  * @param num_threads Number of threads (0 = auto)
172  */
173 void mlp_parallel(const void *ln2_q8,
174  const void *W_gate,
175  const void *W_up,
176  const void *W_down,
177  float *gate_buf,
178  float *up_buf,
179  float *swiglu_buf,
180  void *down_q8,
181  float *mlp_out,
182  int intermediate,
183  int embed_dim,
184  int num_threads)
185 {
186  const int aligned_embed = ((embed_dim + 255) / 256) * 256;
187  const int aligned_inter = ((intermediate + 255) / 256) * 256;
188 
189  if (num_threads <= 0) {
190  num_threads = omp_get_max_threads();
191  }
192 
193  #pragma omp parallel num_threads(num_threads)
194  {
195  const int ith = omp_get_thread_num();
196  const int nth = omp_get_num_threads();
197 
198  /* Gate and Up projections (can run in parallel, no dependency) */
199  gemv_q4_k_q8_k_parallel_simd(gate_buf, W_gate, ln2_q8, aligned_inter, aligned_embed, ith, nth);
200  gemv_q4_k_q8_k_parallel_simd(up_buf, W_up, ln2_q8, aligned_inter, aligned_embed, ith, nth);
201 
202  #pragma omp barrier /* Wait for gate and up to complete */
203 
204  /* SwiGLU: swiglu = silu(gate) * up */
205  /* This is element-wise, parallelize across elements */
206  const int dr = (aligned_inter + nth - 1) / nth;
207  const int r0 = dr * ith;
208  const int r1 = (r0 + dr < aligned_inter) ? (r0 + dr) : aligned_inter;
209 
210  for (int i = r0; i < r1 && i < intermediate; i++) {
211  float g = gate_buf[i];
212  float silu_g = g / (1.0f + expf(-g)); /* SiLU activation */
213  swiglu_buf[i] = silu_g * up_buf[i];
214  }
215 
216  #pragma omp barrier /* Wait for SwiGLU to complete */
217 
218  /* Down projection: only thread 0 quantizes (single-threaded) */
219  /* TODO: Add parallel quantization */
220  #pragma omp single
221  {
222  quantize_row_q8_k(swiglu_buf, down_q8, aligned_inter);
223  }
224  /* Implicit barrier after omp single */
225 
226  /* Down projection */
227  gemv_q4_k_q8_k_parallel_simd(mlp_out, W_down, down_q8, aligned_embed, aligned_inter, ith, nth);
228  }
229 }

238  * Process one transformer layer in parallel.
239  *
240  * This demonstrates the full parallel pattern for a single layer.
241  * In production, this would be called in a loop for all layers.
242  */
243 void decode_layer_parallel(
244  /* Inputs */
245  float *hidden, /* [embed_dim] - modified in place */
246  const void *ln1_weight, /* RMSNorm weights */
247  const void *ln2_weight, /* RMSNorm weights */
248  const void *WQ, /* Q4_K weights */
249  const void *WK,
250  const void *WV,
251  const void *WO,
252  const void *W_gate,
253  const void *W_up,
254  const void *W_down,
255  /* KV cache */
256  float *k_cache, /* [H_kv, max_seq, head_dim] */
257  float *v_cache,
258  int token_index, /* Current position in sequence */
259  /* Scratch buffers */
260  float *scratch, /* Aligned scratch space */
261  /* Model config */
262  int embed_dim,
263  int intermediate,
264  int H, int H_kv,
265  int head_dim,
266  int max_seq,
267  float eps, /* RMSNorm epsilon */
268  /* Threading */
269  int num_threads)
270 {
271  if (num_threads <= 0) {
272  num_threads = omp_get_max_threads();
273  }
274 
275  /* Align dimensions */
276  const int aligned_embed = ((embed_dim + 255) / 256) * 256;
277  const int aligned_inter = ((intermediate + 255) / 256) * 256;
278  const int aligned_head = ((head_dim + 31) / 32) * 32;
279 
280  /* Partition scratch buffer */
281  float *ln1_out = scratch;
282  float *q_vec = ln1_out + aligned_embed;
283  float *k_vec = q_vec + H * aligned_head;
284  float *v_vec = k_vec + H_kv * aligned_head;
285  float *attn_out = v_vec + H_kv * aligned_head;
286  float *o_out = attn_out + H * aligned_head;
287  float *ln2_out = o_out + aligned_embed;
288  float *gate_buf = ln2_out + aligned_embed;
289  float *up_buf = gate_buf + aligned_inter;
290  float *swiglu_buf = up_buf + aligned_inter;
291  float *mlp_out = swiglu_buf + aligned_inter;
292 
293  /* Q8_K buffers for quantized input */
294  const size_t q8_embed_bytes = ((aligned_embed + 255) / 256) * 292;
295  const size_t q8_inter_bytes = ((aligned_inter + 255) / 256) * 292;
296  uint8_t *ln1_q8 = (uint8_t *)(mlp_out + aligned_embed);
297  uint8_t *ln2_q8 = ln1_q8 + q8_embed_bytes;
298  uint8_t *down_q8 = ln2_q8 + q8_embed_bytes;
299 
300  #pragma omp parallel num_threads(num_threads)
301  {
302  const int ith = omp_get_thread_num();
303  const int nth = omp_get_num_threads();
304 
305  /* ============================================================
306  * ATTENTION BLOCK
307  * ============================================================ */
308 
309  /* Step 1: RMSNorm (TODO: parallelize reduction) */
310  #pragma omp single
311  {
312  rmsnorm(hidden, (const float *)ln1_weight, ln1_out, embed_dim, eps);
313  quantize_row_q8_k(ln1_out, ln1_q8, aligned_embed);
314  }
315  /* Implicit barrier after single */
316 
317  /* Step 2: QKV projections (parallel) */
318  gemv_q4_k_q8_k_parallel_simd(q_vec, WQ, ln1_q8, H * head_dim, aligned_embed, ith, nth);
319  gemv_q4_k_q8_k_parallel_simd(k_vec, WK, ln1_q8, H_kv * head_dim, aligned_embed, ith, nth);
320  gemv_q4_k_q8_k_parallel_simd(v_vec, WV, ln1_q8, H_kv * head_dim, aligned_embed, ith, nth);
321 
322  #pragma omp barrier
323 
324  /* Step 3: RoPE, attention, etc. would go here (single-threaded for now) */
325  #pragma omp single
326  {
327  /* Copy K/V to cache */
328  const int kv_head_stride = max_seq * aligned_head;
329  for (int h = 0; h < H_kv; h++) {
330  memcpy(k_cache + h * kv_head_stride + token_index * aligned_head,
331  k_vec + h * head_dim, head_dim * sizeof(float));
332  memcpy(v_cache + h * kv_head_stride + token_index * aligned_head,
333  v_vec + h * head_dim, head_dim * sizeof(float));
334  }
335 
336  /* TODO: RoPE, attention decode, etc. */
337  /* For now, just copy q_vec to attn_out as placeholder */
338  memcpy(attn_out, q_vec, H * head_dim * sizeof(float));
339  }
340 
341  #pragma omp barrier
342 
343  /* Step 4: Output projection (parallel) */
344  /* First quantize attention output */
345  #pragma omp single
346  {
347  quantize_row_q8_k(attn_out, ln1_q8, H * aligned_head); /* Reuse ln1_q8 */
348  }
349 
350  gemv_q4_k_q8_k_parallel_simd(o_out, WO, ln1_q8, aligned_embed, H * head_dim, ith, nth);
351 
352  #pragma omp barrier
353 
354  /* Step 5: Residual add (parallel) */
355  residual_add_parallel(hidden, o_out, hidden, embed_dim, ith, nth);
356 
357  #pragma omp barrier
358 
359  /* ============================================================
360  * MLP BLOCK
361  * ============================================================ */
362 
363  /* Step 6: RMSNorm */
364  #pragma omp single
365  {
366  rmsnorm(hidden, (const float *)ln2_weight, ln2_out, embed_dim, eps);
367  quantize_row_q8_k(ln2_out, ln2_q8, aligned_embed);
368  }
369 
370  #pragma omp barrier
371 
372  /* Step 7: Gate and Up projections (parallel) */
373  gemv_q4_k_q8_k_parallel_simd(gate_buf, W_gate, ln2_q8, aligned_inter, aligned_embed, ith, nth);
374  gemv_q4_k_q8_k_parallel_simd(up_buf, W_up, ln2_q8, aligned_inter, aligned_embed, ith, nth);
375 
376  #pragma omp barrier
377 
378  /* Step 8: SwiGLU (parallel element-wise) */
379  const int dr = (intermediate + nth - 1) / nth;
380  const int r0 = dr * ith;
381  const int r1 = (r0 + dr < intermediate) ? (r0 + dr) : intermediate;
382 
383  for (int i = r0; i < r1; i++) {
384  float g = gate_buf[i];
385  float silu_g = g / (1.0f + expf(-g));
386  swiglu_buf[i] = silu_g * up_buf[i];
387  }
388 
389  #pragma omp barrier
390 
391  /* Step 9: Down projection */
392  #pragma omp single
393  {
394  quantize_row_q8_k(swiglu_buf, down_q8, aligned_inter);
395  }
396 
397  gemv_q4_k_q8_k_parallel_simd(mlp_out, W_down, down_q8, aligned_embed, aligned_inter, ith, nth);
398 
399  #pragma omp barrier
400 
401  /* Step 10: Final residual add */
402  residual_add_parallel(hidden, mlp_out, hidden, embed_dim, ith, nth);
403  }
404 }

410 /* Optimal thread count for decode on memory-bound systems */
411 int get_optimal_decode_threads(void)
412 {
413  int max_threads = omp_get_max_threads();
414 
415  /* For memory-bound workloads, 4 threads is often optimal.
416  * More threads hit memory bandwidth limits with diminishing returns.
417  * See MEMORY_BANDWIDTH_ANALYSIS.md for details. */
418  if (max_threads >= 4) {
419  return 4;
420  }
421  return max_threads;
422 }
