📖 How to Generate Parity Report

Option 1: Check CKE Only (No llama.cpp Needed)

# Enable parity dumping in generated code
python version/v6.6/scripts/codegen_v6_6.py \
    --decode lowered_decode_call.json \
    --init init_call.json \
    -o model_parity.c \
    --parity-dump

# Compile and run
gcc -o model_parity model_parity.c -lckernel_engine -lm -DCK_PARITY_DUMP
./model_parity model.gguf "Hello" 5

# Generate report
python version/v6.6/scripts/parity_report.py \
    --ck-dump ck_parity_dumps/dump.bin \
    --output report.json

Option 2: Full Comparison with llama.cpp

# Run llama.cpp with dump (requires patch)
./llama-cli -m model.gguf -p "Hello" -n 5 --parity-dump

# Run CKE with dump
./ck_model model.gguf "Hello" 5

# Compare
python version/v6.6/scripts/parity_report.py \
    --ck-dump ck_parity_dumps/dump.bin \
    --ref-dump llama_parity_dumps/dump.bin \
    --manifest weights_manifest.json \
    --output report.json

Option 3: Quick Check via ck-cli

# ck-cli has built-in parity mode
./build/ck-cli-v6.6 --model Qwen3-0.6B-GGUF --parity-dump --prompt "Hello"
