1 /**
2  * @file gemm_kernels_q4_1.c
3  * @brief GEMM/GEMV kernels with Q4_1 quantized weights
4  *
5  * Q4_1 Format:
6  * - 32 weights per block
7  * - 1 FP16 scale (d) per block
8  * - 1 FP16 minimum (m) per block
9  * - 20 bytes per 32 weights = 5.0 bits/weight
10  *
11  * Dequantization: w = d * q + m
12  * where q is the 4-bit unsigned value (0-15)
13  *
14  * Operations:
15  * Forward: Y = W @ X (W is Q4_1, X and Y are FP32)
16  * Backward: dX = W^T @ dY (gradient w.r.t. input)
17  */

32 /**
33  * @brief Matrix-vector multiply with Q4_1 weights (scalar reference)
34  *
35  * @param y Output vector [M]
36  * @param W Weight matrix in Q4_1 format [M x K]
37  * @param x Input vector [K]
38  * @param M Number of output rows
39  * @param K Number of columns (must be multiple of 32)
40  */
41 void gemv_q4_1_ref(float *y,
42  const void *W,
43  const float *x,
44  int M, int K)
45 {
46  const block_q4_1 *blocks = (const block_q4_1 *)W;
47  const int blocks_per_row = K / QK4_1;
48 
49  for (int row = 0; row < M; row++) {
50  float sum = 0.0f;
51 
52  for (int b = 0; b < blocks_per_row; b++) {
53  const block_q4_1 *block = &blocks[row * blocks_per_row + b];
54  const float d = CK_FP16_TO_FP32(block->d);
55  const float m = CK_FP16_TO_FP32(block->m);
56  const float *xp = &x[b * QK4_1];
57 
58  for (int i = 0; i < QK4_1 / 2; i++) {
59  const uint8_t packed = block->qs[i];
60  const int q0 = (packed & 0x0F);
61  const int q1 = (packed >> 4);
62 
63  /* Dequantize: w = d * q + m */
64  const float w0 = d * (float)q0 + m;
65  const float w1 = d * (float)q1 + m;
66 
67  sum += w0 * xp[2*i + 0];
68  sum += w1 * xp[2*i + 1];
69  }
70  }
71 
72  y[row] = sum;
73  }
74 }

77 /**
78  * @brief Matrix-vector multiply with Q4_1 weights (AVX-512)
79  */
80 void gemv_q4_1_avx512(float *y,
81  const void *W,
82  const float *x,
83  int M, int K)
84 {
85  const block_q4_1 *blocks = (const block_q4_1 *)W;
86  const int blocks_per_row = K / QK4_1;
87  const __m512i mask_lo = _mm512_set1_epi32(0x0F);
88 
89  for (int row = 0; row < M; row++) {
90  __m512 acc = _mm512_setzero_ps();
91 
92  for (int b = 0; b < blocks_per_row; b++) {
93  const block_q4_1 *block = &blocks[row * blocks_per_row + b];
94  const __m512 vscale = _mm512_set1_ps(CK_FP16_TO_FP32(block->d));
95  const __m512 vmin = _mm512_set1_ps(CK_FP16_TO_FP32(block->m));
96  const float *xp = &x[b * QK4_1];
97 
98  /* Load 16 bytes = 32 x 4-bit weights */
99  __m128i packed = _mm_loadu_si128((const __m128i *)block->qs);
100  __m512i bytes = _mm512_cvtepu8_epi32(packed);
101 
102  /* Extract nibbles */
103  __m512i lo = _mm512_and_epi32(bytes, mask_lo);
104  __m512i hi = _mm512_srli_epi32(bytes, 4);
105 
106  /* Dequantize: w = d * q + m */
107  __m512 w_lo = _mm512_fmadd_ps(_mm512_cvtepi32_ps(lo), vscale, vmin);
108  __m512 w_hi = _mm512_fmadd_ps(_mm512_cvtepi32_ps(hi), vscale, vmin);
109 
110  /* Load interleaved input */
111  __m512 x_even = _mm512_set_ps(
112  xp[30], xp[28], xp[26], xp[24], xp[22], xp[20], xp[18], xp[16],
113  xp[14], xp[12], xp[10], xp[8], xp[6], xp[4], xp[2], xp[0]);
114  __m512 x_odd = _mm512_set_ps(
115  xp[31], xp[29], xp[27], xp[25], xp[23], xp[21], xp[19], xp[17],
116  xp[15], xp[13], xp[11], xp[9], xp[7], xp[5], xp[3], xp[1]);
117 
118  acc = _mm512_fmadd_ps(w_lo, x_even, acc);
119  acc = _mm512_fmadd_ps(w_hi, x_odd, acc);
120  }
121 
122  y[row] = _mm512_reduce_add_ps(acc);
123  }
124 }

163 /**
164  * @brief Backward pass: compute input gradient
165  *
166  * @param dX Output gradient w.r.t. input [K]
167  * @param W Weight matrix in Q4_1 format [M x K]
168  * @param dY Gradient w.r.t. output [M]
169  * @param M Number of output rows
170  * @param K Number of columns (input dimension)
171  */
172 void gemv_q4_1_backward_ref(float *dX,
173  const void *W,
174  const float *dY,
175  int M, int K)
176 {
177  const block_q4_1 *blocks = (const block_q4_1 *)W;
178  const int blocks_per_row = K / QK4_1;
179 
180  /* Zero output gradient */
181  memset(dX, 0, K * sizeof(float));
182 
183  /* Accumulate: dX += W^T @ dY */
184  for (int row = 0; row < M; row++) {
185  const float dy = dY[row];
186 
187  for (int b = 0; b < blocks_per_row; b++) {
188  const block_q4_1 *block = &blocks[row * blocks_per_row + b];
189  const float d = CK_FP16_TO_FP32(block->d);
190  const float m = CK_FP16_TO_FP32(block->m);
191  float *dxp = &dX[b * QK4_1];
192 
193  for (int i = 0; i < QK4_1 / 2; i++) {
194  const uint8_t packed = block->qs[i];
195  const int q0 = (packed & 0x0F);
196  const int q1 = (packed >> 4);
197 
198  const float w0 = d * (float)q0 + m;
199  const float w1 = d * (float)q1 + m;
200 
201  dxp[2*i + 0] += w0 * dy;
202  dxp[2*i + 1] += w1 * dy;
203  }
204  }
205  }

236 /**
237  * @brief GEMM with transposed Q4_1 weights: C = A @ B^T
238  *
239  * @param A Input activations [M x K], row-major FP32
240  * @param B Weight matrix in Q4_1 format [N x K], row-major quantized
241  * @param bias Optional bias [N], NULL if not used
242  * @param C Output [M x N], row-major FP32
243  * @param M Batch size (number of tokens)
244  * @param N Output dimension
245  * @param K Input dimension
246  */
247 void gemm_nt_q4_1(const float *A,
248  const void *B,
249  const float *bias,
250  float *C,
251  int M, int N, int K)
252 {
253  const block_q4_1 *blocks = (const block_q4_1 *)B;
254  const int blocks_per_row = K / QK4_1;
255 
256  for (int m = 0; m < M; m++) {
257  const float *a_row = &A[m * K];
258 
259  for (int n = 0; n < N; n++) {
260  float sum = 0.0f;
261 
262  for (int b = 0; b < blocks_per_row; b++) {
263  const block_q4_1 *block = &blocks[n * blocks_per_row + b];
264  const float d = CK_FP16_TO_FP32(block->d);
265  const float min = CK_FP16_TO_FP32(block->m);
266  const float *ap = &a_row[b * QK4_1];
267 
268  for (int i = 0; i < QK4_1 / 2; i++) {
269  const uint8_t packed = block->qs[i];
270  const int q0 = (packed & 0x0F);
271  const int q1 = (packed >> 4);
272 
273  const float w0 = d * (float)q0 + min;
274  const float w1 = d * (float)q1 + min;
275 
276  sum += w0 * ap[2 * i + 0];
277  sum += w1 * ap[2 * i + 1];
278  }
279  }
280 
281  C[m * N + n] = sum + (bias ? bias[n] : 0.0f);
282  }
283  }
