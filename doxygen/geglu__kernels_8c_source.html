16static inline float ck_gelu_tanh_parity_f32(float x)
17{
18 const float sqrt_2_over_pi = 0.7978845608f;
19 const float coeff = 0.044715f;
20 const float x3 = x * x * x;
21 const float inner = sqrt_2_over_pi * (x + coeff * x3);
22 return 0.5f * x * (1.0f + tanhf(inner));
23}

25void geglu_forward_exact(const float *x, float *out, int tokens, int dim)
26{

82void geglu_backward_fp32(const float *x,
83 const float *d_out,
84 float *d_x,
85 int tokens,
86 int dim)
87{
88 if (!x || !d_out || !d_x || tokens <= 0 || dim <= 0) {
89 return;
90 }
91
92 const float sqrt_2_over_pi = 0.7978845608f;
93 const float coeff = 0.044715f;
94 const int inner_dim = dim * 2;
95
96 for (int t = 0; t < tokens; ++t) {
97 const float *x_ptr = x + (size_t)t * (size_t)inner_dim;
98 const float *d_out_ptr = d_out + (size_t)t * (size_t)dim;
99 float *d_x_ptr = d_x + (size_t)t * (size_t)inner_dim;
100
101 for (int d = 0; d < dim; ++d) {
102 float a = x_ptr[d];
103 float b = x_ptr[dim + d];
104 float dout = d_out_ptr[d];
105
106 float a2 = a * a;
107 float a3 = a2 * a;
108 float g = sqrt_2_over_pi * (a + coeff * a3);
109 float tanh_g = tanhf(g);
110 float sech2_g = 1.0f - tanh_g * tanh_g;
111 float g_prime = sqrt_2_over_pi * (1.0f + 3.0f * coeff * a2);
112
113 float d_gelu = 0.5f * (1.0f + tanh_g) + 0.5f * a * sech2_g * g_prime;
114 d_x_ptr[d] = dout * d_gelu * b;
115
116 float gelu_a = 0.5f * a * (1.0f + tanh_g);
117 d_x_ptr[dim + d] = dout * gelu_a;
118 }
119 }
