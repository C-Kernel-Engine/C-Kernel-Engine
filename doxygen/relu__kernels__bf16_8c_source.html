23void relu_forward_bf16(const uint16_t *input, uint16_t *output, size_t n)
24{
25 if (!input || !output) {
26 return;
27 }
28 for (size_t i = 0; i < n; ++i) {
29 float x = bf16_to_float(input[i]);

31 }
32}
