21static inline int8_t decode_int4(uint8_t packed, int index)
22{
23 int8_t nibble;
24 if ((index & 1) == 0) {
25 nibble = packed & 0x0F;
26 } else {
27 nibble = (packed >> 4) & 0x0F;
28 }
29 if (nibble >= 8) {
30 nibble -= 16;
31 }
32 return nibble;
33}

35static inline uint8_t encode_int4_nibble(int8_t value)
36{
37 if (value > 7) {
38 value = 7;
39 } else if (value < -8) {
40 value = -8;
41 }
42 return (uint8_t)(value & 0x0F);
43}

45static void convert_int4_to_float(const uint8_t *src,
46 float *dst,
47 size_t count)
48{
49 for (size_t i = 0; i < count; ++i) {
50 uint8_t packed = src[i >> 1];
51 dst[i] = (float)decode_int4(packed, (int)(i & 1));
52 }
53}

55static void convert_float_to_int4(const float *src,
56 uint8_t *dst,
57 size_t count)
58{
59 size_t bytes = (count + 1) / 2;
60 for (size_t i = 0; i < bytes; ++i) {
61 dst[i] = 0;
62 }
63 for (size_t i = 0; i < count; ++i) {
64 uint8_t quant = encode_int4_nibble((int8_t)lrintf(src[i]));
65 size_t byte_idx = i >> 1;
66 if ((i & 1) == 0) {
67 dst[byte_idx] = (dst[byte_idx] & 0xF0) | quant;
68 } else {
69 dst[byte_idx] = (dst[byte_idx] & 0x0F) | (quant << 4);
70 }
71 }
72}
