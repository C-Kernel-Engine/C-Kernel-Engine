25/* AVX1 horizontal sum helper (no _mm256_reduce_add_ps in AVX1) */
26#if defined(__AVX__) && !defined(__AVX512F__)
27static inline float hsum256_ps_rmsnorm(__m256 v) {
28 // Sum upper and lower 128-bit lanes
29 __m128 hi = _mm256_extractf128_ps(v, 1);
30 __m128 lo = _mm256_castps256_ps128(v);
31 __m128 sum128 = _mm_add_ps(lo, hi);
32 // Horizontal add within 128-bit lane
33 sum128 = _mm_hadd_ps(sum128, sum128);
34 sum128 = _mm_hadd_ps(sum128, sum128);
35 return _mm_cvtss_f32(sum128);
36}
37#endif
38static void rmsnorm_forward_strict_scalar(const float *input,
39 const float *gamma,
40 float *output,
41 float *rstd_cache,
42 int tokens,
43 int d_model,
44 int aligned_embed_dim,
45 float eps)
46{
47 const float inv_d = 1.0f / (float)d_model;
48 for (int t = 0; t < tokens; ++t) {
49 const float *x = input + (size_t)t * (size_t)aligned_embed_dim;
50 float *y = output + (size_t)t * (size_t)aligned_embed_dim;
51
52 float sum_sq = 0.0f;
53 for (int d = 0; d < d_model; ++d) {
54 const float v = x[d];
55 sum_sq += v * v;
56 }
57 const float mean_sq = sum_sq * inv_d;
58 const float rstd = 1.0f / sqrtf(mean_sq + eps);
59 if (rstd_cache) {
60 rstd_cache[t] = rstd;
61 }
62
63 for (int d = 0; d < d_model; ++d) {
64 const float x_hat = x[d] * rstd;
65 y[d] = x_hat * gamma[d];
66 }
67 for (int d = d_model; d < aligned_embed_dim; ++d) {
68 y[d] = 0.0f;
69 }
70 }
71}

73static void rmsnorm_backward_strict_scalar(const float *d_output,
74 const float *input,
75 const float *gamma,
76 const float *rstd_cache,
77 float *d_input,
78 float *d_gamma,
79 int tokens,
80 int d_model,
81 int aligned_embed_dim)
82{
83 const float inv_d = 1.0f / (float)d_model;
84 for (int d = 0; d < d_model; ++d) {
85 d_gamma[d] = 0.0f;
86 }
87
88 for (int t = 0; t < tokens; ++t) {
89 const float *x = input + (size_t)t * (size_t)aligned_embed_dim;
90 const float *dY = d_output + (size_t)t * (size_t)aligned_embed_dim;
91 float *dX = d_input + (size_t)t * (size_t)aligned_embed_dim;
92 const float rstd = rstd_cache[t];
93
94 float sum_dY_g_xhat = 0.0f;
95 for (int d = 0; d < d_model; ++d) {
96 const float x_hat = x[d] * rstd;
97 const float grad_x_hat = dY[d] * gamma[d];
98 sum_dY_g_xhat += x_hat * grad_x_hat;
99 }
100
101 for (int d = 0; d < d_model; ++d) {
102 const float x_hat = x[d] * rstd;
103 const float grad_x_hat = dY[d] * gamma[d];
104 dX[d] = (grad_x_hat - (x_hat * inv_d) * sum_dY_g_xhat) * rstd;
105 d_gamma[d] += dY[d] * x_hat;
106 }
107 for (int d = d_model; d < aligned_embed_dim; ++d) {
108 dX[d] = 0.0f;
109 }
110 }
111}

114/**
115 * RMSNorm forward pass
116 * @test test_rmsnorm.py::TestRMSNormForward::test_fp32_tokens
117 * @test test_rmsnorm.py::TestRMSNormForward::test_fp32_single
118 * @test test_rmsnorm.py::TestRMSNormForward::test_perf_rolled
119 * @test test_layernorm.py::TestLayerNormForward::test_rmsnorm_compat
120 * @test test_parity.py::test_rmsnorm_parity
121 *
122 * RMSNorm: y[i] = gamma[i] * x[i] / sqrt(mean(x^2) + eps)
123 *
124 * After changes: make test && make llamacpp-parity-full
125 */
126void rmsnorm_forward(const float *input,
127 const float *gamma,
128 float *output,
129 float *rstd_cache,
130 int tokens,
131 int d_model,
132 int aligned_embed_dim,
133 float eps)
134{
135 int T = tokens;
136 int D = d_model;
137 int aligned = aligned_embed_dim;
138
139 if (ck_strict_parity_enabled()) {
140 rmsnorm_forward_strict_scalar(input, gamma, output, rstd_cache, T, D, aligned, eps);
141 return;
142 }
143
144 for (int t = 0; t < T; ++t) {
145 const float *x = input + (size_t)t * aligned;
146 float *y = output + (size_t)t * aligned;
147
148#if defined(__AVX512F__)
149 // AVX-512: Process 16 floats at a time
150 __m512 sum_sq_vec = _mm512_setzero_ps();
151 int d = 0;
152
153 // Vectorized sum of squares
154 for (; d + 16 <= D; d += 16) {
155 __m512 xv = _mm512_loadu_ps(&x[d]);
156 sum_sq_vec = _mm512_fmadd_ps(xv, xv, sum_sq_vec);
157 }
158 float sum_sq = _mm512_reduce_add_ps(sum_sq_vec);
159
160 // Handle remaining elements
161 for (; d < D; ++d) {
162 sum_sq += x[d] * x[d];
163 }
164
165 float mean_sq = sum_sq / (float)D;
166 float rstd = 1.0f / sqrtf(mean_sq + eps);
167 if (rstd_cache) {
168 rstd_cache[t] = rstd;
169 }
170
171 // Apply normalization and scale (vectorized)
172 __m512 rstd_vec = _mm512_set1_ps(rstd);
173 d = 0;
174 for (; d + 16 <= D; d += 16) {
175 __m512 xv = _mm512_loadu_ps(&x[d]);
176 __m512 gv = _mm512_loadu_ps(&gamma[d]);
177 __m512 x_hat = _mm512_mul_ps(xv, rstd_vec);
178 __m512 yv = _mm512_mul_ps(x_hat, gv);
179 _mm512_storeu_ps(&y[d], yv);
180 }
181 // Handle remaining elements
182 for (; d < D; ++d) {
183 y[d] = x[d] * rstd * gamma[d];
184 }
185
186#elif defined(__AVX__)
187 // AVX: Process 8 floats at a time
188 __m256 sum_sq_vec = _mm256_setzero_ps();
189 int d = 0;
190
191 // Vectorized sum of squares (no FMA in AVX1, use mul + add)
192 for (; d + 8 <= D; d += 8) {
193 __m256 xv = _mm256_loadu_ps(&x[d]);
194 __m256 xv_sq = _mm256_mul_ps(xv, xv);
195 sum_sq_vec = _mm256_add_ps(sum_sq_vec, xv_sq);
196 }
197 float sum_sq = hsum256_ps_rmsnorm(sum_sq_vec);
198
199 // Handle remaining elements
200 for (; d < D; ++d) {
201 sum_sq += x[d] * x[d];
202 }
203
204 float mean_sq = sum_sq / (float)D;
205 float rstd = 1.0f / sqrtf(mean_sq + eps);
206 if (rstd_cache) {
207 rstd_cache[t] = rstd;
208 }
209
210 // Apply normalization and scale (vectorized)
211 __m256 rstd_vec = _mm256_set1_ps(rstd);
212 d = 0;
213 for (; d + 8 <= D; d += 8) {
214 __m256 xv = _mm256_loadu_ps(&x[d]);
215 __m256 gv = _mm256_loadu_ps(&gamma[d]);
216 __m256 x_hat = _mm256_mul_ps(xv, rstd_vec);
217 __m256 yv = _mm256_mul_ps(x_hat, gv);
218 _mm256_storeu_ps(&y[d], yv);
219 }
220 // Handle remaining elements
221 for (; d < D; ++d) {
222 y[d] = x[d] * rstd * gamma[d];
223 }
224
225#else
226 // Scalar fallback
227 float sum_sq = 0.0f;
228 for (int d = 0; d < D; ++d) {
229 float v = x[d];
230 sum_sq += v * v;
231 }
232 float mean_sq = sum_sq / (float)D;
233 float rstd = 1.0f / sqrtf(mean_sq + eps);
234 if (rstd_cache) {
235 rstd_cache[t] = rstd;
236 }
237
238 // Apply normalization and scale
239 for (int d = 0; d < D; ++d) {
240 float x_hat = x[d] * rstd;
241 y[d] = x_hat * gamma[d];
242 }
243#endif
244
245 // Zero padding (if any)
246 for (int d = D; d < aligned; ++d) {
247 y[d] = 0.0f;
248 }
249 }
250}

252/**
253 * RMSNorm backward pass
254 * @test test_rmsnorm.py::TestRMSNormBackward::test_backward_tokens
255 * @test test_rmsnorm.py::TestRMSNormBackward::test_backward_single
256 * @test test_parity.py::test_rmsnorm_backward_parity
257 *
258 * Computes dX and dGamma given dY, X, gamma, and cached rstd.
259 * dX_i = rstd * (dY_i * gamma_i - x_hat_i * m)
260 * dGamma_i = sum_t (dY_i * x_hat_i)
261 *
262 * After changes: make test && make llamacpp-parity-full
263 */
264void rmsnorm_backward(const float *d_output,
265 const float *input,
266 const float *gamma,
267 const float *rstd_cache,
268 float *d_input,
269 float *d_gamma,
270 int tokens,
271 int d_model,
272 int aligned_embed_dim)
273{
274 int T = tokens;
275 int D = d_model;
276 int aligned = aligned_embed_dim;
277
278 if (ck_strict_parity_enabled()) {
279 rmsnorm_backward_strict_scalar(d_output, input, gamma, rstd_cache, d_input, d_gamma, T, D, aligned);
280 return;
281 }
282
283 // Zero parameter gradients
284#if defined(__AVX512F__)
285 {
286 int d = 0;
287 for (; d + 16 <= D; d += 16) {
288 _mm512_storeu_ps(&d_gamma[d], _mm512_setzero_ps());
289 }
290 for (; d < D; ++d) {
291 d_gamma[d] = 0.0f;
292 }
293 }
294#elif defined(__AVX__)
295 {
296 int d = 0;
297 for (; d + 8 <= D; d += 8) {
298 _mm256_storeu_ps(&d_gamma[d], _mm256_setzero_ps());
299 }
300 for (; d < D; ++d) {
301 d_gamma[d] = 0.0f;
302 }
303 }
304#else
305 for (int d = 0; d < D; ++d) {
306 d_gamma[d] = 0.0f;
307 }
308#endif
309
310 for (int t = 0; t < T; ++t) {
311 const float *x = input + (size_t)t * aligned;
312 const float *dY = d_output + (size_t)t * aligned;
313 float *dX = d_input + (size_t)t * aligned;
314
315 float rstd = rstd_cache[t];
316
317#if defined(__AVX512F__)
318 // Compute m = (1/D) * sum_j (dY_j * gamma_j * x_hat_j)
319 __m512 rstd_vec = _mm512_set1_ps(rstd);
320 __m512 sum_vec = _mm512_setzero_ps();
321 int d = 0;
322
323 for (; d + 16 <= D; d += 16) {
324 __m512 xv = _mm512_loadu_ps(&x[d]);
325 __m512 dyv = _mm512_loadu_ps(&dY[d]);
326 __m512 gv = _mm512_loadu_ps(&gamma[d]);
327 __m512 x_hat = _mm512_mul_ps(xv, rstd_vec);
328 // sum += dY * gamma * x_hat
329 __m512 prod = _mm512_mul_ps(dyv, gv);
330 sum_vec = _mm512_fmadd_ps(prod, x_hat, sum_vec);
331 }
332 float sum_dY_g_xhat = _mm512_reduce_add_ps(sum_vec);
333
334 // Handle remaining elements
335 for (; d < D; ++d) {
336 float x_hat = x[d] * rstd;
337 sum_dY_g_xhat += dY[d] * gamma[d] * x_hat;
338 }
339 float m = sum_dY_g_xhat / (float)D;
340
341 // Compute dX and accumulate dGamma (vectorized)
342 __m512 m_vec = _mm512_set1_ps(m);
343 d = 0;
344 for (; d + 16 <= D; d += 16) {
345 __m512 xv = _mm512_loadu_ps(&x[d]);
346 __m512 dyv = _mm512_loadu_ps(&dY[d]);
347 __m512 gv = _mm512_loadu_ps(&gamma[d]);
348 __m512 dgv = _mm512_loadu_ps(&d_gamma[d]);
349
350 __m512 x_hat = _mm512_mul_ps(xv, rstd_vec);
351
352 // dX = rstd * (dY * gamma - x_hat * m)
353 __m512 dy_g = _mm512_mul_ps(dyv, gv);
354 __m512 xhat_m = _mm512_mul_ps(x_hat, m_vec);
355 __m512 diff = _mm512_sub_ps(dy_g, xhat_m);
356 __m512 dxv = _mm512_mul_ps(rstd_vec, diff);
357 _mm512_storeu_ps(&dX[d], dxv);
358
359 // d_gamma += dY * x_hat
360 dgv = _mm512_fmadd_ps(dyv, x_hat, dgv);
361 _mm512_storeu_ps(&d_gamma[d], dgv);
362 }
363 // Handle remaining elements
364 for (; d < D; ++d) {
365 float x_hat = x[d] * rstd;
366 float dy = dY[d];
367 dX[d] = rstd * (dy * gamma[d] - x_hat * m);
368 d_gamma[d] += dy * x_hat;
369 }
370
371#elif defined(__AVX__)
372 // Compute m = (1/D) * sum_j (dY_j * gamma_j * x_hat_j)
373 __m256 rstd_vec = _mm256_set1_ps(rstd);
374 __m256 sum_vec = _mm256_setzero_ps();
375 int d = 0;
376
377 for (; d + 8 <= D; d += 8) {
378 __m256 xv = _mm256_loadu_ps(&x[d]);
379 __m256 dyv = _mm256_loadu_ps(&dY[d]);
380 __m256 gv = _mm256_loadu_ps(&gamma[d]);
381 __m256 x_hat = _mm256_mul_ps(xv, rstd_vec);
382 // sum += dY * gamma * x_hat (no FMA, use mul + mul + add)
383 __m256 prod = _mm256_mul_ps(dyv, gv);
384 __m256 prod2 = _mm256_mul_ps(prod, x_hat);
385 sum_vec = _mm256_add_ps(sum_vec, prod2);
386 }
387 float sum_dY_g_xhat = hsum256_ps_rmsnorm(sum_vec);
388
389 // Handle remaining elements
390 for (; d < D; ++d) {
391 float x_hat = x[d] * rstd;
392 sum_dY_g_xhat += dY[d] * gamma[d] * x_hat;
393 }
394 float m = sum_dY_g_xhat / (float)D;
395
396 // Compute dX and accumulate dGamma (vectorized)
397 __m256 m_vec = _mm256_set1_ps(m);
398 d = 0;
399 for (; d + 8 <= D; d += 8) {
400 __m256 xv = _mm256_loadu_ps(&x[d]);
401 __m256 dyv = _mm256_loadu_ps(&dY[d]);
402 __m256 gv = _mm256_loadu_ps(&gamma[d]);
403 __m256 dgv = _mm256_loadu_ps(&d_gamma[d]);
404
405 __m256 x_hat = _mm256_mul_ps(xv, rstd_vec);
406
407 // dX = rstd * (dY * gamma - x_hat * m)
408 __m256 dy_g = _mm256_mul_ps(dyv, gv);
409 __m256 xhat_m = _mm256_mul_ps(x_hat, m_vec);
410 __m256 diff = _mm256_sub_ps(dy_g, xhat_m);
411 __m256 dxv = _mm256_mul_ps(rstd_vec, diff);
412 _mm256_storeu_ps(&dX[d], dxv);
413
414 // d_gamma += dY * x_hat
415 __m256 dy_xhat = _mm256_mul_ps(dyv, x_hat);
416 dgv = _mm256_add_ps(dgv, dy_xhat);
417 _mm256_storeu_ps(&d_gamma[d], dgv);
418 }
419 // Handle remaining elements
420 for (; d < D; ++d) {
421 float x_hat = x[d] * rstd;
422 float dy = dY[d];
423 dX[d] = rstd * (dy * gamma[d] - x_hat * m);
424 d_gamma[d] += dy * x_hat;
425 }
426
427#else
428 // Scalar fallback
429 // Compute m = (1/D) * sum_j (dY_j * gamma_j * x_hat_j)
430 float sum_dY_g_xhat = 0.0f;
431 for (int d = 0; d < D; ++d) {
432 float x_hat = x[d] * rstd;
433 sum_dY_g_xhat += dY[d] * gamma[d] * x_hat;
434 }
435 float m = sum_dY_g_xhat / (float)D;
436
437 // Compute dX and accumulate dGamma
438 for (int d = 0; d < D; ++d) {
439 float x_hat = x[d] * rstd;
440 float dy = dY[d];
441 dX[d] = rstd * (dy * gamma[d] - x_hat * m);
442 d_gamma[d] += dy * x_hat;
443 }
444#endif
445
446 // Zero padding gradients (if any)
447 for (int d = D; d < aligned; ++d) {
448 dX[d] = 0.0f;
449 }
450 }
