45static inline int ck_nearest_int(float fval) {
46 /* Bit-level round-to-nearest from llama.cpp (fast + deterministic). */
47 float val = fval + 12582912.f;
48 int i;
49 memcpy(&i, &val, sizeof(int));
50 return (i & 0x007fffff) - 0x00400000;
51}

53void quantize_row_q8_k_ref(const float *x, void *vy, int k) {
54 if (!x || !vy || k <= 0) {
55 return;
56 }
57 assert(k % QK_K == 0);
58 const int nb = k / QK_K;
59 block_q8_K *y = (block_q8_K *)vy;
60
61 for (int i = 0; i < nb; ++i) {
62 float max = 0.0f;
63 float amax = 0.0f;
64 for (int j = 0; j < QK_K; ++j) {
65 float ax = fabsf(x[j]);
66 if (ax > amax) {
67 amax = ax;
68 max = x[j];
69 }
70 }
71 if (!amax) {
72 y[i].d = 0.0f;
73 memset(y[i].qs, 0, sizeof(y[i].qs));
74 memset(y[i].bsums, 0, sizeof(y[i].bsums));
75 x += QK_K;
76 continue;
77 }
78
79 const float iscale = -127.0f / max;
80 for (int j = 0; j < QK_K; ++j) {
81 int v = ck_nearest_int(iscale * x[j]);
82 if (v > 127) {
83 v = 127;
84 }
85 if (v < -128) {
86 v = -128;
87 }
88 y[i].qs[j] = (int8_t)v;
89 }
90
91 for (int j = 0; j < QK_K / 16; ++j) {
92 int sum = 0;
93 const int8_t *qs = &y[i].qs[j * 16];
94 for (int ii = 0; ii < 16; ++ii) {
95 sum += qs[ii];
96 }
97 y[i].bsums[j] = (int16_t)sum;
98 }
99
100 y[i].d = 1.0f / iscale;
101 x += QK_K;
102 }
103}

121static float dot_q4_k_q8_k_ref(const block_q4_K *w,
122 const block_q8_K *x,
123 int k)
124{
125 const int nb = k / QK_K;
126 float sumf = 0.0f;
127
128 for (int i = 0; i < nb; ++i) {
129 uint8_t sc[8], m_val[8];
130 unpack_q4_k_scales(w[i].scales, sc, m_val);
131
132 const float d = CK_FP16_TO_FP32(w[i].d) * x[i].d;
133 const float dmin = CK_FP16_TO_FP32(w[i].dmin) * x[i].d;
134
135 /* Q4_K layout: process 64 elements at a time
136 * - Low nibbles of qs[0..31] → elements 0..31 → uses sc[0], m[0]
137 * - High nibbles of qs[0..31] → elements 32..63 → uses sc[1], m[1]
138 * - Low nibbles of qs[32..63] → elements 64..95 → uses sc[2], m[2]
139 * - etc.
140 */
141 int is = 0;
142 int q_offset = 0;
143
144 for (int j = 0; j < QK_K; j += 64) {
145 const uint8_t *qs = &w[i].qs[q_offset];
146 const int8_t *q8_lo = &x[i].qs[j]; /* Elements j to j+31 */
147 const int8_t *q8_hi = &x[i].qs[j + 32]; /* Elements j+32 to j+63 */
148
149 /* Sum for low nibbles (elements j to j+31) */
150 int32_t sum_q4q8_lo = 0;
151 for (int l = 0; l < 32; ++l) {
152 int q4_val = qs[l] & 0x0F;
153 sum_q4q8_lo += q4_val * q8_lo[l];
154 }
155
156 /* Sum for high nibbles (elements j+32 to j+63) */
157 int32_t sum_q4q8_hi = 0;
158 for (int l = 0; l < 32; ++l) {
159 int q4_val = qs[l] >> 4;
160 sum_q4q8_hi += q4_val * q8_hi[l];
161 }
162
163 /* bsums: each bsum is 16 elements */
164 int32_t bsum_lo = (int32_t)x[i].bsums[j / 16] +
165 (int32_t)x[i].bsums[j / 16 + 1];
166 int32_t bsum_hi = (int32_t)x[i].bsums[(j + 32) / 16] +
167 (int32_t)x[i].bsums[(j + 32) / 16 + 1];
168
169 /* Accumulate: d * sc * sum(q4*q8) - dmin * m * sum(q8) */
170 sumf += d * (float)sc[is] * (float)sum_q4q8_lo;
171 sumf -= dmin * (float)m_val[is] * (float)bsum_lo;
172 sumf += d * (float)sc[is + 1] * (float)sum_q4q8_hi;
173 sumf -= dmin * (float)m_val[is + 1] * (float)bsum_hi;
174
175 q_offset += 32;
176 is += 2;
177 }
178 }
179
180 return sumf;
181}

183void gemv_q4_k_q8_k_ref(float *y,
184 const void *W,
185 const void *x_q8,
186 int M, int K)
187{
188 if (!y || !W || !x_q8 || M <= 0 || K <= 0) {
189 return;
190 }
191
192 const block_q4_K *blocks = (const block_q4_K *)W;
193 const block_q8_K *x = (const block_q8_K *)x_q8;
194 const int blocks_per_row = K / QK_K;
195
196 for (int row = 0; row < M; ++row) {
197 const block_q4_K *w_row = blocks + (size_t)row * (size_t)blocks_per_row;
198 y[row] = dot_q4_k_q8_k_ref(w_row, x, K);
199 }
200}

202/* ============================================================================
203 * PARALLEL VERSIONS (for parallel orchestration)
204 *
205 * These receive ith (thread index) and nth (total threads) from orchestration.
206 * OpenMP lives in orchestration layer, NOT here.
207 *
208 * Naming: *_parallel = receives ith/nth, processes only its portion
209 * *_ref/_avx = single-threaded, processes all rows
210 * ============================================================================ */
211
212void gemv_q4_k_q8_k_parallel(float *y,
213 const void *W,
214 const void *x_q8,
215 int M, int K,
216 int ith, int nth)
217{
218 if (!y || !W || !x_q8 || M <= 0 || K <= 0) {
219 return;
220 }
221 if (ith < 0 || nth <= 0 || ith >= nth) {
222 return;
223 }
224
225 /* Compute row range for this thread */
226 const int dr = (M + nth - 1) / nth;
227 const int r0 = dr * ith;
228 const int r1 = (r0 + dr < M) ? (r0 + dr) : M;
229
230 if (r0 >= M) {
231 return; /* This thread has no work */
232 }
233
234 const block_q4_K *blocks = (const block_q4_K *)W;
235 const block_q8_K *x = (const block_q8_K *)x_q8;
236 const int blocks_per_row = K / QK_K;
237
238 /* Only process rows [r0, r1) */
239 for (int row = r0; row < r1; ++row) {
240 const block_q4_K *w_row = blocks + (size_t)row * (size_t)blocks_per_row;
241 y[row] = dot_q4_k_q8_k_ref(w_row, x, K);
242 }
243}
