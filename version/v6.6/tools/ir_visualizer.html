Workflows Common Debugging Workflows

New Parity Issue

# 1. Find first failure
python test_layer_by_layer.py --token 25

# 2. Debug at that layer
python v6_6_comprehensive_debug.py --stop-at LAYER

# 3. Trace divergence
python trace_divergence.py --token 25

NaN Detected

# 1. Find layer with NaN
python trace_nan_layer.py --token 25

# 2. Find exact op with NaN
python trace_nan_step.py --token 25

# 3. Check weight integrity
python test_weight_offset_consistency.py

Memory Corruption

# 1. Quick memory check
python test_memory_planner.py --layout=...

# 2. Deep validation
python advanced_memory_validator.py --deep

# 3. BUMP sync check
python test_bump_layout_sync.py

Converter Bug

# 1. Check weight offsets
python test_weight_offset_consistency.py

# 2. Verify dtype consistency
python test_dtype_consistency.py

# 3. GEMV specific test
python test_gemv_q8_0_q8_0.py
