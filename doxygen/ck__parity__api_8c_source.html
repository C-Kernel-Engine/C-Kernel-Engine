411void ck_test_gemm_q4_k(const void *weight_q4k,
412 const float *input_f32,
413 float *output,
414 int rows, int cols, int n_tokens)
415{
416 /* Allocate Q8_K buffer for quantized activations */
417 int n_blocks_per_row = cols / CK_QK_K;
418 block_q8_K *q8_data = (block_q8_K *)malloc(n_tokens * n_blocks_per_row * sizeof(block_q8_K));
419 if (!q8_data) {
420 memset(output, 0, n_tokens * rows * sizeof(float));
421 return;
422 }
423
424 /* Quantize all input tokens */
425 for (int t = 0; t < n_tokens; t++) {
426 quantize_row_q8_k(input_f32 + t * cols,
427 q8_data + t * n_blocks_per_row, cols);
428 }
429
430 /* Use gemm_nt_q4_k_q8_k: C[M,N] = A[M,K] * B[N,K]^T
431 * Our layout: output[n_tokens, rows] = input[n_tokens, cols] * weight[rows, cols]^T
432 * So: M = n_tokens, N = rows, K = cols
433 */
434 gemm_nt_q4_k_q8_k(q8_data, weight_q4k, NULL, output, n_tokens, rows, cols);
435
436 free(q8_data);
437}

552void ck_test_rope(float *q, float *k,
553 int n_tokens, int n_heads, int n_heads_kv, int head_dim,
554 int pos_offset, float theta)
555{
556 /* Precompute cos/sin cache */
557 int half_dim = head_dim / 2;
558 int max_seq = pos_offset + n_tokens;
559
560 float *cos_cache = (float *)malloc(max_seq * half_dim * sizeof(float));
561 float *sin_cache = (float *)malloc(max_seq * half_dim * sizeof(float));
562 if (!cos_cache || !sin_cache) {
563 free(cos_cache);
564 free(sin_cache);
565 return;
566 }
567
568 rope_precompute_cache(cos_cache, sin_cache, max_seq, head_dim, theta,
569 head_dim, "none", 1.0f);
570
571 /* CK RoPE expects layout [num_heads, num_tokens, head_dim]
572 * Reshape from [n_tokens, n_heads * head_dim] to [n_heads, n_tokens, head_dim]
573 */
574 float *q_reorder = (float *)malloc(n_heads * n_tokens * head_dim * sizeof(float));
575 float *k_reorder = (float *)malloc(n_heads_kv * n_tokens * head_dim * sizeof(float));
576
577 if (q_reorder && k_reorder) {
578 /* Reorder Q: [T, H*D] -> [H, T, D] */
579 for (int t = 0; t < n_tokens; t++) {
580 for (int h = 0; h < n_heads; h++) {
581 for (int d = 0; d < head_dim; d++) {
582 q_reorder[h * n_tokens * head_dim + t * head_dim + d] =
583 q[t * n_heads * head_dim + h * head_dim + d];
584 }
585 }
586 }
587
588 /* Reorder K: [T, H_kv*D] -> [H_kv, T, D] */
589 for (int t = 0; t < n_tokens; t++) {
590 for (int h = 0; h < n_heads_kv; h++) {
591 for (int d = 0; d < head_dim; d++) {
592 k_reorder[h * n_tokens * head_dim + t * head_dim + d] =
593 k[t * n_heads_kv * head_dim + h * head_dim + d];
594 }
595 }
596 }
597
598 /* Apply RoPE */
599 rope_forward_qk(q_reorder, k_reorder,
600 cos_cache, sin_cache,
601 n_heads, n_heads_kv, n_tokens,
602 head_dim, head_dim, pos_offset);
603
604 /* Reorder back: [H, T, D] -> [T, H*D] */
605 for (int t = 0; t < n_tokens; t++) {
606 for (int h = 0; h < n_heads; h++) {
607 for (int d = 0; d < head_dim; d++) {
608 q[t * n_heads * head_dim + h * head_dim + d] =
609 q_reorder[h * n_tokens * head_dim + t * head_dim + d];
610 }
611 }
612 }
613
614 for (int t = 0; t < n_tokens; t++) {
615 for (int h = 0; h < n_heads_kv; h++) {
616 for (int d = 0; d < head_dim; d++) {
617 k[t * n_heads_kv * head_dim + h * head_dim + d] =
618 k_reorder[h * n_tokens * head_dim + t * head_dim + d];
619 }
620 }
621 }
622 }
623
624 free(q_reorder);
625 free(k_reorder);
626 free(cos_cache);
627 free(sin_cache);
628}

630void ck_test_rope_interleaved(float *q, float *k,
631 int n_tokens, int n_heads, int n_heads_kv, int head_dim,
632 int pos_offset, float theta)
633{
634 /* Interleaved RoPE format (matches llama.cpp):
635 * (x0, x1) -> (x0*cos - x1*sin, x0*sin + x1*cos)
636 * Applied to consecutive pairs of elements
637 */
638
639 /* Precompute inverse frequencies */
640 float *inv_freq = (float *)malloc((head_dim / 2) * sizeof(float));
641 if (!inv_freq) return;
642
643 for (int i = 0; i < head_dim / 2; i++) {
644 inv_freq[i] = 1.0f / powf(theta, (float)(2 * i) / head_dim);
645 }
646
647 /* Apply RoPE to Q */
648 for (int t = 0; t < n_tokens; t++) {
649 int pos = pos_offset + t;
650 for (int h = 0; h < n_heads; h++) {
651 float *qh = q + t * n_heads * head_dim + h * head_dim;
652
653 for (int i = 0; i < head_dim / 2; i++) {
654 float freq = pos * inv_freq[i];
655 float cos_val = cosf(freq);
656 float sin_val = sinf(freq);
657
658 /* Interleaved format */
659 float x0 = qh[i * 2];
660 float x1 = qh[i * 2 + 1];
661 qh[i * 2] = x0 * cos_val - x1 * sin_val;
662 qh[i * 2 + 1] = x0 * sin_val + x1 * cos_val;
663 }
664 }
665 }
666
667 /* Apply RoPE to K */
668 for (int t = 0; t < n_tokens; t++) {
669 int pos = pos_offset + t;
670 for (int h = 0; h < n_heads_kv; h++) {
671 float *kh = k + t * n_heads_kv * head_dim + h * head_dim;
672
673 for (int i = 0; i < head_dim / 2; i++) {
674 float freq = pos * inv_freq[i];
675 float cos_val = cosf(freq);
676 float sin_val = sinf(freq);
677
678 float x0 = kh[i * 2];
679 float x1 = kh[i * 2 + 1];
680 kh[i * 2] = x0 * cos_val - x1 * sin_val;
681 kh[i * 2 + 1] = x0 * sin_val + x1 * cos_val;
682 }
683 }
684 }
685
686 free(inv_freq);
687}

696void ck_test_softmax(const float *input, float *output, int n)
697{
698 /* Find max for numerical stability */
699 float max_val = input[0];
700 for (int i = 1; i < n; i++) {
701 if (input[i] > max_val) max_val = input[i];
702 }
703
704 /* Compute exp and sum */
705 float sum = 0.0f;
706 for (int i = 0; i < n; i++) {
707 output[i] = expf(input[i] - max_val);
708 sum += output[i];
709 }
710
711 /* Normalize */
712 float inv_sum = 1.0f / sum;
713 for (int i = 0; i < n; i++) {
714 output[i] *= inv_sum;
715 }
716}

913/**
914 * @brief Test mega-fused OutProj + MLP kernel (Q5_0 weights)
915 *
916 * This is a simplified wrapper for parity testing that:
917 * - Uses Q5_0 for W_o and W1 weights
918 * - Uses Q4_K for W2 weights
919 * - Allocates scratch internally
920 *
921 * @param attn_out Attention output [num_heads, tokens, head_dim] (FP32, head-major)
922 * @param residual Residual input [tokens, embed_dim] (FP32)
923 * @param ln2_gamma RMSNorm gamma [embed_dim] (FP32)
924 * @param wo OutProj weights [embed_dim, embed_dim] (Q5_0)
925 * @param w1 MLP W1 weights [2*intermediate, embed_dim] (Q5_0)
926 * @param w2 MLP W2 weights [embed_dim, intermediate] (Q4_K or Q6_K)
927 * @param output Output [tokens, embed_dim] (FP32)
928 * @param tokens Number of tokens
929 * @param num_heads Number of attention heads
930 * @param head_dim Dimension per head
931 * @param embed_dim Embedding dimension (= num_heads * head_dim)
932 * @param intermediate MLP intermediate dimension
933 * @param eps RMSNorm epsilon
934 * @param w2_is_q6k If true, W2 is Q6_K; if false, W2 is Q4_K
935 */
936void ck_test_outproj_mlp_fused_q5_0(
937 const float *attn_out,
938 const float *residual,
939 const float *ln2_gamma,
940 const void *wo,
941 const void *w1,
942 const void *w2,
943 float *output,

947 int embed_dim,

954 const int CK_DT_Q4_K_VAL = 7;
955 const int CK_DT_Q6_K_VAL = 8;

959 int aligned_head_dim = head_dim;

962 /* Ensure intermediate is multiple of 256 (QK_K) for K-quants */

964 aligned_intermediate = ((intermediate + 255) / 256) * 256;
965 }

973 return;
974 }
975
976 /* Call the mega-fused kernel */
977 mega_fused_outproj_mlp_prefill(
978 output,
979 attn_out,
980 residual,
981 ln2_gamma,
982 wo, NULL, CK_DT_Q5_0_VAL, /* W_o with Q5_0 */

984 w2, NULL, w2_is_q6k ? CK_DT_Q6_K_VAL : CK_DT_Q4_K_VAL, /* W2 with Q4_K or Q6_K */
985 tokens,
986 embed_dim,

988 num_heads,
989 aligned_head_dim,

992 eps,
993 scratch
994 );
