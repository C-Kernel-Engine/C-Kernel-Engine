26/**
27 * @brief Wrapper to call TRUE flash attention from orchestration layer
28 *
29 * @param q_token Query token [H, D_h]
30 * @param k_cache Cached keys [T_k, H, D_h]
31 * @param v_cache Cached values [T_k, H, D_h]
32 * @param out_token Output [H, D_h]
33 * @param num_heads Number of heads
34 * @param num_kv_heads Number of KV heads (for GQA)
35 * @param kv_tokens Number of tokens in KV cache
36 * @param cache_capacity Cache capacity
37 * @param head_dim Head dimension
38 * @param aligned_head_dim Aligned head dimension
39 */
40void ck_attention_flash_decode_wrapper(
41 const float *q_token,
42 const float *k_cache,
43 const float *v_cache,
44 float *out_token,
45 int num_heads,
46 int num_kv_heads,
47 int kv_tokens,
48 int cache_capacity,
49 int head_dim,
50 int aligned_head_dim)
51{
52 if (!q_token || !k_cache || !v_cache || !out_token) {
53 return;
54 }
55 if (num_heads <= 0 || num_kv_heads <= 0 || kv_tokens <= 0 || cache_capacity <= 0) {
56 return;
57 }
58 if (kv_tokens > cache_capacity || head_dim <= 0 || aligned_head_dim <= 0) {
59 return;
60 }
61
62 static int use_strict = -1;
63 if (use_strict < 0) {
64 const char *env = getenv("CK_FLASH_ATTN_STRICT");
65 use_strict = (env && env[0] && env[0] != '0') ? 1 : 0;
66 }
67
68 if (use_strict) {
69 attention_forward_decode_head_major_gqa_regular(q_token,
70 k_cache,
71 v_cache,
72 out_token,
73 num_heads,
74 num_kv_heads,
75 kv_tokens,
76 cache_capacity,
77 head_dim,
78 aligned_head_dim);
79 return;
80 }
81
82 // Scale factor: 1/sqrt(head_dim)
83 const float scale = 1.0f / sqrtf((float)head_dim);
84 const size_t head_stride = (size_t)cache_capacity * (size_t)aligned_head_dim;
85
86#pragma omp parallel for schedule(static) if(num_heads > 1)
87 for (int h = 0; h < num_heads; ++h) {
88 const int kv_head = (int)((long long)h * (long long)num_kv_heads / (long long)num_heads);
89 const float *q_head = q_token + (size_t)h * (size_t)aligned_head_dim;
90 const float *k_head = k_cache + (size_t)kv_head * head_stride;
91 const float *v_head = v_cache + (size_t)kv_head * head_stride;
92 float *out_head = out_token + (size_t)h * (size_t)aligned_head_dim;
93
94 // Use aligned_head_dim as D_h so per-token stride matches the cache layout.
95 attention_flash_decode(out_head,
96 q_head,
97 k_head,
98 v_head,
99 1,
100 kv_tokens,
101 1,
102 aligned_head_dim,
103 scale);
104 }
105}

342 * ============================================================================
343 *
344 * These helpers keep the same activation layouts as the fp32 code paths, but
345 * accept weight matrices stored as GGML-compatible Q4_K blocks. This is meant
346 * for weight-only quantized inference: activations remain fp32 by default, but

349 * Important constraints:
350 * - Q4_K kernels require K (the input dimension) to be a multiple of 256.
351 * - For attention output projection we assume the concatenated head vector

772 float *d_wo,

778{
779 if (!d_out || !attn_out || !wo || !d_attn_out || !d_wo || !d_bo) {
780 return;
781 }

783 // Bias gradient: sum over tokens once (bias is applied once in forward).
784 for (int d = 0; d < aligned_embed_dim; ++d) {

786 }

799 return;

811 wo_h,
812 d_head_in,
813 d_wo_h,

816 aligned_head_dim,
817 aligned_embed_dim,
818 1);
819 }
