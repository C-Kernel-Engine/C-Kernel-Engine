73 /**
74  * @brief Fused RMSNorm + QKV for decode (single token)
75  *
76  * AVX-optimized version. Intermediates stay in L1/L2.
77  *
78  * @param q_out Output Q [num_heads * head_dim] - to RoPE
79  * @param k_out Output K [num_kv_heads * head_dim] - to RoPE
80  * @param v_out Output V [num_kv_heads * head_dim] - to KV cache
81  * @param input Input token [hidden] - from DRAM
82  * @param gamma RMSNorm gamma [hidden]
83  * @param W_qkv QKV weights [3*hidden, hidden]
84  * @param b_qkv QKV bias [3*hidden] or NULL
85  * @param hidden Model hidden dimension
86  * @param num_heads Number of attention heads
87  * @param num_kv_heads Number of KV heads
88  * @param head_dim Head dimension
89  * @param eps RMSNorm epsilon
90  */
91 void mega_fuse_rmsnorm_qkv_avx(
92  float *q_out,
93  float *k_out,
94  float *v_out,
95  const float *input,
96  const float *gamma,
97  const float *W_qkv,
98  const float *b_qkv,
99  int hidden,
100  int num_heads,
101  int num_kv_heads,
102  int head_dim,
103  float eps)
104 {
105  const int H = num_heads;
106  const int H_kv = num_kv_heads;
107  const int hd = head_dim;
108  const int ad = (head_dim + MEGA_VLEN - 1) / MEGA_VLEN * MEGA_VLEN; /* aligned */
109 
110  /* Stack buffer for RMSNorm output - stays in L1! */
111  float ln1_row[hidden];
112  float rstd;
113  float sum_sq = 0.0f;
114 
115  /* RMSNorm: compute in registers, store to L1 stack buffer */
116 #if defined(__AVX512F__)
117  __m512 sum = _mm512_setzero_ps();
118  for (int i = 0; i < hidden; i += 16) {
119  __m512 xv = _mm512_loadu_ps(input + i);
120  sum = _mm512_add_ps(sum, _mm512_mul_ps(xv, xv));
121  }
122  float rowsum = _mm512_reduce_add_ps(sum);
123  rstd = 1.0f / sqrtf(rowsum / hidden + eps);
124 #elif defined(__AVX__)
125  __m256 sum = _mm256_setzero_ps();
126  for (int i = 0; i < hidden; i += 8) {
127  __m256 xv = _mm256_loadu_ps(input + i);
128  sum = _mm256_add_ps(sum, _mm256_mul_ps(xv, xv));
129  }
130  /* Horizontal sum of 256-bit vector */
131  __m128 sum128 = _mm256_extractf128_ps(sum, 0);
132  __m128 hi = _mm256_extractf128_ps(sum, 1);
133  sum128 = _mm_add_ps(sum128, hi);
134  float rowsum = _mm_cvtss_f32(_mm_hadd_ps(sum128, sum128));
135  rstd = 1.0f / sqrtf(rowsum / hidden + eps);
136 #endif
137 
138  /* Normalize and project - no DRAM write of ln1_row! */
139  size_t q_elems = (size_t)H * ad;
140  size_t kv_elems = (size_t)H_kv * ad;
141 
142  /* Q projection: ln1_row is still hot in L1 */
143  for (int j = 0; j < q_elems; j += 4) {
144  float dot = 0.0f;
145  for (int i = 0; i < hidden; i++) {
146  dot += ln1_row[i] * W_qkv[j * (size_t)hidden + i];
147  }
148  q_out[j] = dot + (b_qkv ? b_qkv[j] : 0.0f);
149  }
150 
151  /* K projection - same L1 buffer */
152  const float *W_k = W_qkv + (size_t)H * ad * hidden;
153  for (int j = 0; j < kv_elems; j += 4) {
154  float dot = 0.0f;
155  for (int i = 0; i < hidden; i++) {
156  dot += ln1_row[i] * W_k[j * (size_t)hidden + i];
157  }
158  k_out[j] = dot + (b_qkv ? b_qkv[H * ad + j] : 0.0f);
159  }
160 
161  /* V projection - same L1 buffer */
162  const float *W_v = W_qkv + (size_t)(H + H_kv) * ad * hidden;
163  for (int j = 0; j < kv_elems; j += 4) {
164  float dot = 0.0f;
165  for (int i = 0; i < hidden; i++) {
166  dot += ln1_row[i] * W_v[j * (size_t)hidden + i];
167  }
168  v_out[j] = dot + (b_qkv ? b_qkv[(H + H_kv) * ad + j] : 0.0f);
169  }
170 
171  /* q_out, k_out, v_out are in L1 - ready for RoPE! */
172 }

178 /**
179  * @brief Apply RoPE to Q and K (in-place, from L1)
180  *
181  * Q and K are already in L1 from QKV projection.
182  * Just apply rotation in-place.
183  */
184 void mega_fuse_rope_inplace_avx(
185  float *q, /* [num_heads * head_dim] - in L1 */
186  float *k, /* [num_kv_heads * head_dim] - in L1 */
187  const float *rope_cos,
188  const float *rope_sin,
189  int pos,
190  int num_heads,
191  int num_kv_heads,
192  int head_dim)
193 {
194  int half = head_dim / 2;
195 
196  for (int h = 0; h < num_heads; h++) {
197  float *q_h = q + h * head_dim;
198  float *k_h = k + (h % num_kv_heads) * head_dim;
199 
200  const float *cos_ptr = rope_cos + pos * half;
201  const float *sin_ptr = rope_sin + pos * half;
202 
203  for (int i = 0; i < half; i += MEGA_VLEN) {
204 #if defined(__AVX512F__)
205  __m512 q0 = _mm512_loadu_ps(q_h + i);
206  __m512 q1 = _mm512_loadu_ps(q_h + i + half);
207  __m512 k0 = _mm512_loadu_ps(k_h + i);
208  __m512 k1 = _mm512_loadu_ps(k_h + i + half);
209  __m512 cos = _mm512_loadu_ps(cos_ptr + i);
210  __m512 sin = _mm512_loadu_ps(sin_ptr + i);
211 
212  __m512 q_rot0 = _mm512_fmsub_ps(q0, cos, _mm512_mul_ps(q1, sin));
213  __m512 q_rot1 = _mm512_fmadd_ps(q0, sin, _mm512_mul_ps(q1, cos));
214  __m512 k_rot0 = _mm512_fmsub_ps(k0, cos, _mm512_mul_ps(k1, sin));
215  __m512 k_rot1 = _mm512_fmadd_ps(k0, sin, _mm512_mul_ps(k1, cos));
216 
217  _mm512_storeu_ps(q_h + i, q_rot0);
218  _mm512_storeu_ps(q_h + i + half, q_rot1);
219  _mm512_storeu_ps(k_h + i, k_rot0);
220  _mm512_storeu_ps(k_h + i + half, k_rot1);
221 #elif defined(__AVX__)
222  __m256 q0 = _mm256_loadu_ps(q_h + i);
223  __m256 q1 = _mm256_loadu_ps(q_h + i + half);
224  __m256 k0 = _mm256_loadu_ps(k_h + i);
225  __m256 k1 = _mm256_loadu_ps(k_h + i + half);
226  __m256 cos = _mm256_loadu_ps(cos_ptr + i);
227  __m256 sin = _mm256_loadu_ps(sin_ptr + i);
228 
229  __m256 q_rot0 = _mm256_fmsub_ps(q0, cos, _mm256_mul_ps(q1, sin));
230  __m256 q_rot1 = _mm256_fmadd_ps(q0, sin, _mm256_mul_ps(q1, cos));
231  __m256 k_rot0 = _mm256_fmsub_ps(k0, cos, _mm256_mul_ps(k1, sin));
232  __m256 k_rot1 = _mm256_fmadd_ps(k0, sin, _mm256_mul_ps(k1, cos));
233 
234  _mm256_storeu_ps(q_h + i, q_rot0);
235  _mm256_storeu_ps(q_h + i + half, q_rot1);
236  _mm256_storeu_ps(k_h + i, k_rot0);
237  _mm256_storeu_ps(k_h + i + half, k_rot1);
238 #endif
239  }
240  }
241 }

243 /*============================================================================
244  * Phase 3: Flash Attention with Online Softmax
245  *
246  * O, m, l stay in registers across all KV tiles.
247  * K/V stream from KV cache (in L2).
248  *============================================================================*/
249 
250 /**
251  * @brief Flash attention with online softmax (AVX version)
252  *
253  * Key insight: O, m, l stay in registers throughout!
254  * K/V tiles stream from L2 cache.
255  *
256  * @param o_out Output [num_heads * head_dim] - in registers/L1
257  * @param q Q tensor [num_heads * head_dim] - from L1
258  * @param kv_cache_k KV cache K [seq, num_kv_heads * head_dim] - in L2
259  * @param kv_cache_v KV cache V [seq, num_kv_heads * head_dim] - in L2
260  * @param num_heads Number of heads
261  * @param num_kv_heads Number of KV heads
262  * @param seq_len Current sequence length
263  * @param head_dim Head dimension
264  */
265 void mega_fuse_flash_attention_avx(
266  float *o_out,
267  const float *q,
268  const float *kv_cache_k,
269  const float *kv_cache_v,
270  int num_heads,
271  int num_kv_heads,
272  int seq_len,
273  int head_dim)
274 {
275  const int hd = head_dim;
276  const float scale = 1.0f / sqrtf((float)hd);
277 
278  for (int h = 0; h < num_heads; h++) {
279  const float *q_h = q + h * hd;
280  const int kv_idx = h % num_kv_heads;
281  const float *k_cache = kv_cache_k + kv_idx * hd;
282  const float *v_cache = kv_cache_v + kv_idx * hd;
283 
284  /* O, m, l in registers for this head */
285  float o_h[hd]; /* in L1 */
286  float m = -INFINITY; /* running max */
287  float l = 0.0f; /* running sum */
288 
289  /* Initialize O to zeros */
290  memset(o_h, 0, hd * sizeof(float));
291 
292  /* Iterate over KV cache tiles */
293  for (int t = 0; t < seq_len; t += MEGA_KV_TILE) {
294  int tile_end = t + MEGA_KV_TILE;
295  if (tile_end > seq_len) tile_end = seq_len;
296  int tile_size = tile_end - t;
297 
298  /* Load K tile from L2 cache */
299  float k_tile[MEGA_KV_TILE * hd];
300  for (int i = 0; i < tile_size; i++) {
301  memcpy(k_tile + i * hd, k_cache + (t + i) * hd, hd * sizeof(float));
302  }
303 
304  /* S_ij = Q @ K_tile.T / sqrt(d) - in registers */
305  float s_row[MEGA_KV_TILE];
306  for (int j = 0; j < tile_size; j++) {
307  s_row[j] = 0.0f;
308  for (int i = 0; i < hd; i++) {
309  s_row[j] += q_h[i] * k_tile[j * hd + i];
310  }
311  s_row[j] *= scale;
312  }
313 
314  /* Online softmax update */
315  float m_new = m;
316  for (int j = 0; j < tile_size; j++) {
317  if (s_row[j] > m_new) m_new = s_row[j];
318  }
319 
320  float l_new = 0.0f;
321  for (int j = 0; j < tile_size; j++) {
322  float p = expf(s_row[j] - m_new);
323  s_row[j] = p;
324  l_new += p;
325  }
326 
327  /* Scale O by exp(m - m_new) and add P @ V_tile */
328  float exp_m_diff = expf(m - m_new);
329  for (int i = 0; i < hd; i++) {
330  o_h[i] *= exp_m_diff;
331  }
332 
333  /* Load V tile and accumulate */
334  for (int j = 0; j < tile_size; j++) {
335  float p = s_row[j];
336  for (int i = 0; i < hd; i++) {
337  o_h[i] += p * v_cache[(t + j) * hd + i];
338  }
339  }
340 
341  l = l_new * exp_m_diff + l;
342  m = m_new;
343  }
344 
345  /* Normalize by l */
346  for (int i = 0; i < hd; i++) {
347  o_h[i] /= l;
348  }
349 
350  /* Store O - still in L1, goes to output projection */
351  memcpy(o_out + h * hd, o_h, hd * sizeof(float));
352  }
353 }

362 /**
363  * @brief Full mega-fused attention for decode
364  *
365  * THE HOLY GRAIL: All operations fused, no intermediates to DRAM.
366  *
367  * Reads: input[hidden], W_qkv, W_o, KV cache, RoPE tables
368  * Writes: output[hidden]
369  *
370  * Total DRAM traffic: 4KB input + 4KB output = 8KB
371  * (vs ~800KB with unfused attention)
372  */
373 void mega_fused_attention_decode_avx(
374  float *output, /* [hidden] - single DRAM write */
375  const float *input, /* [hidden] - single DRAM read */
376  const float *residual, /* [hidden] - for residual add */
377  const float *W_qkv, /* QKV weights */
378  const float *b_qkv, /* QKV bias or NULL */
379  const float *W_o, /* Output projection */
380  const float *b_o, /* Output bias or NULL */
381  float *kv_cache_k, /* KV cache K - in L2 */
382  float *kv_cache_v, /* KV cache V - in L2 */
383  const float *rope_cos,
384  const float *rope_sin,
385  int pos,
386  int hidden,
387  int num_heads,
388  int num_kv_heads,
389  int head_dim,
390  int max_seq,
391  float eps)
392 {
393  /* Allocate output buffers in L1 - not DRAM! */
394  size_t q_elems = (size_t)num_heads * head_dim;
395  size_t kv_elems = (size_t)num_kv_heads * head_dim;
396  float *q = (float *)malloc(q_elems * sizeof(float));
397  float *k = (float *)malloc(kv_elems * sizeof(float));
398  float *v = (float *)malloc(kv_elems * sizeof(float));
399  float *o = (float *)malloc(q_elems * sizeof(float));
400 
401  /* Phase 1: RMSNorm + QKV (ln1_row in L1 stack) */
402  mega_fuse_rmsnorm_qkv_avx(q, k, v, input, NULL, W_qkv, b_qkv,
403  hidden, num_heads, num_kv_heads, head_dim, eps);
404 
405  /* Phase 2: RoPE in-place (q, k still hot in L1) */
406  mega_fuse_rope_inplace_avx(q, k, rope_cos, rope_sin, pos,
407  num_heads, num_kv_heads, head_dim);
408 
409  /* Write K/V to KV cache (L2) */
410  for (int h = 0; h < num_kv_heads; h++) {
411  memcpy(kv_cache_k + pos * kv_elems + h * head_dim,
412  k + h * head_dim, head_dim * sizeof(float));
413  memcpy(kv_cache_v + pos * kv_elems + h * head_dim,
414  v + h * head_dim, head_dim * sizeof(float));
415  }
416 
417  /* Phase 3: Flash Attention (o in L1) */
418  mega_fuse_flash_attention_avx(o, q, kv_cache_k, kv_cache_v,
419  num_heads, num_kv_heads, pos + 1, head_dim);
420 
421  /* Phase 4: Output Projection + Residual (in final store) */
422  for (int j = 0; j < hidden; j++) {
423  float dot = 0.0f;
424  for (int h = 0; h < num_heads; h++) {
425  const float *o_h = o + h * head_dim;
426  const float *W_o_h = W_o + h * head_dim * hidden + j * head_dim;
427  for (int i = 0; i < head_dim; i++) {
428  dot += o_h[i] * W_o_h[i];
429  }
430  }
431  float proj = dot + (b_o ? b_o[j] : 0.0f);
432  output[j] = proj + residual[j]; /* Residual add in store! */
433  }
434 
435  free(q);
436  free(k);
437  free(v);
438  free(o);
439 
440  /* Total DRAM traffic: 4KB input + 4KB output = 8KB */
441 }
