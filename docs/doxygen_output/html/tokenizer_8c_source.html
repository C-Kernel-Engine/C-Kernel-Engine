63 /* Create a new tokenizer */
64 CKTokenizer *ck_tokenizer_create(CKTokenizerType type) {
65  CKTokenizer *tok = (CKTokenizer *)malloc(sizeof(CKTokenizer));
66  if (!tok) {
67  return NULL;
68  }
69 
70  memset(tok, 0, sizeof(*tok));
71 
72  /* Create hash table for vocabulary */
73  tok->vocab = ck_tokenizer_hash_table_create(CK_TOKENIZER_HT_BUCKETS_LARGE);
74  if (!tok->vocab) {
75  free(tok);
76  return NULL;
77  }
78 
79  /* Create trie for fast lookups (1M nodes for ~50k vocab) */
80  tok->vocab_trie = ck_trie_create(1000000);
81  if (!tok->vocab_trie) {
82  ck_tokenizer_hash_table_free(tok->vocab, true);
83  free(tok);
84  return NULL;
85  }
86 
87  /* Initialize reverse vocab */
88  tok->vocab_capacity = 4096;
89  tok->id_to_token = (char **)calloc(tok->vocab_capacity, sizeof(char *));
90  if (!tok->id_to_token) {
91  ck_tokenizer_hash_table_free(tok->vocab, true);
92  free(tok);
93  return NULL;
94  }
95 
96  /* Set default special tokens */
97  tok->unk_id = 0;
98  tok->bos_id = 1;
99  tok->eos_id = 2;
100  tok->pad_id = -1;
101  tok->mask_id = -1;
102 
103  /* Set config */
104  tok->config.type = type;
105  tok->config.add_bos = false;
106  tok->config.add_eos = false;
107  tok->config.unk_score = -1e10f;
108 
109  ck_tokenizer_mempool_init(&tok->pool, 1024 * 1024);
110 
111  return tok;
112 }

114 /* Free a tokenizer */
115 void ck_tokenizer_free(CKTokenizer *tok) {
116  if (!tok) return;
117 
118  /* Free vocabulary entries */
119  if (tok->vocab) {
120  ck_tokenizer_hash_table_free(tok->vocab, true);
121  }
122 
123  /* Free trie */
124  if (tok->vocab_trie) {
125  ck_trie_free(tok->vocab_trie);
126  }
127 
128  /* Free reverse vocab strings */
129  if (tok->id_to_token) {
130  /* Note: strings were strdup'd in add_token */
131  for (size_t i = 0; i < tok->vocab_size; i++) {
132  if (tok->id_to_token[i]) {
133  free(tok->id_to_token[i]);
134  }
135  }
136  free(tok->id_to_token);
137  }
138 
139  ck_tokenizer_mempool_free(&tok->pool);
140  free(tok);
141 }

163 /* Add a token to vocabulary */
164 int ck_tokenizer_add_token(CKTokenizer *tok, const char *token, int32_t id, float score) {
165  if (!tok || !token) {
166  return -1;
167  }
168 
169  /* Ensure we have space in reverse vocab */
170  if (id >= (int32_t)tok->vocab_capacity) {
171  size_t new_cap = tok->vocab_capacity * 2;
172  while (new_cap <= (size_t)id) {
173  new_cap *= 2;
174  }
175  char **new_array = (char **)realloc(tok->id_to_token, new_cap * sizeof(char *));
176  if (!new_array) {
177  return -1;
178  }
179  memset(new_array + tok->vocab_capacity, 0, (new_cap - tok->vocab_capacity) * sizeof(char *));
180  tok->id_to_token = new_array;
181  tok->vocab_capacity = new_cap;
182  }
183 
184  /* Check if token already exists */
185  TokenInfo *existing = (TokenInfo *)ck_tokenizer_hash_table_lookup(tok->vocab, token);
186  if (existing) {
187  existing->id = id;
188  existing->score = score;
189  if (id >= (int32_t)tok->vocab_size) tok->vocab_size = id + 1;
190  if (tok->id_to_token[id]) free(tok->id_to_token[id]);
191  tok->id_to_token[id] = strdup(token);
192  return 0;
193  }
194 
195  /* Create new token info */
196  TokenInfo *info = (TokenInfo *)malloc(sizeof(TokenInfo));
197  if (!info) return -1;
198  info->id = id;
199  info->score = score;
200  info->is_special = false;
201 
202  if (ck_tokenizer_hash_table_insert(tok->vocab, token, info) != 0) {
203  free(info);
204  return -1;
205  }
206 
207  /* Also add to trie for fast longest-match lookups */
208  if (tok->vocab_trie) {
209  ck_trie_insert(tok->vocab_trie, token, id, false, 0);
210  }
211 
212  if (id >= (int32_t)tok->vocab_size) tok->vocab_size = id + 1;
213  if (tok->id_to_token[id]) free(tok->id_to_token[id]);
214  tok->id_to_token[id] = strdup(token);
215 
216  return 0;
217 }

219 /* Add special token */
220 int ck_tokenizer_add_special_token(CKTokenizer *tok, const char *name, int32_t id) {
221  if (!tok || !name) return -1;
222  if (ck_tokenizer_add_token(tok, name, id, -1e10f) != 0) return -1;
223 
224  TokenInfo *info = (TokenInfo *)ck_tokenizer_hash_table_lookup(tok->vocab, name);
225  if (info) info->is_special = true;
226 
227  /* Also add to trie as special */
228  if (tok->vocab_trie) {
229  ck_trie_insert(tok->vocab_trie, name, id, true, 0);
230  }
231 
232  if (strcmp(name, "<unk>") == 0 || strcmp(name, "[UNK]") == 0) tok->unk_id = id;
233  else if (strcmp(name, "<s>") == 0 || strcmp(name, "<bos>") == 0 || strcmp(name, "[BOS]") == 0) tok->bos_id = id;
234  else if (strcmp(name, "</s>") == 0 || strcmp(name, "<eos>") == 0 || strcmp(name, "[EOS]") == 0) tok->eos_id = id;
235  else if (strcmp(name, "<pad>") == 0 || strcmp(name, "[PAD]") == 0) tok->pad_id = id;
236 
237  return 0;
238 }

270 static int32_t find_longest_match_trie(const CKTokenizer *tok, const char *text, size_t text_len, size_t pos, size_t *match_len) {
271  if (!tok || !tok->vocab_trie || !text || pos >= text_len) {
272  *match_len = 0;
273  return tok ? tok->unk_id : -1;
274  }
275 
276  int32_t token_id = ck_trie_find_longest(tok->vocab_trie, text, text_len, pos, match_len);
277  return token_id >= 0 ? token_id : tok->unk_id;
278 }

280 /* Find longest matching token at position using hash table (O(n*k) worst case) */
281 static int32_t find_longest_match_hash(const CKTokenizer *tok, const char *text, size_t text_len, size_t pos, size_t *match_len) {
282  if (!tok || !text || pos >= text_len) {
283  *match_len = 0;
284  return tok ? tok->unk_id : -1;
285  }
286 
287  size_t max_len = 64;
288  if (pos + max_len > text_len) max_len = text_len - pos;
289 
290  int32_t best_id = tok->unk_id;
291  size_t best_len = 0;
292 
293  for (size_t len = max_len; len >= 1; len--) {
294  char tmp[65];
295  memcpy(tmp, text + pos, len);
296  tmp[len] = '\0';
297 
298  TokenInfo *info = (TokenInfo *)ck_tokenizer_hash_table_lookup(tok->vocab, tmp);
299  if (info) {
300  best_id = info->id;
301  best_len = len;
302  break;
303  }
304  }
305 
306  *match_len = best_len;
307  return best_id;
308 }

319 /* Encode text to token IDs using greedy longest-match */
320 int ck_tokenizer_encode(const CKTokenizer *tok, const char *text, int text_len, int32_t *ids, int max_ids) {
321  if (!tok || !text || !ids || max_ids <= 0) return 0;
322  if (text_len < 0) text_len = (int)strlen(text);
323 
324  int out_idx = 0;
325  if (tok->config.add_bos && tok->bos_id >= 0 && out_idx < max_ids) {
326  ids[out_idx++] = tok->bos_id;
327  }
328 
329  size_t pos = 0;
330  while (pos < (size_t)text_len && out_idx < max_ids) {
331  /* Skip whitespace for BPE/SPM (not WordPiece) */
332  if (tok->config.type != CK_TOKENIZER_WORDPIECE) {
333  while (pos < (size_t)text_len && isspace((unsigned char)text[pos])) pos++;
334  if (pos >= (size_t)text_len) break;
335  }
336 
337  size_t match_len = 0;
338  int32_t id = find_longest_match(tok, text, text_len, pos, &match_len);
339 
340  if (match_len == 0) {
341  /* Emit UNK for unknown characters */
342  if (tok->unk_id >= 0) ids[out_idx++] = tok->unk_id;
343  pos++;
344  } else {
345  ids[out_idx++] = id;
346  pos += match_len;
347  }
348  }
349 
350  if (tok->config.add_eos && tok->eos_id >= 0 && out_idx < max_ids) {
351  ids[out_idx++] = tok->eos_id;
352  }
353 
354  return out_idx;
355 }

357 /* Decode token IDs to text */
358 int ck_tokenizer_decode(const CKTokenizer *tok, const int32_t *ids, int num_ids, char *text, int max_len) {
359  if (!tok || !ids || !text || max_len <= 0) return 0;
360  int len = 0;
361  for (int i = 0; i < num_ids; i++) {
362  int32_t id = ids[i];
363  if (id < 0) continue;
364  const char *token = ck_tokenizer_id_to_token(tok, id);
365  if (!token) continue;
366  int token_len = (int)strlen(token);
367  if ((unsigned char)token[0] == 0xC4 && (unsigned char)token[1] == 0xA0) {
368  if (len < max_len - 1) text[len++] = ' ';
369  token++; token_len--;
370  }
371  for (int j = 0; j < token_len && len < max_len - 1; j++) text[len++] = token[j];
372  }
373  text[len] = '\0';
374  return len;
375 }
