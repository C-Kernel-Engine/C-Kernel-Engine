1 /**
2  * @file gemm_kernels_q8_0.c
3  * @brief GEMM/GEMV kernels with Q8_0 quantized weights
4  *
5  * Q8_0 Format:
6  * - 32 weights per block
7  * - 1 FP16 scale per block
8  * - 34 bytes per 32 weights = 8.5 bits/weight
9  * - Weights stored as signed 8-bit integers
10  *
11  * Operations:
12  * Forward: Y = W @ X (W is Q8_0, X and Y are FP32)
13  * Backward: dX = W^T @ dY (gradient w.r.t. input)
14  *
15  * Note: Q8_0 is often used for activation quantization or as an
16  * intermediate format. Higher precision than Q4_0/Q4_K.
17  */

34 /**
35  * @brief Matrix-vector multiply with Q8_0 weights (scalar reference)
36  *
37  * @param y Output vector [M]
38  * @param W Weight matrix in Q8_0 format [M x K]
39  * @param x Input vector [K]
40  * @param M Number of output rows
41  * @param K Number of columns (must be multiple of 32)
42  */
43 void gemv_q8_0_ref(float *y,
44  const void *W,
45  const float *x,
46  int M, int K)
47 {
48  const block_q8_0 *blocks = (const block_q8_0 *)W;
49  const int blocks_per_row = K / QK8_0;
50 
51  for (int row = 0; row < M; row++) {
52  float sum = 0.0f;
53 
54  for (int b = 0; b < blocks_per_row; b++) {
55  const block_q8_0 *block = &blocks[row * blocks_per_row + b];
56  const float d = CK_FP16_TO_FP32(block->d);
57  const float *xp = &x[b * QK8_0];
58 
59  for (int i = 0; i < QK8_0; i++) {
60  sum += d * (float)block->qs[i] * xp[i];
61  }
62  }
63 
64  y[row] = sum;
65  }
66 }

72 void gemv_q8_0_avx512(float *y,
73  const void *W,
74  const float *x,
75  int M, int K)
76 {
77  const block_q8_0 *blocks = (const block_q8_0 *)W;
78  const int blocks_per_row = K / QK8_0;
79 
80  for (int row = 0; row < M; row++) {
81  __m512 acc = _mm512_setzero_ps();
82 
83  for (int b = 0; b < blocks_per_row; b++) {
84  const block_q8_0 *block = &blocks[row * blocks_per_row + b];
85  const __m512 vscale = _mm512_set1_ps(CK_FP16_TO_FP32(block->d));
86  const float *xp = &x[b * QK8_0];
87 
88  /* Process 32 weights in two batches of 16 */
89  for (int chunk = 0; chunk < 2; chunk++) {
90  /* Load 16 x int8 weights */
91  __m128i q8 = _mm_loadu_si128((const __m128i *)&block->qs[chunk * 16]);
92 
93  /* Sign-extend to 32-bit */
94  __m512i q32 = _mm512_cvtepi8_epi32(q8);
95 
96  /* Convert to float and scale */
97  __m512 w = _mm512_mul_ps(_mm512_cvtepi32_ps(q32), vscale);
98 
99  /* Load input */
100  __m512 x_vec = _mm512_loadu_ps(&xp[chunk * 16]);
101 
102  /* FMA */
103  acc = _mm512_fmadd_ps(w, x_vec, acc);
104  }
105  }
106 
107  y[row] = _mm512_reduce_add_ps(acc);
108  }
109 }

115 /* Helper macro: extract 4 int8 weights at byte offset, convert to float, multiply with x */
116 #define SSE_Q8_BLOCK(q8_reg, offset, xp, d_val, acc) do { \
117  __m128 vx = _mm_loadu_ps(&(xp)[offset]); \
118  __m128i qw = _mm_cvtepi8_epi32(_mm_srli_si128(q8_reg, offset)); \
119  __m128 vw = _mm_cvtepi32_ps(qw); \
120  acc = _mm_add_ps(acc, _mm_mul_ps(_mm_mul_ps(vw, vx), _mm_set1_ps(d_val))); \
121 } while(0)

123 void gemv_q8_0_sse(float *y,
124  const void *W,
125  const float *x,
126  int M, int K)
127 {
128  const block_q8_0 *blocks = (const block_q8_0 *)W;
129  const int blocks_per_row = K / QK8_0;
130 
131  for (int row = 0; row < M; row++) {
132  __m128 acc = _mm_setzero_ps();
133 
134  for (int b = 0; b < blocks_per_row; b++) {
135  const block_q8_0 *block = &blocks[row * blocks_per_row + b];
136  const float d_val = CK_FP16_TO_FP32(block->d);
137  const float *xp = &x[b * QK8_0];
138 
139  /* Load 32 weights (signed 8-bit) in two 16-byte chunks */
140  __m128i q8_0 = _mm_loadu_si128((const __m128i *)&block->qs[0]);
141  __m128i q8_1 = _mm_loadu_si128((const __m128i *)&block->qs[16]);
142 
143  /* Process first 16 weights (q8_0) - unrolled with compile-time constants */
144  SSE_Q8_BLOCK(q8_0, 0, xp, d_val, acc);
145  SSE_Q8_BLOCK(q8_0, 4, xp, d_val, acc);
146  SSE_Q8_BLOCK(q8_0, 8, xp, d_val, acc);
147  SSE_Q8_BLOCK(q8_0, 12, xp, d_val, acc);
148 
149  /* Process second 16 weights (q8_1) - offset xp by 16 */
150  const float *xp1 = xp + 16;
151  SSE_Q8_BLOCK(q8_1, 0, xp1, d_val, acc);
152  SSE_Q8_BLOCK(q8_1, 4, xp1, d_val, acc);
153  SSE_Q8_BLOCK(q8_1, 8, xp1, d_val, acc);
154  SSE_Q8_BLOCK(q8_1, 12, xp1, d_val, acc);
155  }
156 
157  /* Horizontal sum */
158  acc = _mm_add_ps(acc, _mm_shuffle_ps(acc, acc, _MM_SHUFFLE(1, 0, 3, 2)));
159  acc = _mm_add_ps(acc, _mm_shuffle_ps(acc, acc, _MM_SHUFFLE(0, 1, 0, 1)));
160  _mm_store_ss(&y[row], acc);
161  }
162 }

205 /**
206  * @brief Matrix-matrix multiply: C[M,N] = A[M,K] @ B[N,K]^T + bias
207  *
208  * @param A Input matrix [M x K], row-major FP32
209  * @param B Weight matrix in Q8_0 format, [N x K] stored row-major
210  * @param bias Optional bias [N], NULL if not used
211  * @param C Output [M x N], row-major FP32
212  * @param M Batch size (number of tokens)
213  * @param N Output dimension (number of rows in B)
214  * @param K Input dimension
215  */
216 void gemm_nt_q8_0(const float *A,
217  const void *B,
218  const float *bias,
219  float *C,
220  int M, int N, int K)
221 {
222 #if defined(__SSE4_1__)
223  // For now use gemv unrolled as it's safer
224  for (int m = 0; m < M; m++) {
225  gemv_q8_0_sse(&C[m * N], B, &A[m * K], N, K);
226  if (bias) {
227  for (int n = 0; n < N; n++) C[m * N + n] += bias[n];
228  }
229  }
230  return;
231 #endif
232 
233  const block_q8_0 *blocks = (const block_q8_0 *)B;
234  const int blocks_per_row = K / QK8_0;
235 
236  for (int m = 0; m < M; m++) {
237  const float *a_row = &A[m * K];
238 
239  for (int n = 0; n < N; n++) {
240  float sum = 0.0f;
241 
242  for (int b = 0; b < blocks_per_row; b++) {
243  const block_q8_0 *block = &blocks[n * blocks_per_row + b];
244  const float d = CK_FP16_TO_FP32(block->d);
245  const float *ap = &a_row[b * QK8_0];
246 
247  for (int i = 0; i < QK8_0; i++) {
248  sum += d * (float)block->qs[i] * ap[i];
249  }
250  }
251 
252  C[m * N + n] = sum + (bias ? bias[n] : 0.0f);
253  }
254  }

261 /**
262  * @brief Backward pass: compute input gradient (scalar reference)
263  *
264  * @param dX Output gradient w.r.t. input [K]
265  * @param W Weight matrix in Q8_0 format [M x K]
266  * @param dY Gradient w.r.t. output [M]
267  * @param M Number of output rows
268  * @param K Number of columns (input dimension)
269  */
270 void gemv_q8_0_backward_ref(float *dX,
271  const void *W,
272  const float *dY,
273  int M, int K)
274 {
275  const block_q8_0 *blocks = (const block_q8_0 *)W;
276  const int blocks_per_row = K / QK8_0;
277 
278  /* Zero output gradient */
279  memset(dX, 0, K * sizeof(float));
280 
281  /* Accumulate: dX += W^T @ dY */
282  for (int row = 0; row < M; row++) {
283  const float dy = dY[row];
284 
285  for (int b = 0; b < blocks_per_row; b++) {
286  const block_q8_0 *block = &blocks[row * blocks_per_row + b];
287  const float d = CK_FP16_TO_FP32(block->d);
288  float *dxp = &dX[b * QK8_0];
289 
290  for (int i = 0; i < QK8_0; i++) {
291  dxp[i] += d * (float)block->qs[i] * dy;
292  }
293  }
294  }
295 }

305 {
306  const block_q8_0 *blocks = (const block_q8_0 *)W;
307  const int blocks_per_row = K / QK8_0;
308 
309  /* Zero output */
310  memset(dX, 0, K * sizeof(float));
311 
312  for (int row = 0; row < M; row++) {
313  const __m512 vdy = _mm512_set1_ps(dY[row]);
314 
315  for (int b = 0; b < blocks_per_row; b++) {
316  const block_q8_0 *block = &blocks[row * blocks_per_row + b];
317  const __m512 vscale = _mm512_set1_ps(CK_FP16_TO_FP32(block->d));
318  float *dxp = &dX[b * QK8_0];
319 
320  /* Process 32 weights in two batches of 16 */
321  for (int chunk = 0; chunk < 2; chunk++) {
322  /* Load and dequantize weights */
323  __m128i q8 = _mm_loadu_si128((const __m128i *)&block->qs[chunk * 16]);
324  __m512i q32 = _mm512_cvtepi8_epi32(q8);
325  __m512 w = _mm512_mul_ps(_mm512_cvtepi32_ps(q32), vscale);
326 
327  /* Compute gradient */
328  __m512 grad = _mm512_mul_ps(w, vdy);
329 
330  /* Accumulate */
331  __m512 dx_cur = _mm512_loadu_ps(&dxp[chunk * 16]);
332  _mm512_storeu_ps(&dxp[chunk * 16], _mm512_add_ps(dx_cur, grad));
333  }
334  }
335  }
336 }
