30  * Fused RMSNorm + Q8_K Quantization
31  *
32  * Benefits:
33  * 1. Single pass over input data (reduces DRAM pressure)
34  * 2. Normalization results stay in registers for quantization
35  * 3. Keeps hot data in L1/L2 cache
36  */
37 void rmsnorm_q8_k_fused(const float *input,
38  const float *gamma,
39  void *vy,
40  int tokens,
41  int d_model,
42  int aligned_embed_dim,
43  float eps)
44 {
45  const int T = tokens;
46  const int D = d_model;
47  block_q8_K *y = (block_q8_K *)vy;
48 
49  for (int t = 0; t < T; ++t) {
50  const float *x = input + (size_t)t * aligned_embed_dim;
51 
52  // 1. Compute sum of squares using AVX
53  __m256 sum_sq_vec = _mm256_setzero_ps();
54  for (int d = 0; d < D; d += 8) {
55  __m256 xv = _mm256_loadu_ps(&x[d]);
56  sum_sq_vec = _mm256_add_ps(sum_sq_vec, _mm256_mul_ps(xv, xv));
57  }
58  float sum_sq = hsum256_ps_fused(sum_sq_vec);
59  float rstd = 1.0f / sqrtf(sum_sq / (float)D + eps);
60  __m256 vrstd = _mm256_set1_ps(rstd);
61 
62  // 2. We need the max absolute value of the NORMALIZED data for quantization
63  // y_i = gamma_i * (x_i * rstd)
64  // We do this in blocks of QK_K (256) to match Q8_K layout
65  for (int b = 0; b < D / QK_K; ++b) {
66  const float *xb = x + b * QK_K;
67  const float *gb = gamma + b * QK_K;
68  block_q8_K *out_block = &y[t * (D / QK_K) + b];
69 
70  // Local normalization and max search
71  __m256 v_max_abs = _mm256_setzero_ps();
72  float norm_buf[QK_K];
73 
74  for (int d = 0; d < QK_K; d += 8) {
75  __m256 xv = _mm256_loadu_ps(&xb[d]);
76  __m256 gv = _mm256_loadu_ps(&gb[d]);
77  __m256 normalized = _mm256_mul_ps(_mm256_mul_ps(xv, vrstd), gv);
78 
79  _mm256_storeu_ps(&norm_buf[d], normalized);
80 
81  __m256 v_abs = _mm256_andnot_ps(_mm256_set1_ps(-0.0f), normalized);
82  v_max_abs = _mm256_max_ps(v_max_abs, v_abs);
83  }
84 
85  float max_val = hmax256_ps_fused(v_max_abs);
86  if (max_val == 0.0f) {
87  out_block->d = 0.0f;
88  memset(out_block->qs, 0, QK_K);
89  memset(out_block->bsums, 0, sizeof(out_block->bsums));
90  continue;
91  }
92 
93  // 3. Quantize to Q8_K
94  float iscale = -127.0f / max_val;
95  __m256 v_iscale = _mm256_set1_ps(iscale);
96  out_block->d = 1.0f / iscale;
97 
98  for (int j = 0; j < QK_K; j += 16) {
99  // AVX1 doesn't have 256-bit integer conversion, so we use 128-bit SSE for packing
100  __m128 n0 = _mm_loadu_ps(&norm_buf[j + 0]);
101  __m128 n1 = _mm_loadu_ps(&norm_buf[j + 4]);
102  __m128 n2 = _mm_loadu_ps(&norm_buf[j + 8]);
103  __m128 n3 = _mm_loadu_ps(&norm_buf[j + 12]);
104 
105  __m128i q0 = _mm_cvtps_epi32(_mm_mul_ps(n0, _mm256_castps256_ps128(v_iscale)));
106  __m128i q1 = _mm_cvtps_epi32(_mm_mul_ps(n1, _mm256_castps256_ps128(v_iscale)));
107  __m128i q2 = _mm_cvtps_epi32(_mm_mul_ps(n2, _mm256_castps256_ps128(v_iscale)));
108  __m128i q3 = _mm_cvtps_epi32(_mm_mul_ps(n3, _mm256_castps256_ps128(v_iscale)));
109 
110  __m128i q01 = _mm_packs_epi32(q0, q1);
111  __m128i q23 = _mm_packs_epi32(q2, q3);
112  __m128i q0123 = _mm_packs_epi16(q01, q23);
113 
114  _mm_storeu_si128((__m128i *)(out_block->qs + j), q0123);
115 
116  // Compute bsum for 16 elements
117  __m128i p01 = _mm_add_epi16(q01, q23);
118  p01 = _mm_add_epi16(p01, _mm_shuffle_epi32(p01, _MM_SHUFFLE(1, 0, 3, 2)));
119  p01 = _mm_add_epi16(p01, _mm_shufflelo_epi16(p01, _MM_SHUFFLE(1, 0, 3, 2)));
120  int16_t bsum = (int16_t)_mm_extract_epi16(p01, 0) + (int16_t)_mm_extract_epi16(p01, 1);
