1/**
2 * @file optimizer_kernels.c
3 * @brief Optimizer kernels for training (AdamW, SGD)
4 *
5 * CK-ENGINE KERNEL RULES:
6 * =======================
7 * 1. NO malloc/free - memory via bump allocator, pointers passed in
8 * 2. NO OpenMP - parallelization at orchestrator/codegen layer
9 * 3. API must define: inputs, outputs, workspace, and memory layouts
10 * 4. Pure computation - deterministic, no side effects
11 *
12 * After changes: make test && make llamacpp-parity-full
13 *
14 * AdamW Algorithm:
15 * m_t = beta1 * m_{t-1} + (1 - beta1) * g_t
16 * v_t = beta2 * v_{t-1} + (1 - beta2) * g_t^2
17 * m_hat = m_t / (1 - beta1^t)
18 * v_hat = v_t / (1 - beta2^t)
19 * w_t = w_{t-1} - lr * (m_hat / (sqrt(v_hat) + eps) + weight_decay * w_{t-1})
20 *
21 * Note: AdamW applies weight decay directly to weights, not to gradients.
22 * This is different from L2 regularization (Adam with L2 adds decay to gradient).
23 */

189static void ck_adamw_multi_parallel_work(int ith, int nth, void *argp)
190{
191 ck_adamw_multi_parallel_args_t *a = (ck_adamw_multi_parallel_args_t *)argp;
192 if (!a || !a->grads || !a->weights || !a->m_states || !a->v_states || !a->numels ||
193 a->tensor_count <= 0) {
194 return;
195 }
196
197 const int per = (a->tensor_count + nth - 1) / nth;
198 const int t0 = per * ith;
199 int t1 = t0 + per;
200 if (t0 >= a->tensor_count) {
201 return;
202 }
203 if (t1 > a->tensor_count) {
204 t1 = a->tensor_count;
205 }
206
207 for (int ti = t0; ti < t1; ++ti) {
208 float *g = a->grads[ti];
209 float *w = a->weights[ti];
210 float *m = a->m_states[ti];
211 float *v = a->v_states[ti];
212 size_t n = a->numels[ti];
213 if (!g || !w || !m || !v || n == 0) {
214 continue;
215 }
216
217 if (a->max_grad_norm > 0.0f) {
218 double sum_sq = gradient_sum_sq_f32_impl(g, n);
219 float norm = sqrtf((float)sum_sq);
220 if (norm > a->max_grad_norm) {
221 float scale = a->max_grad_norm / norm;
222 gradient_scale_f32_impl(g, n, scale);
223 }
224 }

237 a->step);

242/**
243 * @brief AdamW optimizer update (fp32 version)
244 *
245 * Updates weights in-place using the AdamW algorithm.
246 * Momentum (m) and variance (v) are stored in fp32 for numerical stability.
247 *
248 * @param grad Gradient tensor (fp32) [numel]
249 * @param weight Weight tensor to update (fp32, in-place) [numel]
250 * @param m First moment (momentum) buffer (fp32, in-place) [numel]
251 * @param v Second moment (variance) buffer (fp32, in-place) [numel]
252 * @param numel Number of elements
253 * @param lr Learning rate
254 * @param beta1 Exponential decay rate for first moment (typically 0.9)
255 * @param beta2 Exponential decay rate for second moment (typically 0.999)
256 * @param eps Small constant for numerical stability (typically 1e-8)
257 * @param weight_decay Weight decay coefficient (typically 0.01)
258 * @param step Current step number (1-indexed for bias correction)
259 */
260static void adamw_update_f32_impl(
261 const float *grad,
262 float *weight,
263 float *m,
264 float *v,
265 size_t numel,
266 float lr,
267 float beta1,
268 float beta2,
269 float eps,
270 float weight_decay,
271 int step)
272{
273 if (!grad || !weight || !m || !v || numel == 0) {
274 return;
275 }
276
277 // Bias correction terms
278 float bias_correction1 = 1.0f - powf(beta1, (float)step);
279 float bias_correction2 = 1.0f - powf(beta2, (float)step);
280
281 // Precompute constants
282 float one_minus_beta1 = 1.0f - beta1;
283 float one_minus_beta2 = 1.0f - beta2;
284
285 if (ck_strict_parity_enabled()) {
286 const double beta1_d = (double)beta1;
287 const double beta2_d = (double)beta2;
288 const double one_minus_beta1_d = 1.0 - beta1_d;
289 const double one_minus_beta2_d = 1.0 - beta2_d;
290 const double lr_d = (double)lr;
291 const double eps_d = (double)eps;
292 const double wd_d = (double)weight_decay;
293
294 const double bc1 = 1.0 - pow(beta1_d, (double)step);
295 const double bc2 = 1.0 - pow(beta2_d, (double)step);
296 const double step_size = lr_d / bc1;
297 const double bc2_sqrt = sqrt(bc2);
298 const double wd_scale = 1.0 - lr_d * wd_d;
299
300 for (size_t i = 0; i < numel; ++i) {
301 double g = (double)grad[i];
302 double w = (double)weight[i];
303 double m_i = (double)m[i];
304 double v_i = (double)v[i];
305
306 m_i = beta1_d * m_i + one_minus_beta1_d * g;
307 v_i = beta2_d * v_i + one_minus_beta2_d * g * g;
308
309 // Match PyTorch AdamW op order: decoupled weight decay first.
310 w *= wd_scale;
311
312 double denom = sqrt(v_i) / bc2_sqrt + eps_d;
313 w -= step_size * (m_i / denom);
314
315 m[i] = (float)m_i;
316 v[i] = (float)v_i;
317 weight[i] = (float)w;
318 }
319 return;
320 }
321
322#if defined(__AVX512F__)
323 // AVX-512 path: process 16 floats at a time
324 __m512 v_beta1 = _mm512_set1_ps(beta1);
325 __m512 v_beta2 = _mm512_set1_ps(beta2);
326 __m512 v_one_minus_beta1 = _mm512_set1_ps(one_minus_beta1);
327 __m512 v_one_minus_beta2 = _mm512_set1_ps(one_minus_beta2);
328 __m512 v_lr = _mm512_set1_ps(lr);
329 __m512 v_eps = _mm512_set1_ps(eps);
330 __m512 v_weight_decay = _mm512_set1_ps(weight_decay);
331 __m512 v_bc1_inv = _mm512_set1_ps(1.0f / bias_correction1);
332 __m512 v_bc2_inv = _mm512_set1_ps(1.0f / bias_correction2);
333
334 size_t i = 0;
335 for (; i + 16 <= numel; i += 16) {
336 __m512 g = _mm512_loadu_ps(&grad[i]);
337 __m512 w = _mm512_loadu_ps(&weight[i]);
338 __m512 m_val = _mm512_loadu_ps(&m[i]);
339 __m512 v_val = _mm512_loadu_ps(&v[i]);
340
341 // m = beta1 * m + (1 - beta1) * g
342 m_val = _mm512_fmadd_ps(v_beta1, m_val, _mm512_mul_ps(v_one_minus_beta1, g));
343
344 // v = beta2 * v + (1 - beta2) * g^2
345 __m512 g_sq = _mm512_mul_ps(g, g);
346 v_val = _mm512_fmadd_ps(v_beta2, v_val, _mm512_mul_ps(v_one_minus_beta2, g_sq));
347
348 // Bias-corrected estimates
349 __m512 m_hat = _mm512_mul_ps(m_val, v_bc1_inv);
350 __m512 v_hat = _mm512_mul_ps(v_val, v_bc2_inv);
351
352 // w = w - lr * (m_hat / (sqrt(v_hat) + eps) + weight_decay * w)
353 __m512 denom = _mm512_add_ps(_mm512_sqrt_ps(v_hat), v_eps);
354 __m512 update = _mm512_div_ps(m_hat, denom);
355 update = _mm512_fmadd_ps(v_weight_decay, w, update);
356 w = _mm512_fnmadd_ps(v_lr, update, w);
357
358 _mm512_storeu_ps(&weight[i], w);
359 _mm512_storeu_ps(&m[i], m_val);
360 _mm512_storeu_ps(&v[i], v_val);
361 }
362
363 // Scalar tail
364 for (; i < numel; ++i) {
365 float g = grad[i];
366 float w = weight[i];
367 m[i] = beta1 * m[i] + one_minus_beta1 * g;
368 v[i] = beta2 * v[i] + one_minus_beta2 * g * g;
369 float m_hat = m[i] / bias_correction1;
370 float v_hat = v[i] / bias_correction2;
371 weight[i] = w - lr * (m_hat / (sqrtf(v_hat) + eps) + weight_decay * w);
372 }
373
374#elif defined(__AVX__)
375 // AVX path: process 8 floats at a time (no FMA on older CPUs like Ivy Bridge)
376 __m256 v_beta1 = _mm256_set1_ps(beta1);
377 __m256 v_beta2 = _mm256_set1_ps(beta2);
378 __m256 v_one_minus_beta1 = _mm256_set1_ps(one_minus_beta1);
379 __m256 v_one_minus_beta2 = _mm256_set1_ps(one_minus_beta2);
380 __m256 v_lr = _mm256_set1_ps(lr);
381 __m256 v_eps = _mm256_set1_ps(eps);
382 __m256 v_weight_decay = _mm256_set1_ps(weight_decay);
383 __m256 v_bc1_inv = _mm256_set1_ps(1.0f / bias_correction1);
384 __m256 v_bc2_inv = _mm256_set1_ps(1.0f / bias_correction2);
385
386 size_t i = 0;
387 for (; i + 8 <= numel; i += 8) {
388 __m256 g = _mm256_loadu_ps(&grad[i]);
389 __m256 w = _mm256_loadu_ps(&weight[i]);
390 __m256 m_val = _mm256_loadu_ps(&m[i]);
391 __m256 v_val = _mm256_loadu_ps(&v[i]);
392
393 // m = beta1 * m + (1 - beta1) * g
394 m_val = _mm256_add_ps(_mm256_mul_ps(v_beta1, m_val),
395 _mm256_mul_ps(v_one_minus_beta1, g));
396
397 // v = beta2 * v + (1 - beta2) * g^2
398 __m256 g_sq = _mm256_mul_ps(g, g);
399 v_val = _mm256_add_ps(_mm256_mul_ps(v_beta2, v_val),
400 _mm256_mul_ps(v_one_minus_beta2, g_sq));
401
402 // Bias-corrected estimates
403 __m256 m_hat = _mm256_mul_ps(m_val, v_bc1_inv);
404 __m256 v_hat = _mm256_mul_ps(v_val, v_bc2_inv);
405
406 // w = w - lr * (m_hat / (sqrt(v_hat) + eps) + weight_decay * w)
407 __m256 denom = _mm256_add_ps(_mm256_sqrt_ps(v_hat), v_eps);
408 __m256 update = _mm256_div_ps(m_hat, denom);
409 update = _mm256_add_ps(update, _mm256_mul_ps(v_weight_decay, w));
410 w = _mm256_sub_ps(w, _mm256_mul_ps(v_lr, update));
411
412 _mm256_storeu_ps(&weight[i], w);
413 _mm256_storeu_ps(&m[i], m_val);
414 _mm256_storeu_ps(&v[i], v_val);
415 }
416
417 // Scalar tail
418 for (; i < numel; ++i) {
419 float g = grad[i];
420 float w = weight[i];
421 m[i] = beta1 * m[i] + one_minus_beta1 * g;
422 v[i] = beta2 * v[i] + one_minus_beta2 * g * g;
423 float m_hat = m[i] / bias_correction1;
424 float v_hat = v[i] / bias_correction2;
425 weight[i] = w - lr * (m_hat / (sqrtf(v_hat) + eps) + weight_decay * w);
426 }
427
428#elif defined(__SSE2__)
429 // SSE2 path: process 4 floats at a time
430 __m128 v_beta1 = _mm_set1_ps(beta1);
431 __m128 v_beta2 = _mm_set1_ps(beta2);
432 __m128 v_one_minus_beta1 = _mm_set1_ps(one_minus_beta1);
433 __m128 v_one_minus_beta2 = _mm_set1_ps(one_minus_beta2);
434 __m128 v_lr = _mm_set1_ps(lr);
435 __m128 v_eps = _mm_set1_ps(eps);
436 __m128 v_weight_decay = _mm_set1_ps(weight_decay);
437 __m128 v_bc1_inv = _mm_set1_ps(1.0f / bias_correction1);
438 __m128 v_bc2_inv = _mm_set1_ps(1.0f / bias_correction2);
439
440 size_t i = 0;
441 for (; i + 4 <= numel; i += 4) {
442 __m128 g = _mm_loadu_ps(&grad[i]);
443 __m128 w = _mm_loadu_ps(&weight[i]);
444 __m128 m_val = _mm_loadu_ps(&m[i]);
445 __m128 v_val = _mm_loadu_ps(&v[i]);
446
447 // m = beta1 * m + (1 - beta1) * g
448 m_val = _mm_add_ps(_mm_mul_ps(v_beta1, m_val),
449 _mm_mul_ps(v_one_minus_beta1, g));
450
451 // v = beta2 * v + (1 - beta2) * g^2
452 __m128 g_sq = _mm_mul_ps(g, g);
453 v_val = _mm_add_ps(_mm_mul_ps(v_beta2, v_val),
454 _mm_mul_ps(v_one_minus_beta2, g_sq));
455
456 // Bias-corrected estimates
457 __m128 m_hat = _mm_mul_ps(m_val, v_bc1_inv);
458 __m128 v_hat = _mm_mul_ps(v_val, v_bc2_inv);
459
460 // w = w - lr * (m_hat / (sqrt(v_hat) + eps) + weight_decay * w)
461 __m128 denom = _mm_add_ps(_mm_sqrt_ps(v_hat), v_eps);
462 __m128 update = _mm_div_ps(m_hat, denom);
463 update = _mm_add_ps(update, _mm_mul_ps(v_weight_decay, w));
464 w = _mm_sub_ps(w, _mm_mul_ps(v_lr, update));
465
466 _mm_storeu_ps(&weight[i], w);
467 _mm_storeu_ps(&m[i], m_val);
468 _mm_storeu_ps(&v[i], v_val);
469 }
470
471 // Scalar tail
472 for (; i < numel; ++i) {
473 float g = grad[i];
474 float w = weight[i];
475 m[i] = beta1 * m[i] + one_minus_beta1 * g;
476 v[i] = beta2 * v[i] + one_minus_beta2 * g * g;
477 float m_hat = m[i] / bias_correction1;
478 float v_hat = v[i] / bias_correction2;
479 weight[i] = w - lr * (m_hat / (sqrtf(v_hat) + eps) + weight_decay * w);
480 }
481
482#else
483 // Scalar path
484 for (size_t i = 0; i < numel; ++i) {
485 float g = grad[i];
486 float w = weight[i];
487 m[i] = beta1 * m[i] + one_minus_beta1 * g;
488 v[i] = beta2 * v[i] + one_minus_beta2 * g * g;
489 float m_hat = m[i] / bias_correction1;
490 float v_hat = v[i] / bias_correction2;
491 weight[i] = w - lr * (m_hat / (sqrtf(v_hat) + eps) + weight_decay * w);
492 }

494}

497void adamw_update_f32(
498 const float *grad,
499 float *weight,
500 float *m,
501 float *v,
502 size_t numel,
503 float lr,
504 float beta1,
505 float beta2,
506 float eps,
507 float weight_decay,
508 int step)
509{
510 if (!grad || !weight || !m || !v || numel == 0) {
511 return;
512 }
513
514 ck_threadpool_t *pool = ck_threadpool_global();
515 int nth = pool ? ck_threadpool_n_threads(pool) : 1;
516 if (!pool || nth <= 1 || nth > CK_OPT_PAR_MAX_THREADS || numel < CK_OPT_PAR_MIN_NUMEL) {
517 adamw_update_f32_impl(grad, weight, m, v, numel, lr, beta1, beta2, eps, weight_decay, step);
518 return;
519 }
520
521 ck_adamw_parallel_args_t args = {
522 .grad = grad,
523 .weight = weight,
524 .m = m,
525 .v = v,
526 .numel = numel,
527 .lr = lr,
528 .beta1 = beta1,
529 .beta2 = beta2,
530 .eps = eps,
531 .weight_decay = weight_decay,
532 .step = step,
533 };
534 ck_threadpool_dispatch(pool, ck_adamw_parallel_work, &args);
535}

538void adamw_clip_update_multi_f32(
539 float *const *grads,
540 float *const *weights,
541 float *const *m_states,
542 float *const *v_states,
543 const size_t *numels,
544 int tensor_count,
545 float lr,
546 float beta1,
547 float beta2,
548 float eps,
549 float weight_decay,
550 float max_grad_norm,
551 int step)
552{
553 if (!grads || !weights || !m_states || !v_states || !numels || tensor_count <= 0) {
554 return;
555 }
556
557 size_t total_numel = 0;
558 int valid_tensors = 0;
559 for (int i = 0; i < tensor_count; ++i) {
560 if (grads[i] && weights[i] && m_states[i] && v_states[i] && numels[i] > 0) {
561 total_numel += numels[i];
562 valid_tensors += 1;
563 }
564 }
565 if (valid_tensors == 0 || total_numel == 0) {
566 return;
567 }
568
569 ck_threadpool_t *pool = ck_threadpool_global();
570 int nth = pool ? ck_threadpool_n_threads(pool) : 1;
571
572 if (!pool || nth <= 1 || nth > CK_OPT_PAR_MAX_THREADS ||
573 total_numel < CK_OPT_PAR_MIN_NUMEL || valid_tensors < 2) {
574 for (int i = 0; i < tensor_count; ++i) {
575 float *g = grads[i];
576 float *w = weights[i];
577 float *m = m_states[i];
578 float *v = v_states[i];
579 size_t n = numels[i];
580 if (!g || !w || !m || !v || n == 0) {
581 continue;
582 }
583 if (max_grad_norm > 0.0f) {
584 double sum_sq = gradient_sum_sq_f32_impl(g, n);
585 float norm = sqrtf((float)sum_sq);
586 if (norm > max_grad_norm) {

615/**
616 * @brief SGD with momentum optimizer update (fp32 version)
617 *
618 * v_t = momentum * v_{t-1} + g_t
619 * w_t = w_{t-1} - lr * (v_t + weight_decay * w_{t-1})
620 *
621 * @param grad Gradient tensor (fp32) [numel]
622 * @param weight Weight tensor to update (fp32, in-place) [numel]
623 * @param velocity Velocity buffer (fp32, in-place) [numel]
624 * @param numel Number of elements
625 * @param lr Learning rate
626 * @param momentum Momentum coefficient (typically 0.9)
627 * @param weight_decay Weight decay coefficient
628 */
629void sgd_momentum_update_f32(
630 const float *grad,
631 float *weight,
632 float *velocity,
633 size_t numel,
634 float lr,
635 float momentum,
636 float weight_decay)
637{
638 if (!grad || !weight || !velocity || numel == 0) {
639 return;
640 }
641
642#if defined(__AVX512F__)
643 // AVX-512 path: process 16 floats at a time
644 __m512 v_lr = _mm512_set1_ps(lr);
645 __m512 v_momentum = _mm512_set1_ps(momentum);
646 __m512 v_weight_decay = _mm512_set1_ps(weight_decay);
647
648 size_t i = 0;
649 for (; i + 16 <= numel; i += 16) {
650 __m512 g = _mm512_loadu_ps(&grad[i]);
651 __m512 w = _mm512_loadu_ps(&weight[i]);
652 __m512 vel = _mm512_loadu_ps(&velocity[i]);
653
654 vel = _mm512_fmadd_ps(v_momentum, vel, g);
655 __m512 update = _mm512_fmadd_ps(v_weight_decay, w, vel);
656 w = _mm512_fnmadd_ps(v_lr, update, w);
657
658 _mm512_storeu_ps(&weight[i], w);
659 _mm512_storeu_ps(&velocity[i], vel);
660 }
661
662 for (; i < numel; ++i) {
663 velocity[i] = momentum * velocity[i] + grad[i];
664 weight[i] = weight[i] - lr * (velocity[i] + weight_decay * weight[i]);
665 }
666
667#elif defined(__AVX__)
668 // AVX path: process 8 floats at a time
669 __m256 v_lr = _mm256_set1_ps(lr);
670 __m256 v_momentum = _mm256_set1_ps(momentum);
671 __m256 v_weight_decay = _mm256_set1_ps(weight_decay);
672
673 size_t i = 0;
674 for (; i + 8 <= numel; i += 8) {
675 __m256 g = _mm256_loadu_ps(&grad[i]);
676 __m256 w = _mm256_loadu_ps(&weight[i]);
677 __m256 vel = _mm256_loadu_ps(&velocity[i]);
678
679 // v = momentum * v + g
680 vel = _mm256_add_ps(_mm256_mul_ps(v_momentum, vel), g);
681
682 // w = w - lr * (v + weight_decay * w)
683 __m256 update = _mm256_add_ps(vel, _mm256_mul_ps(v_weight_decay, w));
684 w = _mm256_sub_ps(w, _mm256_mul_ps(v_lr, update));
685
686 _mm256_storeu_ps(&weight[i], w);
687 _mm256_storeu_ps(&velocity[i], vel);
688 }
689
690 for (; i < numel; ++i) {
691 velocity[i] = momentum * velocity[i] + grad[i];
692 weight[i] = weight[i] - lr * (velocity[i] + weight_decay * weight[i]);
693 }
694
695#elif defined(__SSE2__)
696 // SSE2 path: process 4 floats at a time
697 __m128 v_lr = _mm_set1_ps(lr);
698 __m128 v_momentum = _mm_set1_ps(momentum);
699 __m128 v_weight_decay = _mm_set1_ps(weight_decay);
700
701 size_t i = 0;
702 for (; i + 4 <= numel; i += 4) {
703 __m128 g = _mm_loadu_ps(&grad[i]);
704 __m128 w = _mm_loadu_ps(&weight[i]);
705 __m128 vel = _mm_loadu_ps(&velocity[i]);
706
707 vel = _mm_add_ps(_mm_mul_ps(v_momentum, vel), g);
708 __m128 update = _mm_add_ps(vel, _mm_mul_ps(v_weight_decay, w));
709 w = _mm_sub_ps(w, _mm_mul_ps(v_lr, update));
710
711 _mm_storeu_ps(&weight[i], w);
712 _mm_storeu_ps(&velocity[i], vel);
713 }
714
715 for (; i < numel; ++i) {
716 velocity[i] = momentum * velocity[i] + grad[i];
717 weight[i] = weight[i] - lr * (velocity[i] + weight_decay * weight[i]);
718 }
719
720#else
721 // Scalar path
722 for (size_t i = 0; i < numel; ++i) {
723 velocity[i] = momentum * velocity[i] + grad[i];
724 weight[i] = weight[i] - lr * (velocity[i] + weight_decay * weight[i]);
725 }
726#endif
727}

900static double gradient_sum_sq_f32_impl(const float *grad, size_t numel)
901{
902 if (!grad || numel == 0) {
903 return 0.0;
904 }
905
906 double sum_sq = 0.0;
907#if defined(__AVX512F__)
908 __m512 acc = _mm512_setzero_ps();
909 size_t i = 0;
910 for (; i + 16 <= numel; i += 16) {
911 __m512 g = _mm512_loadu_ps(&grad[i]);
912 acc = _mm512_fmadd_ps(g, g, acc);
913 }
914 sum_sq = _mm512_reduce_add_ps(acc);
915 for (; i < numel; ++i) {
916 sum_sq += (double)grad[i] * (double)grad[i];
917 }
918#elif defined(__AVX__)
919 __m256 acc = _mm256_setzero_ps();
920 size_t i = 0;
921 for (; i + 8 <= numel; i += 8) {
922 __m256 g = _mm256_loadu_ps(&grad[i]);
923 acc = _mm256_add_ps(acc, _mm256_mul_ps(g, g));
924 }
925 __m128 hi = _mm256_extractf128_ps(acc, 1);
926 __m128 lo = _mm256_castps256_ps128(acc);
927 __m128 sum4 = _mm_add_ps(lo, hi);
928 __m128 shuf = _mm_movehdup_ps(sum4);
929 __m128 sums = _mm_add_ps(sum4, shuf);
930 shuf = _mm_movehl_ps(shuf, sums);
931 sums = _mm_add_ss(sums, shuf);
932 sum_sq = _mm_cvtss_f32(sums);
933 for (; i < numel; ++i) {
934 sum_sq += (double)grad[i] * (double)grad[i];
935 }
936#elif defined(__SSE2__)
937 __m128 acc = _mm_setzero_ps();
938 size_t i = 0;
939 for (; i + 4 <= numel; i += 4) {
940 __m128 g = _mm_loadu_ps(&grad[i]);
941 acc = _mm_add_ps(acc, _mm_mul_ps(g, g));
942 }
943 __m128 shuf = _mm_shuffle_ps(acc, acc, _MM_SHUFFLE(2, 3, 0, 1));
944 __m128 sums = _mm_add_ps(acc, shuf);
945 shuf = _mm_movehl_ps(shuf, sums);
946 sums = _mm_add_ss(sums, shuf);
947 sum_sq = _mm_cvtss_f32(sums);
948 for (; i < numel; ++i) {
949 sum_sq += (double)grad[i] * (double)grad[i];
950 }
951#else
952 for (size_t i = 0; i < numel; ++i) {
953 sum_sq += (double)grad[i] * (double)grad[i];
954 }
955#endif
956 return sum_sq;
957}

961 * @brief Clip gradient norm (fp32)
962 *
963 * If ||grad||_2 > max_norm, scale grad so that ||grad||_2 = max_norm
964 *
965 * @param grad Gradient tensor to clip (in-place) [numel]
966 * @param numel Number of elements
967 * @param max_norm Maximum allowed L2 norm
968 * @return The original L2 norm before clipping
969 */
970float gradient_clip_norm_f32(float *grad, size_t numel, float max_norm)
971{
972 if (!grad || numel == 0 || max_norm <= 0.0f) {
973 return 0.0f;
974 }
975
976 double sum_sq = 0.0;
977 ck_threadpool_t *pool = ck_threadpool_global();
978 int nth = pool ? ck_threadpool_n_threads(pool) : 1;
979
980 if (pool && nth > 1 && nth <= CK_OPT_PAR_MAX_THREADS && numel >= CK_OPT_PAR_MIN_NUMEL) {
981 ck_sum_sq_parallel_args_t args;
982 args.grad = grad;
983 args.numel = numel;
984 for (int i = 0; i < CK_OPT_PAR_MAX_THREADS; ++i) {
985 args.partial[i] = 0.0;
986 }
987 ck_threadpool_dispatch(pool, ck_sum_sq_parallel_work, &args);
988 for (int i = 0; i < nth; ++i) {
989 sum_sq += args.partial[i];
990 }
991 } else {
992 sum_sq = gradient_sum_sq_f32_impl(grad, numel);
993 }
994
995 float norm = sqrtf((float)sum_sq);
996 if (norm > max_norm) {
997 float scale = max_norm / norm;
998 gradient_scale_f32(grad, numel, scale);
999 }
1000 return norm;
1001}
