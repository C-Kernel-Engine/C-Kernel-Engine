309 float *output,

311{
312 /* Allocate Q8_K buffer for quantized activations */

324 }
325
326 /* Use gemm_nt_q4_k_q8_k: C[M,N] = A[M,K] * B[N,K]^T
327 * Our layout: output[n_tokens, rows] = input[n_tokens, cols] * weight[rows, cols]^T
328 * So: M = n_tokens, N = rows, K = cols
329 */
330 gemm_nt_q4_k_q8_k(q8_data, weight_q4k, NULL, output, n_tokens, rows, cols);
331
332 free(q8_data);

448void ck_test_rope(float *q, float *k,
449 int n_tokens, int n_heads, int n_heads_kv, int head_dim,
450 int pos_offset, float theta)
451{
452 /* Precompute cos/sin cache */
453 int half_dim = head_dim / 2;
454 int max_seq = pos_offset + n_tokens;
455
456 float *cos_cache = (float *)malloc(max_seq * half_dim * sizeof(float));
457 float *sin_cache = (float *)malloc(max_seq * half_dim * sizeof(float));
458 if (!cos_cache || !sin_cache) {
459 free(cos_cache);
460 free(sin_cache);
461 return;
462 }
463
464 rope_precompute_cache(cos_cache, sin_cache, max_seq, head_dim, theta,
465 head_dim, "none", 1.0f);
466
467 /* CK RoPE expects layout [num_heads, num_tokens, head_dim]
468 * Reshape from [n_tokens, n_heads * head_dim] to [n_heads, n_tokens, head_dim]
469 */
470 float *q_reorder = (float *)malloc(n_heads * n_tokens * head_dim * sizeof(float));
471 float *k_reorder = (float *)malloc(n_heads_kv * n_tokens * head_dim * sizeof(float));
472
473 if (q_reorder && k_reorder) {
474 /* Reorder Q: [T, H*D] -> [H, T, D] */
475 for (int t = 0; t < n_tokens; t++) {
476 for (int h = 0; h < n_heads; h++) {
477 for (int d = 0; d < head_dim; d++) {
478 q_reorder[h * n_tokens * head_dim + t * head_dim + d] =
479 q[t * n_heads * head_dim + h * head_dim + d];
480 }
481 }
482 }
483
484 /* Reorder K: [T, H_kv*D] -> [H_kv, T, D] */
485 for (int t = 0; t < n_tokens; t++) {
486 for (int h = 0; h < n_heads_kv; h++) {
487 for (int d = 0; d < head_dim; d++) {
488 k_reorder[h * n_tokens * head_dim + t * head_dim + d] =
489 k[t * n_heads_kv * head_dim + h * head_dim + d];
490 }
491 }
492 }
493
494 /* Apply RoPE */
495 rope_forward_qk(q_reorder, k_reorder,
496 cos_cache, sin_cache,
497 n_heads, n_heads_kv, n_tokens,
498 head_dim, head_dim, pos_offset);
499
500 /* Reorder back: [H, T, D] -> [T, H*D] */
501 for (int t = 0; t < n_tokens; t++) {
502 for (int h = 0; h < n_heads; h++) {
503 for (int d = 0; d < head_dim; d++) {
504 q[t * n_heads * head_dim + h * head_dim + d] =
505 q_reorder[h * n_tokens * head_dim + t * head_dim + d];
506 }
507 }
508 }
509
510 for (int t = 0; t < n_tokens; t++) {
511 for (int h = 0; h < n_heads_kv; h++) {
512 for (int d = 0; d < head_dim; d++) {
513 k[t * n_heads_kv * head_dim + h * head_dim + d] =
514 k_reorder[h * n_tokens * head_dim + t * head_dim + d];
515 }
516 }
517 }
518 }
519
520 free(q_reorder);
521 free(k_reorder);
522 free(cos_cache);
523 free(sin_cache);
524}

526void ck_test_rope_interleaved(float *q, float *k,
527 int n_tokens, int n_heads, int n_heads_kv, int head_dim,
528 int pos_offset, float theta)
529{
530 /* Interleaved RoPE format (matches llama.cpp):
531 * (x0, x1) -> (x0*cos - x1*sin, x0*sin + x1*cos)
532 * Applied to consecutive pairs of elements
533 */
534
535 /* Precompute inverse frequencies */
536 float *inv_freq = (float *)malloc((head_dim / 2) * sizeof(float));
537 if (!inv_freq) return;
538
539 for (int i = 0; i < head_dim / 2; i++) {
540 inv_freq[i] = 1.0f / powf(theta, (float)(2 * i) / head_dim);
541 }
542
543 /* Apply RoPE to Q */
544 for (int t = 0; t < n_tokens; t++) {
545 int pos = pos_offset + t;
546 for (int h = 0; h < n_heads; h++) {
547 float *qh = q + t * n_heads * head_dim + h * head_dim;
548
549 for (int i = 0; i < head_dim / 2; i++) {
550 float freq = pos * inv_freq[i];
551 float cos_val = cosf(freq);
552 float sin_val = sinf(freq);
553
554 /* Interleaved format */
555 float x0 = qh[i * 2];
556 float x1 = qh[i * 2 + 1];
557 qh[i * 2] = x0 * cos_val - x1 * sin_val;
558 qh[i * 2 + 1] = x0 * sin_val + x1 * cos_val;
559 }
560 }
561 }
562
563 /* Apply RoPE to K */
564 for (int t = 0; t < n_tokens; t++) {
565 int pos = pos_offset + t;
566 for (int h = 0; h < n_heads_kv; h++) {
567 float *kh = k + t * n_heads_kv * head_dim + h * head_dim;
568
569 for (int i = 0; i < head_dim / 2; i++) {
570 float freq = pos * inv_freq[i];
571 float cos_val = cosf(freq);
572 float sin_val = sinf(freq);
573
574 float x0 = kh[i * 2];
575 float x1 = kh[i * 2 + 1];
576 kh[i * 2] = x0 * cos_val - x1 * sin_val;
577 kh[i * 2 + 1] = x0 * sin_val + x1 * cos_val;
578 }
579 }
580 }
581
582 free(inv_freq);
583}

592void ck_test_softmax(const float *input, float *output, int n)
593{
594 /* Find max for numerical stability */
595 float max_val = input[0];
596 for (int i = 1; i < n; i++) {
597 if (input[i] > max_val) max_val = input[i];
598 }
599
600 /* Compute exp and sum */
601 float sum = 0.0f;
602 for (int i = 0; i < n; i++) {
603 output[i] = expf(input[i] - max_val);
604 sum += output[i];
605 }
606
607 /* Normalize */
608 float inv_sum = 1.0f / sum;
609 for (int i = 0; i < n; i++) {
610 output[i] *= inv_sum;
611 }
612}

671 * @brief Test mega-fused OutProj + MLP kernel (Q5_0 weights)
672 *
673 * This is a simplified wrapper for parity testing that:
674 * - Uses Q5_0 for W_o and W1 weights
675 * - Uses Q4_K for W2 weights
676 * - Allocates scratch internally
677 *
678 * @param attn_out Attention output [num_heads, tokens, head_dim] (FP32, head-major)
679 * @param residual Residual input [tokens, embed_dim] (FP32)
680 * @param ln2_gamma RMSNorm gamma [embed_dim] (FP32)
681 * @param wo OutProj weights [embed_dim, embed_dim] (Q5_0)
682 * @param w1 MLP W1 weights [2*intermediate, embed_dim] (Q5_0)
683 * @param w2 MLP W2 weights [embed_dim, intermediate] (Q4_K or Q6_K)
684 * @param output Output [tokens, embed_dim] (FP32)
685 * @param tokens Number of tokens
686 * @param num_heads Number of attention heads
687 * @param head_dim Dimension per head
688 * @param embed_dim Embedding dimension (= num_heads * head_dim)
689 * @param intermediate MLP intermediate dimension
690 * @param eps RMSNorm epsilon
691 * @param w2_is_q6k If true, W2 is Q6_K; if false, W2 is Q4_K
692 */
693void ck_test_outproj_mlp_fused_q5_0(
694 const float *attn_out,
695 const float *residual,
696 const float *ln2_gamma,
697 const void *wo,
698 const void *w1,
699 const void *w2,
700 float *output,
701 int tokens,
702 int num_heads,
703 int head_dim,
704 int embed_dim,
705 int intermediate,
706 float eps,
707 int w2_is_q6k)
708{
709 /* CK uses dtype enum: CK_DT_Q5_0 = 11, CK_DT_Q4_K = 7, CK_DT_Q6_K = 8 */
710 const int CK_DT_Q5_0_VAL = 11;
711 const int CK_DT_Q4_K_VAL = 7;
712 const int CK_DT_Q6_K_VAL = 8;
713
714 /* For parity testing, aligned = actual (no padding) */
715 int aligned_embed_dim = embed_dim;
716 int aligned_head_dim = head_dim;
717 int aligned_intermediate = intermediate;
718
719 /* Ensure intermediate is multiple of 256 (QK_K) for K-quants */
720 if ((intermediate % 256) != 0) {
721 aligned_intermediate = ((intermediate + 255) / 256) * 256;
722 }
723
724 /* Allocate scratch */
725 size_t scratch_size = mega_fused_outproj_mlp_prefill_scratch_size(
726 tokens, aligned_embed_dim, num_heads, aligned_head_dim, aligned_intermediate);
727
728 void *scratch = malloc(scratch_size);
729 if (!scratch) {
730 return;
731 }
732
733 /* Call the mega-fused kernel */
734 mega_fused_outproj_mlp_prefill(
735 output,
736 attn_out,
737 residual,
738 ln2_gamma,
739 wo, NULL, CK_DT_Q5_0_VAL, /* W_o with Q5_0 */
740 w1, NULL, CK_DT_Q5_0_VAL, /* W1 with Q5_0 */
741 w2, NULL, w2_is_q6k ? CK_DT_Q6_K_VAL : CK_DT_Q4_K_VAL, /* W2 with Q4_K or Q6_K */
742 tokens,
743 embed_dim,
744 aligned_embed_dim,
745 num_heads,
746 aligned_head_dim,
747 intermediate,
748 aligned_intermediate,
749 eps,
750 scratch
751 );
752
753 free(scratch);
754}
