79static int ck_attention_oracle_exact_dump_layer(int *layer_id_out)
80{
81 const char *v = getenv("CK_STRICT_ATTN_DUMP_LAYER");
82 if (!v || !v[0]) {
83 return 0;
84 }
85 char *end = NULL;
86 long parsed = strtol(v, &end, 10);
87 if (end == v || (end && *end != '\0') || parsed < 0 || parsed > INT32_MAX) {
88 return 0;
89 }
90 if (layer_id_out) {
91 *layer_id_out = (int) parsed;
92 }
93 return 1;
94}

122static float ck_attention_oracle_tensor_f32_at(const struct ggml_tensor *t,
123 size_t i0,
124 size_t i1,
125 size_t i2,
126 size_t i3)
127{
128 const uint8_t *base = (const uint8_t *) t->data;
129 const size_t off =
130 i0 * (size_t) t->nb[0] +
131 i1 * (size_t) t->nb[1] +
132 i2 * (size_t) t->nb[2] +
133 i3 * (size_t) t->nb[3];
134 return *(const float *) (base + off);
135}

137static void ck_attention_oracle_dump_tensor(const char *name,
138 int layer_id,
139 const struct ggml_tensor *t)
140{
141 const char *dir = getenv("CK_PARITY_DIR");
142 if (!ck_attention_oracle_dump_enabled() ||
143 !dir || !dir[0] || !name || !name[0] || !t || !t->data ||
144 !ck_attention_oracle_should_dump_layer(layer_id)) {
145 return;
146 }
147 if (t->type != GGML_TYPE_F32) {
148 return;
149 }
150
151 const int64_t ne0 = t->ne[0] > 0 ? t->ne[0] : 1;
152 const int64_t ne1 = t->ne[1] > 0 ? t->ne[1] : 1;
153 const int64_t ne2 = t->ne[2] > 0 ? t->ne[2] : 1;
154 const int64_t ne3 = t->ne[3] > 0 ? t->ne[3] : 1;
155 const size_t elem_count =
156 (size_t) ne0 * (size_t) ne1 * (size_t) ne2 * (size_t) ne3;
157 if (elem_count == 0) {
158 return;
159 }
160 const size_t nbytes = elem_count * sizeof(float);
161 float *host = NULL;
162 const uint8_t *base = NULL;
163
164 if (t->buffer != NULL) {
165 ck_ggml_backend_tensor_get_fn ggml_backend_tensor_get_fn = ck_resolve_ggml_backend_tensor_get();
166 if (!ggml_backend_tensor_get_fn) {
167 return;
168 }
169 host = (float *) malloc(nbytes);
170 if (!host) {
171 return;
172 }
173 ggml_backend_tensor_get_fn(t, host, 0, nbytes);
174 base = (const uint8_t *) host;
175 } else if (t->data != NULL) {
176 base = (const uint8_t *) t->data;
177 }
178 if (!base) {
179 free(host);
180 return;
181 }
182
183 char path[4096];
184 snprintf(path, sizeof(path), "%s/%s", dir, "strict_internal.bin");
185 FILE *f = fopen(path, "ab");
186 if (!f) {
187 free(host);
188 return;
189 }
190
191 ck_attention_oracle_dump_header_t h;
192 memset(&h, 0, sizeof(h));
193 memcpy(h.magic, ck_attention_oracle_magic, sizeof(ck_attention_oracle_magic));
194 h.version = ck_attention_oracle_version;
195 h.layer_id = layer_id;
196 strncpy(h.op_name, name, sizeof(h.op_name) - 1);
197 h.dtype = 0u;
198 h.rank = 1u;
199 h.shape[0] = (int64_t) elem_count;
200 h.elem_count = (uint32_t) elem_count;
201 h.token_id = 0;
202
203 fwrite(&h, sizeof(h), 1, f);
204 for (int64_t i3 = 0; i3 < ne3; ++i3) {
205 for (int64_t i2 = 0; i2 < ne2; ++i2) {
206 for (int64_t i1 = 0; i1 < ne1; ++i1) {
207 for (int64_t i0 = 0; i0 < ne0; ++i0) {
208 const size_t off =
209 (size_t) i0 * (size_t) t->nb[0] +
210 (size_t) i1 * (size_t) t->nb[1] +
211 (size_t) i2 * (size_t) t->nb[2] +
212 (size_t) i3 * (size_t) t->nb[3];
213 const float v = *(const float *) (base + off);
214 fwrite(&v, sizeof(v), 1, f);
215 }
216 }
217 }
218 }
219 fclose(f);
220 free(host);
221}

223static void ck_attention_oracle_dump_meta(const char *name,
224 int layer_id,
225 const struct ggml_tensor *t)
226{
227 const char *dir = getenv("CK_PARITY_DIR");
228 if (!ck_attention_oracle_meta_dump_enabled() ||
229 !dir || !dir[0] || !name || !name[0] || !t ||
230 !ck_attention_oracle_should_dump_layer(layer_id)) {
231 return;
232 }
233
234 char path[4096];
235 snprintf(path, sizeof(path), "%s/%s", dir, "meta.jsonl");
236 FILE *f = fopen(path, "a");
237 if (!f) {
238 return;
239 }
240
241 fprintf(
242 f,
243 "{\"name\":\"%s\",\"layer_id\":%d,\"type\":%d,"
244 "\"ne\":[%lld,%lld,%lld,%lld],"
245 "\"nb\":[%zu,%zu,%zu,%zu]}\n",
246 name,
247 layer_id,
248 (int) t->type,
249 (long long) t->ne[0], (long long) t->ne[1], (long long) t->ne[2], (long long) t->ne[3],
250 (size_t) t->nb[0], (size_t) t->nb[1], (size_t) t->nb[2], (size_t) t->nb[3]);
251 fclose(f);
252}

540int ck_attention_head_full_ggml_graph_oracle_regular(const float *q_head,
541 const float *k_head,
542 const float *v_head,
543 float *out_head,
544 int num_tokens,
545 int head_dim,
546 int aligned_head_dim,
547 float scale)
548{
549 const char *disable_env = getenv("CK_STRICT_DISABLE_REGULAR_ATTN_ORACLE");
550 if (disable_env && disable_env[0] && strcmp(disable_env, "0") != 0) {
551 return 0;
552 }
553 ck_ggml_cpu_init_fn ggml_cpu_init_fn = ck_resolve_ggml_cpu_init();
554 ck_ggml_init_fn ggml_init_fn = ck_resolve_ggml_init();
555 ck_ggml_free_fn ggml_free_fn = ck_resolve_ggml_free();
556 ck_ggml_new_tensor_1d_fn ggml_new_tensor_1d_fn = ck_resolve_ggml_new_tensor_1d();
557 ck_ggml_new_tensor_2d_fn ggml_new_tensor_2d_fn = ck_resolve_ggml_new_tensor_2d();
558 ck_ggml_view_3d_fn ggml_view_3d_fn = ck_resolve_ggml_view_3d();
559 ck_ggml_permute_fn ggml_permute_fn = ck_resolve_ggml_permute();
560 ck_ggml_cont_fn ggml_cont_fn = ck_resolve_ggml_cont();
561 ck_ggml_cont_2d_fn ggml_cont_2d_fn = ck_resolve_ggml_cont_2d();
562 ck_ggml_mul_mat_graph_fn ggml_mul_mat_fn = ck_resolve_ggml_mul_mat_graph();
563 ck_ggml_soft_max_ext_fn ggml_soft_max_ext_fn = ck_resolve_ggml_soft_max_ext();
564 ck_ggml_new_graph_fn ggml_new_graph_fn = ck_resolve_ggml_new_graph();
