20static int tile_order_index_2d(int linear_idx,
21 int grid_h,
22 int grid_w,
23 int merge_size)
24{
25 if (linear_idx < 0 || grid_h <= 0 || grid_w <= 0 || merge_size <= 0) {
26 return 0;
27 }
28
29 const int tiles_w = (grid_w + merge_size - 1) / merge_size;
30 const int tile_area = merge_size * merge_size;
31 const int tile_idx = linear_idx / tile_area;
32 const int intra_idx = linear_idx % tile_area;
33 const int tile_y = tile_idx / tiles_w;
34 const int tile_x = tile_idx % tiles_w;
35 const int dy = intra_idx / merge_size;
36 const int dx = intra_idx % merge_size;
37 const int y = tile_y * merge_size + dy;
38 const int x = tile_x * merge_size + dx;
39
40 return y * grid_w + x;
41}

44/**
45 * im2patch: Transforms an image into a sequence of flattened patches.
46 *
47 * Image Layout: [C, H, W] (Row-major: W is fastest moving)
48 * Output Layout: [num_patches, C * P * P]
49 *
50 * num_patches = (H/P) * (W/P)
51 * P = patch_size
52 */
53void im2patch(const float *image,
54 float *patches,
55 int C, int H, int W, int P)
56{
57 int num_patches_h = H / P;
58 int num_patches_w = W / P;
59 int patch_dim = C * P * P;
60
61 // ph, pw: patch grid coordinates
62 for (int ph = 0; ph < num_patches_h; ++ph) {
63 for (int pw = 0; pw < num_patches_w; ++pw) {
64
65 int patch_idx = ph * num_patches_w + pw;
66 float *dst_patch = patches + (size_t)patch_idx * patch_dim;
67
68 // For each patch, grab pixels from all channels
69 for (int c = 0; c < C; ++c) {
70 for (int py = 0; py < P; ++py) {
71 int y = ph * P + py;
72 int x = pw * P;
73
74 // Input row start in the image
75 const float *src_row = image + (size_t)c * H * W + (size_t)y * W + x;
76
77 // Destination row in the flattened patch sequence
78 float *dst_row = dst_patch + (size_t)c * P * P + (size_t)py * P;
79
80 // Copy P pixels (one row of the patch)
81 memcpy(dst_row, src_row, P * sizeof(float));
82 }
83 }
84 }
85 }
86}

88/**
89 * patch2im: Accumulates gradients from patches back into the image. (Backward pass)
90 *
91 * d_patches: [num_patches, C * P * P]
92 * d_image: [C, H, W] (Accumulated)
93 */
94void patch2im(const float *d_patches,
95 float *d_image,
96 int C, int H, int W, int P)
97{
98 int num_patches_h = H / P;
99 int num_patches_w = W / P;
100 int patch_dim = C * P * P;
101
102 // Zero out the image first as we are accumulating gradients
103 memset(d_image, 0, (size_t)C * H * W * sizeof(float));
104
105 for (int ph = 0; ph < num_patches_h; ++ph) {
106 for (int pw = 0; pw < num_patches_w; ++pw) {
107
108 int patch_idx = ph * num_patches_w + pw;
109 const float *src_patch = d_patches + (size_t)patch_idx * patch_dim;
110
111 for (int c = 0; c < C; ++c) {
112 for (int py = 0; py < P; ++py) {
113 int y = ph * P + py;
114 int x = pw * P;
115
116 float *dst_row = d_image + (size_t)c * H * W + (size_t)y * W + x;
117 const float *src_row = src_patch + (size_t)c * P * P + (size_t)py * P;
118
119 // Add the patch gradient to the image gradient
120 for (int px = 0; px < P; ++px) {
121 dst_row[px] += src_row[px];
122 }
123 }
124 }
125 }
126 }
127}

129/**
130 * Add learned absolute position embeddings in-place.
131 *
132 * x layout: [num_tokens, embed_dim]
133 * position_embd: [num_positions, embed_dim]
134 *
135 * This first v8 vision path intentionally assumes native resized embeddings are
136 * already materialized in the weight tensor, so token i maps directly to
137 * position_embd[i]. This is the correct contract for fixed-size bring-up.
138 */
139void position_embeddings_add(float *x,
140 const float *position_embd,
141 int num_tokens,
142 int embed_dim,
143 int num_positions)
144{
145 if (x == NULL || position_embd == NULL || num_tokens <= 0 || embed_dim <= 0) {
146 return;
147 }
148 const int limit = num_tokens < num_positions ? num_tokens : num_positions;
149 for (int tok = 0; tok < limit; ++tok) {
150 float *dst = x + (size_t)tok * embed_dim;
151 const float *src = position_embd + (size_t)tok * embed_dim;
152 for (int d = 0; d < embed_dim; ++d) {
153 dst[d] += src[d];
154 }
155 }
156}

158void position_embeddings_add_tiled_2d(float *x,
159 const float *position_embd,
160 int grid_h,
161 int grid_w,
162 int embed_dim,
163 int merge_size,
164 int source_grid_size)
165{
166 if (x == NULL || position_embd == NULL || grid_h <= 0 || grid_w <= 0 || embed_dim <= 0 || merge_size <= 0) {
167 return;
168 }
169
170 if (source_grid_size <= 0) {
171 source_grid_size = grid_h == grid_w ? grid_h : (grid_h > grid_w ? grid_h : grid_w);
172 }
173
174 const int num_tokens = grid_h * grid_w;
175 const int source_tokens = source_grid_size * source_grid_size;
176 const int needs_resize = source_grid_size != grid_h || source_grid_size != grid_w;
177
178 const float sf_x = needs_resize ? (float) grid_w / (float) source_grid_size : 1.0f;
179 const float sf_y = needs_resize ? (float) grid_h / (float) source_grid_size : 1.0f;
180 const float pixel_offset = 0.5f;
181 const float support_x = needs_resize ? fmaxf(1.0f, 1.0f / sf_x) : 1.0f;
182 const float support_y = needs_resize ? fmaxf(1.0f, 1.0f / sf_y) : 1.0f;
183 const float invscale_x = needs_resize ? 1.0f / support_x : 1.0f;
184 const float invscale_y = needs_resize ? 1.0f / support_y : 1.0f;
185
186 for (int tok = 0; tok < num_tokens; ++tok) {
187 const int src_tok = tile_order_index_2d(tok, grid_h, grid_w, merge_size);
188 const int dst_y = src_tok / grid_w;
189 const int dst_x = src_tok % grid_w;
190 float *dst = x + (size_t) tok * (size_t) embed_dim;
191
192 if (!needs_resize) {
193 const int flat = dst_y * source_grid_size + dst_x;
194 if (flat < 0 || flat >= source_tokens) {
195 continue;
196 }
197 const float *src = position_embd + (size_t) flat * (size_t) embed_dim;
198 for (int d = 0; d < embed_dim; ++d) {
199 dst[d] += src[d];
200 }
201 continue;
202 }
203
204 const float x_src = ((float) dst_x + pixel_offset) / sf_x;
205 const float y_src = ((float) dst_y + pixel_offset) / sf_y;
206 int x_min = (int) (x_src - support_x + pixel_offset);
207 int x_max = (int) (x_src + support_x + pixel_offset);
208 int y_min = (int) (y_src - support_y + pixel_offset);
209 int y_max = (int) (y_src + support_y + pixel_offset);
210 if (x_min < 0) x_min = 0;
211 if (y_min < 0) y_min = 0;
212 if (x_max > source_grid_size) x_max = source_grid_size;
213 if (y_max > source_grid_size) y_max = source_grid_size;
214
215 for (int d = 0; d < embed_dim; ++d) {
216 float val = 0.0f;
217 float total_weight = 0.0f;
218 for (int sy = y_min; sy < y_max; ++sy) {
219 const float wy = fmaxf(1.0f - fabsf(((float) sy - y_src + pixel_offset) * invscale_y), 0.0f);
220 if (wy <= 0.0f) {
221 continue;
222 }
223 for (int sx = x_min; sx < x_max; ++sx) {
224 const float wx = fmaxf(1.0f - fabsf(((float) sx - x_src + pixel_offset) * invscale_x), 0.0f);
225 const float weight = wx * wy;
226 if (weight <= 0.0f) {
227 continue;
228 }
229 const float sample = position_embd[((size_t) sy * (size_t) source_grid_size + (size_t) sx) * (size_t) embed_dim + (size_t) d];
230 val += sample * weight;
231 total_weight += weight;
232 }
233 }
234 if (total_weight > 0.0f) {
235 dst[d] += val / total_weight;
236 }
237 }
238 }
239}

241/**
242 * Build merged 2D vision position IDs in the layout expected by vision M-RoPE.
243 *
244 * Output layout: [4, grid_h * grid_w] flattened as
245 * [y_stream | x_stream | y_stream_dup | x_stream_dup]
246 *
247 * Tokens are emitted in merged-tile traversal order so the position buffer
248 * matches the same 2x2 grouping used by Qwen-style vision encoders.
249 */
250void vision_position_ids_2d_merge(int32_t *positions,
251 int grid_h,
252 int grid_w,
253 int merge_size)
254{
255 if (!positions || grid_h <= 0 || grid_w <= 0 || merge_size <= 0) {
256 return;
257 }
258
259 const int num_tokens = grid_h * grid_w;
260 int ptr = 0;
261
262 for (int y = 0; y < grid_h; y += merge_size) {
263 for (int x = 0; x < grid_w; x += merge_size) {
264 for (int dy = 0; dy < merge_size; ++dy) {
265 for (int dx = 0; dx < merge_size; ++dx) {
266 const int yy = y + dy;
267 const int xx = x + dx;
268 if (yy >= grid_h || xx >= grid_w || ptr >= num_tokens) {
269 continue;
270 }
271 positions[ptr] = yy;
272 positions[num_tokens + ptr] = xx;
273 positions[2 * num_tokens + ptr] = yy;
274 positions[3 * num_tokens + ptr] = xx;
275 ++ptr;
276 }
277 }
278 }
279 }
280
281 for (; ptr < num_tokens; ++ptr) {
282 positions[ptr] = 0;
283 positions[num_tokens + ptr] = 0;
284 positions[2 * num_tokens + ptr] = 0;
285 positions[3 * num_tokens + ptr] = 0;
286 }
287}

289/**
290 * Merge 2x2 neighboring tokens into a single wider token.
291 *
292 * Input layout: [grid_h * grid_w, embed_dim]
293 * Output layout: [(grid_h/2) * (grid_w/2), embed_dim * 4]
294 *
295 * Pack order within each merged token:
296 * top-left, top-right, bottom-left, bottom-right
297 */
298void spatial_merge_2x2(const float *input,
299 float *output,
300 int grid_h,
301 int grid_w,
302 int embed_dim)
303{
304 if (input == NULL || output == NULL || grid_h <= 0 || grid_w <= 0 || embed_dim <= 0) {
305 return;
306 }
307
308 const int merged_h = grid_h / 2;
309 const int merged_w = grid_w / 2;
310 const size_t token_stride = (size_t)embed_dim;
311 const size_t merged_stride = (size_t)embed_dim * 4;
312
313 for (int mh = 0; mh < merged_h; ++mh) {
314 for (int mw = 0; mw < merged_w; ++mw) {
315 const int y0 = mh * 2;
316 const int x0 = mw * 2;
317 const int in_idx00 = y0 * grid_w + x0;
318 const int in_idx01 = in_idx00 + 1;
319 const int in_idx10 = in_idx00 + grid_w;
320 const int in_idx11 = in_idx10 + 1;
321 const int out_idx = mh * merged_w + mw;
322
323 const float *src00 = input + (size_t)in_idx00 * token_stride;
324 const float *src01 = input + (size_t)in_idx01 * token_stride;
325 const float *src10 = input + (size_t)in_idx10 * token_stride;
326 const float *src11 = input + (size_t)in_idx11 * token_stride;
327 float *dst = output + (size_t)out_idx * merged_stride;
328
329 memcpy(dst + 0 * embed_dim, src00, (size_t)embed_dim * sizeof(float));
330 memcpy(dst + 1 * embed_dim, src01, (size_t)embed_dim * sizeof(float));
331 memcpy(dst + 2 * embed_dim, src10, (size_t)embed_dim * sizeof(float));
332 memcpy(dst + 3 * embed_dim, src11, (size_t)embed_dim * sizeof(float));
333 }
334 }
335}

384void add_stream_reorder_2d(float *main_inout,
385 float *aux_scratch,
386 int grid_h,
387 int grid_w,
388 int embed_dim,
389 int merge_size)
390{
391 if (!main_inout || !aux_scratch || grid_h <= 0 || grid_w <= 0 || embed_dim <= 0 || merge_size <= 0) {
392 return;
393 }
394
395 const int num_tokens = grid_h * grid_w;
396 const size_t total_elems = (size_t) num_tokens * (size_t) embed_dim;
397
398 for (size_t i = 0; i < total_elems; ++i) {
399 main_inout[i] += aux_scratch[i];
400 }
401
402 for (int tok = 0; tok < num_tokens; ++tok) {
403 const int src_tok = tile_order_index_2d(tok, grid_h, grid_w, merge_size);
404 const float *src_main = main_inout + (size_t) src_tok * (size_t) embed_dim;
405 float *dst = aux_scratch + (size_t) tok * (size_t) embed_dim;
406 for (int d = 0; d < embed_dim; ++d) {
407 dst[d] = src_main[d];
408 }
409 }
410
411 memcpy(main_inout, aux_scratch, total_elems * sizeof(float));
412}
