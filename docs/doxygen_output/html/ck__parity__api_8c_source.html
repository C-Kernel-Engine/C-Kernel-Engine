209 void ck_test_gemm_q4_k(const void *weight_q4k,
210  const float *input_f32,
211  float *output,
212  int rows, int cols, int n_tokens)
213 {
214  /* Allocate Q8_K buffer for quantized activations */
215  int n_blocks_per_row = cols / CK_QK_K;
216  block_q8_K *q8_data = (block_q8_K *)malloc(n_tokens * n_blocks_per_row * sizeof(block_q8_K));
217  if (!q8_data) {
218  memset(output, 0, n_tokens * rows * sizeof(float));
219  return;
220  }
221 
222  /* Quantize all input tokens */
223  for (int t = 0; t < n_tokens; t++) {
224  quantize_row_q8_k(input_f32 + t * cols,
225  q8_data + t * n_blocks_per_row, cols);
226  }
227 
228  /* Use gemm_nt_q4_k_q8_k: C[M,N] = A[M,K] * B[N,K]^T
229  * Our layout: output[n_tokens, rows] = input[n_tokens, cols] * weight[rows, cols]^T
230  * So: M = n_tokens, N = rows, K = cols
231  */
232  gemm_nt_q4_k_q8_k(q8_data, weight_q4k, NULL, output, n_tokens, rows, cols);
233 
234  free(q8_data);
235 }

251 void ck_test_rope(float *q, float *k,
252  int n_tokens, int n_heads, int n_heads_kv, int head_dim,
253  int pos_offset, float theta)
254 {
255  /* Precompute cos/sin cache */
256  int half_dim = head_dim / 2;
257  int max_seq = pos_offset + n_tokens;
258 
259  float *cos_cache = (float *)malloc(max_seq * half_dim * sizeof(float));
260  float *sin_cache = (float *)malloc(max_seq * half_dim * sizeof(float));
261  if (!cos_cache || !sin_cache) {
262  free(cos_cache);
263  free(sin_cache);
264  return;
265  }
266 
267  rope_precompute_cache(cos_cache, sin_cache, max_seq, head_dim, theta);
268 
269  /* CK RoPE expects layout [num_heads, num_tokens, head_dim]
270  * Reshape from [n_tokens, n_heads * head_dim] to [n_heads, n_tokens, head_dim]
271  */
272  float *q_reorder = (float *)malloc(n_heads * n_tokens * head_dim * sizeof(float));
273  float *k_reorder = (float *)malloc(n_heads_kv * n_tokens * head_dim * sizeof(float));
274 
275  if (q_reorder && k_reorder) {
276  /* Reorder Q: [T, H*D] -> [H, T, D] */
277  for (int t = 0; t < n_tokens; t++) {
278  for (int h = 0; h < n_heads; h++) {
279  for (int d = 0; d < head_dim; d++) {
280  q_reorder[h * n_tokens * head_dim + t * head_dim + d] =
281  q[t * n_heads * head_dim + h * head_dim + d];
282  }
283  }
284  }
285 
286  /* Reorder K: [T, H_kv*D] -> [H_kv, T, D] */
287  for (int t = 0; t < n_tokens; t++) {
288  for (int h = 0; h < n_heads_kv; h++) {
289  for (int d = 0; d < head_dim; d++) {
290  k_reorder[h * n_tokens * head_dim + t * head_dim + d] =
291  k[t * n_heads_kv * head_dim + h * head_dim + d];
292  }
293  }
294  }
295 
296  /* Apply RoPE */
297  rope_forward_qk(q_reorder, k_reorder,
298  cos_cache, sin_cache,
299  n_heads, n_heads_kv, n_tokens,
300  head_dim, head_dim, pos_offset);
301 
302  /* Reorder back: [H, T, D] -> [T, H*D] */
303  for (int t = 0; t < n_tokens; t++) {
304  for (int h = 0; h < n_heads; h++) {
305  for (int d = 0; d < head_dim; d++) {
306  q[t * n_heads * head_dim + h * head_dim + d] =
307  q_reorder[h * n_tokens * head_dim + t * head_dim + d];
308  }
309  }
310  }
311 
312  for (int t = 0; t < n_tokens; t++) {
313  for (int h = 0; h < n_heads_kv; h++) {
314  for (int d = 0; d < head_dim; d++) {
315  k[t * n_heads_kv * head_dim + h * head_dim + d] =
316  k_reorder[h * n_tokens * head_dim + t * head_dim + d];
317  }
318  }
319  }
320  }
321 
322  free(q_reorder);
323  free(k_reorder);
324  free(cos_cache);
325  free(sin_cache);
326 }

328 void ck_test_rope_interleaved(float *q, float *k,
329  int n_tokens, int n_heads, int n_heads_kv, int head_dim,
330  int pos_offset, float theta)
331 {
332  /* Interleaved RoPE format (matches llama.cpp):
333  * (x0, x1) -> (x0*cos - x1*sin, x0*sin + x1*cos)
334  * Applied to consecutive pairs of elements
335  */
336 
337  /* Precompute inverse frequencies */
338  float *inv_freq = (float *)malloc((head_dim / 2) * sizeof(float));
339  if (!inv_freq) return;
340 
341  for (int i = 0; i < head_dim / 2; i++) {
342  inv_freq[i] = 1.0f / powf(theta, (float)(2 * i) / head_dim);
343  }
344 
345  /* Apply RoPE to Q */
346  for (int t = 0; t < n_tokens; t++) {
347  int pos = pos_offset + t;
348  for (int h = 0; h < n_heads; h++) {
349  float *qh = q + t * n_heads * head_dim + h * head_dim;
350 
351  for (int i = 0; i < head_dim / 2; i++) {
352  float freq = pos * inv_freq[i];
353  float cos_val = cosf(freq);
354  float sin_val = sinf(freq);
355 
356  /* Interleaved format */
357  float x0 = qh[i * 2];
358  float x1 = qh[i * 2 + 1];
359  qh[i * 2] = x0 * cos_val - x1 * sin_val;
360  qh[i * 2 + 1] = x0 * sin_val + x1 * cos_val;
361  }
362  }
363  }
364 
365  /* Apply RoPE to K */
366  for (int t = 0; t < n_tokens; t++) {
367  int pos = pos_offset + t;
368  for (int h = 0; h < n_heads_kv; h++) {
369  float *kh = k + t * n_heads_kv * head_dim + h * head_dim;
370 
371  for (int i = 0; i < head_dim / 2; i++) {
372  float freq = pos * inv_freq[i];
373  float cos_val = cosf(freq);
374  float sin_val = sinf(freq);
375 
376  float x0 = kh[i * 2];
377  float x1 = kh[i * 2 + 1];
378  kh[i * 2] = x0 * cos_val - x1 * sin_val;
379  kh[i * 2 + 1] = x0 * sin_val + x1 * cos_val;
380  }
381  }
382  }
383 
384  free(inv_freq);
385 }

394 void ck_test_softmax(const float *input, float *output, int n)
395 {
396  /* Find max for numerical stability */
397  float max_val = input[0];
398  for (int i = 1; i < n; i++) {
399  if (input[i] > max_val) max_val = input[i];
400  }
401 
402  /* Compute exp and sum */
403  float sum = 0.0f;
404  for (int i = 0; i < n; i++) {
405  output[i] = expf(input[i] - max_val);
406  sum += output[i];
407  }
408 
409  /* Normalize */
410  float inv_sum = 1.0f / sum;
411  for (int i = 0; i < n; i++) {
412  output[i] *= inv_sum;
413  }
414 }
