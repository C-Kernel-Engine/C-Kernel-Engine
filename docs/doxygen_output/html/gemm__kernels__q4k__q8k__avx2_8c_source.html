23  __m128i lo = _mm256_castsi256_si128(v);

57 static inline int32_t dot_q4_q8_32_avx2(const uint8_t *q4,
58  const int8_t *q8) {

60  const __m128i mask4 = _mm_set1_epi8(0x0F);
61  const __m128i q4_lo = _mm_and_si128(q4_packed, mask4);
62  const __m128i q4_hi = _mm_and_si128(_mm_srli_epi16(q4_packed, 4), mask4);
63 
64  const __m256i q4_lo16 = _mm256_cvtepu8_epi16(q4_lo);
65  const __m256i q4_hi16 = _mm256_cvtepu8_epi16(q4_hi);
