31static size_t align_up_bytes(size_t n, size_t align)
32{
33 if (align == 0) return n;
34 return (n + align - 1) & ~(align - 1);

37static size_t align_down_bytes(size_t n, size_t align)
38{
39 if (align == 0) return n;
40 return n & ~(align - 1);

43static int record_allocation(void *ptr, size_t len, int was_mmap)
44{
45 ck_huge_alloc_entry_t *entry = malloc(sizeof(*entry));
46 if (!entry) {
47 return 0;
48 }
49 entry->ptr = ptr;
50 entry->len = len;
51 entry->was_mmap = was_mmap;
52 pthread_mutex_lock(&g_alloc_mutex);
53 entry->next = g_alloc_list;
54 g_alloc_list = entry;
55 pthread_mutex_unlock(&g_alloc_mutex);
56 return 1;
57}

59static ck_huge_alloc_entry_t *detach_allocation(void *ptr)
60{
61 pthread_mutex_lock(&g_alloc_mutex);
62 ck_huge_alloc_entry_t **node = &g_alloc_list;
63 while (*node) {
64 if ((*node)->ptr == ptr) {
65 ck_huge_alloc_entry_t *entry = *node;
66 *node = entry->next;
67 pthread_mutex_unlock(&g_alloc_mutex);
68 return entry;
69 }
70 node = &(*node)->next;
71 }
72 pthread_mutex_unlock(&g_alloc_mutex);
73 return NULL;
74}

76void *ck_huge_alloc(size_t bytes)
77{
78 size_t len = align_up_bytes(bytes, HUGE_PAGE_SIZE);
79
80 /* First, try explicit huge pages via mmap + MAP_HUGETLB. */
81 void *p = mmap(NULL, len,
82 PROT_READ | PROT_WRITE,
83 MAP_PRIVATE | MAP_ANONYMOUS | MAP_HUGETLB,
84 -1, 0);
85 if (p != MAP_FAILED) {
86 if (!record_allocation(p, len, 1)) {
87 munmap(p, len);
88 return NULL;
89 }
90 return p;
91 }
92
93 /* Fallback: aligned_alloc with transparent hugepage hint. */
94 void *q = aligned_alloc(HUGE_PAGE_SIZE, len);
95 if (!q) {
96 fprintf(stderr, "ck_huge_alloc: aligned_alloc failed for %zu bytes: %s\n",
97 len, strerror(errno));
98 return NULL;
99 }
100
101 /* Best-effort hint; ignore errors. */
102 (void)madvise(q, len, MADV_HUGEPAGE);
103 if (!record_allocation(q, len, 0)) {
104 free(q);
105 return NULL;
106 }
107 return q;
108}

110void ck_huge_free(void *ptr, size_t bytes)
111{
112 if (!ptr || bytes == 0) {
113 return;
114 }
115
116 ck_huge_alloc_entry_t *entry = detach_allocation(ptr);
117 if (!entry) {
118 /* Fall back to malloc/free if the allocation wasn't tracked. */
119 free(ptr);
120 return;
121 }
122
123 if (entry->was_mmap) {
124 munmap(ptr, entry->len);
125 } else {
126 free(ptr);
127 }
128
129 free(entry);
130}

159static int ck_bump_alloc_try_mixed(ck_bump_alloc_t *alloc, const char *weights_path)
160{
161 long page_size_raw = sysconf(_SC_PAGESIZE);
162 size_t page_size = page_size_raw > 0 ? (size_t)page_size_raw : 4096U;
163 size_t mapped_len = 0;
164 size_t prefix_len = 0;
165 size_t weights_len = 0;
166 size_t weights_map_len = 0;
167 size_t runtime_map_start = 0;
168 int fd = -1;
169 struct stat st;
170 uint8_t *base = NULL;
171
172 if (!alloc || !weights_path || !*weights_path) {
173 return -1;
174 }
175 if (alloc->weights_base > alloc->activations_base || alloc->activations_base > alloc->total_size) {
176 fprintf(stderr, "ck_bump_alloc_init: invalid bump layout for mixed fallback\n");
177 return -1;
178 }
179 if (alloc->weights_base != 0) {
180 fprintf(stderr,
181 "ck_bump_alloc_init: mixed fallback currently requires weights_base == 0 (got %zu)\n",
182 alloc->weights_base);
183 return -1;
184 }
185 if (alloc->weights_base != align_down_bytes(alloc->weights_base, page_size)) {
186 fprintf(stderr,
187 "ck_bump_alloc_init: mixed fallback requires page-aligned weights_base (got %zu, page %zu)\n",
188 alloc->weights_base, page_size);
189 return -1;
190 }
191
192 mapped_len = align_up_bytes(alloc->total_size, page_size);
193 prefix_len = alloc->weights_base;
194 weights_len = alloc->activations_base - alloc->weights_base;
195 weights_map_len = align_up_bytes(weights_len, page_size);
196 runtime_map_start = align_up_bytes(alloc->activations_base, page_size);
197
198 fd = open(weights_path, O_RDONLY | O_CLOEXEC);
199 if (fd < 0) {
200 fprintf(stderr, "ck_bump_alloc_init: failed to open %s: %s\n", weights_path, strerror(errno));
201 return -1;
202 }
203 if (fstat(fd, &st) != 0) {
204 fprintf(stderr, "ck_bump_alloc_init: fstat failed for %s: %s\n", weights_path, strerror(errno));
205 close(fd);
206 return -1;
207 }
208 if (!S_ISREG(st.st_mode)) {
209 fprintf(stderr, "ck_bump_alloc_init: %s is not a regular file\n", weights_path);
210 close(fd);
211 return -1;
212 }
213 alloc->weights_file_size = (size_t)st.st_size;
214 if (weights_len > 0 && weights_map_len > align_up_bytes(alloc->weights_file_size, page_size)) {
215 fprintf(stderr,
216 "ck_bump_alloc_init: weights file too small for mixed mapping (%zu < %zu bytes)\n",
217 alloc->weights_file_size, weights_map_len);
218 close(fd);
219 return -1;
220 }
221
222 base = mmap(NULL, mapped_len, PROT_NONE, MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
223 if (base == MAP_FAILED) {
224 fprintf(stderr, "ck_bump_alloc_init: reserve mmap failed for %zu bytes: %s\n",
225 mapped_len, strerror(errno));
226 close(fd);
227 return -1;
228 }
229
230 if (prefix_len > 0) {
231 void *prefix = mmap(base, prefix_len, PROT_READ | PROT_WRITE,
232 MAP_PRIVATE | MAP_ANONYMOUS | MAP_FIXED, -1, 0);
233 if (prefix == MAP_FAILED) {
234 fprintf(stderr, "ck_bump_alloc_init: prefix mmap failed: %s\n", strerror(errno));
235 munmap(base, mapped_len);
236 close(fd);
237 return -1;
238 }
239 }
240
241 if (weights_map_len > 0) {
242 void *mapped = mmap(base + alloc->weights_base, weights_map_len,
243 PROT_READ | PROT_WRITE, MAP_PRIVATE | MAP_FIXED, fd, 0);
244 if (mapped == MAP_FAILED) {
245 fprintf(stderr, "ck_bump_alloc_init: weights mmap failed: %s\n", strerror(errno));
246 munmap(base, mapped_len);
247 close(fd);
248 return -1;
249 }
250 }
251
252 if (runtime_map_start < mapped_len) {
253 void *runtime = mmap(base + runtime_map_start, mapped_len - runtime_map_start,
254 PROT_READ | PROT_WRITE,
255 MAP_PRIVATE | MAP_ANONYMOUS | MAP_FIXED, -1, 0);
256 if (runtime == MAP_FAILED) {
257 fprintf(stderr, "ck_bump_alloc_init: runtime mmap failed: %s\n", strerror(errno));
258 munmap(base, mapped_len);
259 close(fd);
260 return -1;
261 }
262 }
263
264 close(fd);
265 alloc->base = base;
266 alloc->mapped_len = mapped_len;
267 alloc->mode = CK_BUMP_MODE_MIXED_FILE_BACKED;
268 fprintf(stderr,
269 "ck_bump_alloc_init: fallback to mixed-backed bump (%zu MiB total, weights file-backed=%zu MiB)\n",
270 alloc->total_size / (1024U * 1024U), weights_len / (1024U * 1024U));
271 return 0;
272}

275int ck_bump_alloc_init(ck_bump_alloc_t *alloc,
276 const char *weights_path,
277 size_t total_size,
278 size_t weights_base,
279 size_t activations_base)
280{
281 int force_mixed = 0;
282 int disable_mixed = 0;
283
284 if (!alloc || total_size == 0) {
285 fprintf(stderr, "ck_bump_alloc_init: invalid arguments\n");
286 return -1;
287 }
288
289 ck_bump_alloc_reset(alloc);
290 alloc->total_size = total_size;
291 alloc->weights_base = weights_base;
292 alloc->activations_base = activations_base;
293 force_mixed = env_flag_enabled("CK_BUMP_FORCE_MIXED");
294 disable_mixed = env_flag_enabled("CK_BUMP_DISABLE_MIXED");
295
296 if (weights_base > activations_base || activations_base > total_size) {
297 fprintf(stderr,
298 "ck_bump_alloc_init: invalid bump layout weights=%zu activations=%zu total=%zu\n",
299 weights_base, activations_base, total_size);
300 ck_bump_alloc_reset(alloc);
301 return -1;
302 }
303
304 if (!force_mixed) {
305 void *anon = ck_huge_alloc(total_size);
306 if (anon) {
307 alloc->base = (uint8_t *)anon;
308 alloc->mapped_len = align_up_bytes(total_size, HUGE_PAGE_SIZE);
309 alloc->mode = CK_BUMP_MODE_ANON;
310 return 0;
311 }
312 }
313
314#ifdef __linux__
315 if (!disable_mixed) {
316 if (ck_bump_alloc_try_mixed(alloc, weights_path) == 0) {
317 return 0;
318 }
319 }
320#else
321 (void)weights_path;
322#endif
323
324 fprintf(stderr,
325 "ck_bump_alloc_init: failed to allocate bump arena (%zu bytes, %.2f MiB)\n",
326 total_size, (double)total_size / (1024.0 * 1024.0));
327 ck_bump_alloc_reset(alloc);
328 return -1;
329}

331void ck_bump_alloc_free(ck_bump_alloc_t *alloc)
332{
333 if (!alloc || !alloc->base) {
334 ck_bump_alloc_reset(alloc);
335 return;
336 }
337
338 if (alloc->mode == CK_BUMP_MODE_MIXED_FILE_BACKED) {
339 if (alloc->mapped_len > 0) {
340 munmap(alloc->base, alloc->mapped_len);
341 }
342 } else if (alloc->mode == CK_BUMP_MODE_ANON) {
343 ck_huge_free(alloc->base, alloc->total_size);
344 }
345
346 ck_bump_alloc_reset(alloc);
347}
