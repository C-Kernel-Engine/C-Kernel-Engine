57 /**
58  * Simple JSON string extraction - finds a key and returns the value string.
59  * Returns pointer to malloc'd string (caller must free) or NULL on error.
60  */
61 static char *extract_json_string(const char *json, const char *key) {
62  if (!json || !key) return NULL;
63 
64  char pattern[128];
65  snprintf(pattern, sizeof(pattern), "\"%s\"", key);
66  const char *p = strstr(json, pattern);
67  if (!p) return NULL;
68 
69  /* Find the colon and opening quote */
70  p = strchr(p, ':');
71  if (!p) return NULL;
72  p = strchr(p + 1, '"');
73  if (!p) return NULL;
74  p++; /* Skip opening quote */
75 
76  /* Find closing quote */
77  const char *end = strchr(p, '"');
78  if (!end) return NULL;
79 
80  size_t len = (size_t)(end - p);
81  char *result = (char *)malloc(len + 1);
82  if (!result) return NULL;
83 
84  memcpy(result, p, len);
85  result[len] = '\0';
86  return result;
87 }

113 /**
114  * Extract an array of integers from JSON.
115  *
116  * @param json JSON string
117  * @param key Key to find
118  * @param out Output array
119  * @param max_count Maximum elements to extract
120  * @return Number of elements extracted
121  */
122 static int extract_json_int_array(const char *json, const char *key, int32_t *out, int max_count) {
123  if (!json || !key || !out) return 0;
124 
125  char pattern[128];
126  snprintf(pattern, sizeof(pattern), "\"%s\"", key);
127  const char *p = strstr(json, pattern);
128  if (!p) return 0;
129 
130  /* Find the opening bracket */
131  p = strchr(p, '[');
132  if (!p) return 0;
133  p++; /* Skip '[' */
134 
135  int count = 0;
136  while (count < max_count && *p) {
137  /* Skip whitespace */
138  while (*p == ' ' || *p == '\t' || *p == '\n') p++;
139 
140  if (*p == ']' || *p == '\0') break;
141 
142  /* Parse integer */
143  long val = strtol(p, NULL, 10);
144  out[count++] = (int32_t)val;
145 
146  /* Skip to next number */
147  while (*p && *p != ',' && *p != ']') p++;
148  if (*p == ',') p++;
149  }
150 
151  return count;
152 }

162 static bool init_stop_token_table(const int32_t *stop_ids, int count, int vocab_size) {
163  /* Free any existing table */
164  if (g_stop_token_table) {
165  free(g_stop_token_table);
166  g_stop_token_table = NULL;
167  }
168 
169  if (count <= 0 || vocab_size <= 0) {
170  return false;
171  }
172 
173  /* Allocate table (zero-initialized) */
174  g_stop_token_table = (bool *)calloc((size_t)vocab_size, sizeof(bool));
175  if (!g_stop_token_table) {
176  fprintf(stderr, "[Error] Failed to allocate stop token table (%d bytes)\n", vocab_size);
177  return false;
178  }
179 
180  g_vocab_size = vocab_size;
181 
182  /* Mark stop tokens */
183  for (int i = 0; i < count; i++) {
184  int id = stop_ids[i];
185  if (id >= 0 && id < vocab_size) {
186  g_stop_token_table[id] = true;
187  }
188  }
189 
190  fprintf(stderr, "[StopTable] Initialized with %d stop tokens, vocab_size=%d\n", count, vocab_size);
191  return true;
192 }

209 static bool load_chat_template(const char *weights_path) {
210  char path[4096];
211  const char *json_content = NULL;
212  char *json_copy = NULL;
213  bool success = false;
214 
215  /* Try chat_template.json first */
216  strncpy(path, weights_path, sizeof(path) - 20);
217  char *slash = strrchr(path, '/');
218  if (slash) {
219  strcpy(slash + 1, "chat_template.json");
220  } else {
221  strcpy(path, "chat_template.json");
222  }
223 
224  FILE *f = fopen(path, "r");
225  if (f) {
226  fseek(f, 0, SEEK_END);
227  long size = ftell(f);
228  fseek(f, 0, SEEK_SET);
229  if (size > 0 && size < (long)sizeof(path)) {
230  json_content = path; /* Reuse buffer */
231  size_t n = fread(path, 1, (size_t)size, f);
232  path[n] = '\0';
233  json_content = path;
234  } else if (size > 0 && size < 32768) {
235  json_copy = (char *)malloc((size_t)size + 1);
236  if (json_copy) {
237  size_t n = fread(json_copy, 1, (size_t)size, f);
238  json_copy[n] = '\0';
239  json_content = json_copy;
240  }
241  }
242  fclose(f);
243  }
244 
245  /* Try weights_manifest.json if chat_template.json not found */
246  if (!json_content) {
247  strncpy(path, weights_path, sizeof(path) - 25);
248  slash = strrchr(path, '/');
249  if (slash) {
250  strcpy(slash + 1, "weights_manifest.json");
251  } else {
252  strcpy(path, "weights_manifest.json");
253  }
254  f = fopen(path, "r");
255  if (f) {
256  fseek(f, 0, SEEK_END);
257  long size = ftell(f);
258  fseek(f, 0, SEEK_SET);
259  if (size > 0 && size < 32768) {
260  json_copy = (char *)malloc((size_t)size + 1);
261  if (json_copy) {
262  size_t n = fread(json_copy, 1, (size_t)size, f);
263  json_copy[n] = '\0';
264  json_content = json_copy;
265  }
266  }
267  fclose(f);
268  }
269  }
270 
271  if (!json_content) {
272  return false;
273  }
274 
275  /* Extract template fields */
276  char *template_str = extract_json_string(json_content, "template");
277  if (template_str) {
278  size_t len = strlen(template_str);
279  if (len < CK_TEMPLATE_MAX_SIZE) {
280  memcpy(g_chat_template, template_str, len + 1);
281  success = true;
282  }
283  free(template_str);
284  }
285 
286  char *system_str = extract_json_string(json_content, "default_system");
287  if (system_str) {
288  size_t len = strlen(system_str);
289  if (len < CK_SYSTEM_MAX_SIZE) {
290  memcpy(g_default_system, system_str, len + 1);
291  }
292  free(system_str);
293  }
294 
295  g_eos_token_id = extract_json_int(json_content, "eos_token_id");
296 
297  /* Extract stop_token_ids and vocab_size for lookup table */
298  int32_t stop_ids[16];
299  int stop_count = extract_json_int_array(json_content, "stop_token_ids", stop_ids, 16);
300  int vocab_size = extract_json_int(json_content, "vocab_size");
301  if (vocab_size <= 0) {
302  vocab_size = 200000; /* Default fallback */
303  }
304 
305  /* Initialize stop token lookup table */
306  if (stop_count > 0) {
307  init_stop_token_table(stop_ids, stop_count, vocab_size);
308  }
309 
310  free(json_copy);
311 
312  g_has_template = (g_chat_template[0] != '\0');
313  return g_has_template;
314 }

316 /**
317  * Apply chat template to user prompt.
318  * Returns allocated string (caller must free) or NULL on error.
319  */
320 static char *apply_chat_template(const char *user_prompt, const char *system_prompt) {
321  if (!user_prompt || !g_has_template) return NULL;
322 
323  const char *system = system_prompt ? system_prompt : g_default_system;
324  if (!system) system = "";
325 
326  const char *tpl = g_chat_template;
327  size_t tpl_len = strlen(tpl);
328  size_t prompt_len = strlen(user_prompt);
329  size_t system_len = strlen(system);
330 
331  /* Count placeholders to calculate buffer size */
332  int system_count = 0;
333  int prompt_count = 0;
334  for (const char *p = tpl; *p; p++) {
335  if (strncmp(p, "{system}", 8) == 0) { system_count++; p += 7; }
336  if (strncmp(p, "{prompt}", 8) == 0) { prompt_count++; p += 7; }
337  }
338 
339  size_t total = tpl_len
340  - (system_count * 8) /* Remove {system} */
341  - (prompt_count * 8) /* Remove {prompt} */
342  + system_len
343  + prompt_len
344  + 1; /* NULL terminator */
345 
346  char *result = (char *)malloc(total);
347  if (!result) return NULL;
348 
349  /* Build result string */
350  char *out = result;
351  const char *p = tpl;
352 
353  while (*p) {
354  if (strncmp(p, "{system}", 8) == 0) {
355  memcpy(out, system, system_len);
356  out += system_len;
357  p += 8;
358  } else if (strncmp(p, "{prompt}", 8) == 0) {
359  memcpy(out, user_prompt, prompt_len);
360  out += prompt_len;
361  p += 8;
362  } else {
363  *out++ = *p++;
364  }
365  }
366  *out = '\0';
367 
368  return result;
369 }

491 static void output_token(char *buf, size_t *len, const char *token) {
492  if (!token || !*token) return;
493  const unsigned char *b = (const unsigned char *)token;
494  /* "Ġ" prefix (0xC4 0xA0) */
495  if (b[0] == 0xC4 && b[1] == 0xA0) {
496  output_append(buf, len, " ");
497  output_append(buf, len, token + 2);
498  return;
499  }
500  /* "▁" prefix (0xE2 0x96 0x81) */
501  if (b[0] == 0xE2 && b[1] == 0x96 && b[2] == 0x81) {
502  output_append(buf, len, " ");
503  output_append(buf, len, token + 3);
504  return;
505  }
506  output_append(buf, len, token);
507 }

551 static int run_prompt(ModelAPI *api, CKTrueBPE *tokenizer, const CLIOptions *opt, const char *input) {
552  if (!api || !tokenizer || !opt || !input) return -1;
553  if (g_exit_requested) return -1;
554 
555  int ctx = opt->context_override;
556  if (ctx <= 0 && api->get_context) {
557  ctx = api->get_context();
558  }
559  if (ctx <= 0) ctx = 4096;
560 
561  /* max_tokens: if -1 (default), use a very large number and let EOS stop us
562  if > 0, use that limit */
563  int max_decode_tokens = (opt->max_tokens > 0) ? opt->max_tokens : 8192; /* generous limit */
564 
565  /* Apply chat template if available */
566  char *formatted_input = NULL;
567  const char *input_to_tokenize = input;
568 
569  if (g_has_template) {
570  formatted_input = apply_chat_template(input, NULL);
571  if (formatted_input) {
572  input_to_tokenize = formatted_input;
573  if (!opt->timing) {
574  /* Optional: show that template was applied */
575  }
576  }
577  }
578 
579  int32_t *ids = (int32_t *)malloc((size_t)ctx * sizeof(int32_t));
580  if (!ids) {
581  free(formatted_input);
582  fprintf(stderr, "Error: failed to allocate token buffer\n");
583  return -1;
584  }
585 
586  int n = ck_true_bpe_encode(tokenizer, input_to_tokenize, -1, ids, ctx);
587  free(formatted_input); /* Free formatted input after tokenization */
588 
589  if (n <= 0) {
590  fprintf(stderr, "[Tokenizer] failed to encode prompt\n");
591  free(ids);
592  return -1;
593  }
594  if (n > ctx) n = ctx;
595 
596  /* Reset timing globals */
597  g_prefill_time_ms = 0.0;
598  g_decode_time_ms = 0.0;
599  g_decode_count = 0;
600 
601  if (api->kv_reset) {
602  api->kv_reset();
603  }
604 
605  if (api->embed(ids, n) != 0) {
606  fprintf(stderr, "[Model] embed failed\n");
607  free(ids);
608  return -1;
609  }
610 
611  /* Time prefill (forward) */
612  struct timespec t0, t1;
613  clock_gettime(CLOCK_MONOTONIC, &t0);
614  if (api->forward(NULL) != 0) {
615  fprintf(stderr, "[Model] forward failed\n");
616  free(ids);
617  return -1;
618  }
619  clock_gettime(CLOCK_MONOTONIC, &t1);
620  g_prefill_time_ms = (t1.tv_sec - t0.tv_sec) * 1000.0 +
621  (t1.tv_nsec - t0.tv_nsec) / 1000000.0;
622 
623  int next_token = api->sample();
624  char out_buf[CK_CLI_OUTPUT_BUF_SIZE];
625  size_t out_len = 0;
626 
627  if (!opt->timing) {
628  printf("Assistant: ");
629  fflush(stdout);
630  }

632  for (int generated = 0; generated < max_decode_tokens && !g_exit_requested; generated++) {
633  if (next_token < 0) break;
634 
635  /* CHECK STOP TOKEN FIRST using O(1) lookup table - don't decode or print stop tokens */
636  if (is_stop_token_id(next_token)) {
637  break;
638  }
639 
640  /* Only decode and print if NOT a stop token */
641  const char *word = ck_true_bpe_id_to_token(tokenizer, next_token);
642  output_token(out_buf, &out_len, word);
643 
644  /* Flush immediately if streaming enabled */
645  if (opt->stream) {
646  output_flush(out_buf, &out_len);
647  fflush(stdout);
648  } else if (out_len > (CK_CLI_OUTPUT_BUF_SIZE / 2)) {
649  output_flush(out_buf, &out_len);
650  fflush(stdout);
651  }
652 
653  if (generated + 1 >= max_decode_tokens) break;
654 
655  /* Time decode */
656  clock_gettime(CLOCK_MONOTONIC, &t0);
657  if (api->decode(next_token, NULL) != 0) {
658  fprintf(stderr, "\n[Model] decode failed\n");
659  break;
660  }
661  clock_gettime(CLOCK_MONOTONIC, &t1);
662  g_decode_time_ms += (t1.tv_sec - t0.tv_sec) * 1000.0 +
663  (t1.tv_nsec - t0.tv_nsec) / 1000000.0;
664  g_decode_count++;
665 
666  next_token = api->sample();
667  }
668 
669  output_flush(out_buf, &out_len);
670 
671  /* Print timing if requested */
672  if (opt->timing) {
673  printf("\n");
674  double total_ms = g_prefill_time_ms + g_decode_time_ms;
675  double prefill_rate = n / (g_prefill_time_ms / 1000.0);
676  double decode_rate = g_decode_count / (g_decode_time_ms / 1000.0);
677  double avg_decode = g_decode_count > 0 ? g_decode_time_ms / g_decode_count : 0.0;
678 
679  printf("Timing:\n");
680  printf(" Prefill: %3d tokens in %7.2f ms (%6.2f tok/s)\n",
681  n, g_prefill_time_ms, prefill_rate);
682  printf(" Decode: %3d tokens in %7.2f ms (%6.2f tok/s) avg %.2f ms/tok\n",
683  g_decode_count, g_decode_time_ms, decode_rate, avg_decode);
684  printf(" Total: %3d tokens in %7.2f ms\n",
685  n + g_decode_count, total_ms);
686  } else {
687  printf("\n");
688  }
689  fflush(stdout);
690 
691  free(ids);
692  return 0;
693 }
