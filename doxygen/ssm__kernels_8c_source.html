1/**
2 * @file ssm_kernels.c
3 * @brief FP32 SSM causal depthwise convolution kernels for qwen3next/Qwen3.5.
4 *
5 * CK-ENGINE KERNEL RULES:
6 * =======================
7 * 1. NO malloc/free - memory via bump allocator, pointers passed in
8 * 2. NO OpenMP - parallelization at orchestrator/codegen layer
9 * 3. API must define: inputs, outputs, workspace, and memory layouts
10 * 4. Pure computation - deterministic, no side effects
11 *
12 * After changes: make test-ssm-conv && make test-kernels
13 *
14 * This file implements the GGML_OP_SSM_CONV semantics used by qwen3next before
15 * the recurrent DeltaNet update:
16 * out[seq, token, ch] = dot(conv_x[seq, ch, token:token+kernel], kernel[ch, :])
17 *
18 * Memory layouts:
19 * conv_x : [num_seqs, num_channels, kernel_size - 1 + num_tokens]
20 * kernel : [num_channels, kernel_size]
21 * out : [num_seqs, num_tokens, num_channels]
22 * d_out : same as out
23 * d_conv_x : same as conv_x
24 * d_kernel : same as kernel
25 */

32void ssm_conv1d_forward_ref(const float *conv_x,
33 const float *kernel,
34 float *out,
35 int kernel_size,
36 int num_channels,
37 int num_tokens,
38 int num_seqs)
39{
40 if (!conv_x || !kernel || !out) {
41 return;
42 }
43 if (kernel_size <= 0 || num_channels <= 0 || num_tokens < 0 || num_seqs <= 0) {
44 return;
45 }
46
47 const size_t seq_width = (size_t)kernel_size - 1u + (size_t)num_tokens;
48 const size_t conv_seq_stride = (size_t)num_channels * seq_width;
49 const size_t out_seq_stride = (size_t)num_tokens * (size_t)num_channels;
50
51 for (int seq = 0; seq < num_seqs; ++seq) {
52 const float *conv_seq = conv_x + (size_t)seq * conv_seq_stride;
53 float *out_seq = out + (size_t)seq * out_seq_stride;
54
55 for (int tok = 0; tok < num_tokens; ++tok) {
56 float *out_tok = out_seq + (size_t)tok * (size_t)num_channels;
57
58 for (int ch = 0; ch < num_channels; ++ch) {
59 const float *conv_row = conv_seq + (size_t)ch * seq_width + (size_t)tok;
60 const float *kernel_row = kernel + (size_t)ch * (size_t)kernel_size;
61 float sumf = 0.0f;
62 for (int k = 0; k < kernel_size; ++k) {
63 sumf += conv_row[k] * kernel_row[k];
64 }
65 out_tok[ch] = sumf;
66 }
67 }
68 }
69}

71void ssm_conv1d_backward_ref(const float *d_out,
72 const float *conv_x,
73 const float *kernel,
74 float *d_conv_x,
75 float *d_kernel,
76 int kernel_size,
77 int num_channels,
78 int num_tokens,
79 int num_seqs)
80{
81 if (!d_out || !conv_x || !kernel || !d_conv_x || !d_kernel) {
82 return;
83 }
84 if (kernel_size <= 0 || num_channels <= 0 || num_tokens < 0 || num_seqs <= 0) {
85 return;
86 }
87
88 const size_t seq_width = (size_t)kernel_size - 1u + (size_t)num_tokens;
89 const size_t conv_total = (size_t)num_seqs * (size_t)num_channels * seq_width;
90 const size_t kernel_total = (size_t)num_channels * (size_t)kernel_size;
91 const size_t conv_seq_stride = (size_t)num_channels * seq_width;
92 const size_t out_seq_stride = (size_t)num_tokens * (size_t)num_channels;
93
94 memset(d_conv_x, 0, conv_total * sizeof(float));
95 memset(d_kernel, 0, kernel_total * sizeof(float));
96
97 for (int seq = 0; seq < num_seqs; ++seq) {
98 const float *d_out_seq = d_out + (size_t)seq * out_seq_stride;
99 const float *conv_seq = conv_x + (size_t)seq * conv_seq_stride;
100 float *d_conv_seq = d_conv_x + (size_t)seq * conv_seq_stride;
101
102 for (int tok = 0; tok < num_tokens; ++tok) {
103 const float *d_out_tok = d_out_seq + (size_t)tok * (size_t)num_channels;
104
105 for (int ch = 0; ch < num_channels; ++ch) {
106 const float grad = d_out_tok[ch];
107 const float *conv_row = conv_seq + (size_t)ch * seq_width + (size_t)tok;
108 float *d_conv_row = d_conv_seq + (size_t)ch * seq_width + (size_t)tok;
109 const float *kernel_row = kernel + (size_t)ch * (size_t)kernel_size;
110 float *d_kernel_row = d_kernel + (size_t)ch * (size_t)kernel_size;
111
112 for (int k = 0; k < kernel_size; ++k) {
113 d_kernel_row[k] += grad * conv_row[k];
114 d_conv_row[k] += grad * kernel_row[k];
115 }
116 }
117 }
118 }
119}
