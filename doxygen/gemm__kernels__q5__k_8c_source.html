1/**
2 * @file gemm_kernels_q5_k.c
3 * @brief GEMM/GEMV kernels with Q5_K quantized weights
4 *
5 * CK-ENGINE KERNEL RULES:
6 * =======================
7 * 1. NO malloc/free - memory via bump allocator, pointers passed in
8 * 2. NO OpenMP - parallelization at orchestrator/codegen layer
9 * 3. API must define: inputs, outputs, workspace, and memory layouts
10 * 4. Pure computation - deterministic, no side effects
11 *
12 * After changes: make test && make llamacpp-parity-full
13 *
14 * Implements matrix multiplication where:
15 * - Activations (input): FP32 (quantized internally to Q8_K for dot path)
16 * - Weights: Q5_K (5-bit super-block quant)
17 * - Output: FP32
18 *
19 * Q5_K Format (256 weights per super-block):
20 * - d: FP16 super-block scale
21 * - dmin: FP16 super-block minimum
22 * - scales[12]: 8 sub-block scales + 8 sub-block mins (6 bits each, packed)
23 * - qh[32]: high bits for 256 weights (1 bit each)
24 * - qs[128]: low 4 bits for 256 weights (4 bits each)
25 *
26 * Total: 2 + 2 + 12 + 32 + 128 = 176 bytes per 256 weights = 5.5 bits/weight
27 *
28 * Dequantization formula (matches llama.cpp):
29 * w = d * scale * q - dmin * mins
30 * where q = qs_val | (qh_bit << 4) = 5-bit value [0, 31]
31 */
32

48 * Keep a local ggml-compatible layout to decouple from shared headers. */
49typedef struct {
50 ck_half d;
51 ck_half dmin;
52 uint8_t scales[K_SCALE_SIZE];
53 uint8_t qh[QK_K / 8];
54 uint8_t qs[QK_K / 2];
55} block_q5_K;

57/* Unpack 8 per-subblock scales and mins from packed Q5_K scale bytes.
58 * This mirrors the packing contract used by llama.cpp. */
59static inline void unpack_q5_k_scales(const uint8_t *scales,
60 uint8_t *sc,
61 uint8_t *m) {
62 sc[0] = scales[0] & 0x3F;
63 sc[1] = scales[1] & 0x3F;
64 sc[2] = scales[2] & 0x3F;
65 sc[3] = scales[3] & 0x3F;
66
67 m[0] = scales[4] & 0x3F;
68 m[1] = scales[5] & 0x3F;
69 m[2] = scales[6] & 0x3F;
70 m[3] = scales[7] & 0x3F;
71
72 sc[4] = (scales[8] & 0x0F) | ((scales[0] >> 6) << 4);
73 sc[5] = (scales[9] & 0x0F) | ((scales[1] >> 6) << 4);
74 sc[6] = (scales[10] & 0x0F) | ((scales[2] >> 6) << 4);
75 sc[7] = (scales[11] & 0x0F) | ((scales[3] >> 6) << 4);
76
77 m[4] = (scales[8] >> 4) | ((scales[4] >> 6) << 4);
78 m[5] = (scales[9] >> 4) | ((scales[5] >> 6) << 4);
79 m[6] = (scales[10] >> 4) | ((scales[6] >> 6) << 4);
80 m[7] = (scales[11] >> 4) | ((scales[7] >> 6) << 4);
81}

86static inline int ck_nearest_int(float fval) {
87 float val = fval + 12582912.f;
88 int i;
89 memcpy(&i, &val, sizeof(int));
90 return (i & 0x007fffff) - 0x00400000;
91}

93static void quantize_row_q8_k_scalar(const float *x, block_q8_K *y) {
94 float max = 0.0f;
95 float amax = 0.0f;
96 for (int j = 0; j < QK_K; ++j) {
97 float ax = x[j] < 0.0f ? -x[j] : x[j];
98 if (ax > amax) {
99 amax = ax;
100 max = x[j];
101 }
102 }
103 if (amax == 0.0f) {
104 y->d = 0.0f;
105 memset(y->qs, 0, sizeof(y->qs));
106 memset(y->bsums, 0, sizeof(y->bsums));
107 return;
108 }
109
110 const float iscale = -127.0f / max;
111 for (int j = 0; j < QK_K; ++j) {
112 int q = ck_nearest_int(iscale * x[j]);
113 if (q > 127) q = 127;
114 if (q < -128) q = -128;
115 y->qs[j] = (int8_t)q;
116 }
117
118 for (int j = 0; j < QK_K / 16; ++j) {
119 int sum = 0;
120 const int8_t *q = &y->qs[j * 16];
121 for (int l = 0; l < 16; ++l) {
122 sum += q[l];
123 }
124 y->bsums[j] = (int16_t)sum;
125 }
126 y->d = 1.0f / iscale;
127}

129/* Llama-compatible dot path: Q5_K weights x Q8_K activations for one 256-value block. */
130static float dot_q5_k_q8_k_block(const block_q5_K *w, const block_q8_K *x) {
131 static const uint32_t kmask1 = 0x3f3f3f3fU;
132 static const uint32_t kmask2 = 0x0f0f0f0fU;
133 static const uint32_t kmask3 = 0x03030303U;
134
135 uint32_t utmp[4] = {0, 0, 0, 0};
136 const uint8_t *scales = (const uint8_t *)&utmp[0];
137 const uint8_t *mins = (const uint8_t *)&utmp[2];
138
139 int8_t aux8[QK_K];
140 int16_t aux16[8];
141 int32_t aux32[8];
142 memset(aux32, 0, sizeof(aux32));
143
144 const uint8_t *q4 = w->qs;
145 const uint8_t *hm = w->qh;
146 int8_t *a = aux8;
147 uint8_t m = 1;
148
149 for (int j = 0; j < QK_K / 64; ++j) {
150 for (int l = 0; l < 32; ++l) a[l] = (int8_t)(q4[l] & 0xF);
151 for (int l = 0; l < 32; ++l) a[l] += (hm[l] & m ? 16 : 0);
152 a += 32;
153 m <<= 1;
154
155 for (int l = 0; l < 32; ++l) a[l] = (int8_t)(q4[l] >> 4);
156 for (int l = 0; l < 32; ++l) a[l] += (hm[l] & m ? 16 : 0);
157 a += 32;
158 m <<= 1;
159
160 q4 += 32;
161 }
162
163 memcpy(utmp, w->scales, 12);
164 utmp[3] = ((utmp[2] >> 4) & kmask2) | (((utmp[1] >> 6) & kmask3) << 4);
165 {
166 const uint32_t uaux = utmp[1] & kmask1;
167 utmp[1] = (utmp[2] & kmask2) | (((utmp[0] >> 6) & kmask3) << 4);
168 utmp[2] = uaux;
169 }
170 utmp[0] &= kmask1;
171
172 int sumi = 0;
173 for (int j = 0; j < QK_K / 16; ++j) {
174 sumi += (int)x->bsums[j] * (int)mins[j / 2];
175 }
176
177 a = aux8;
178 const int8_t *q8 = x->qs;
179 int is = 0;
180 for (int j = 0; j < QK_K / 32; ++j) {
181 const int32_t scale = (int32_t)scales[is++];
182
183 for (int l = 0; l < 8; ++l) aux16[l] = (int16_t)(q8[l] * a[l]);
184 for (int l = 0; l < 8; ++l) aux32[l] += scale * aux16[l];
185 q8 += 8; a += 8;
186
187 for (int l = 0; l < 8; ++l) aux16[l] = (int16_t)(q8[l] * a[l]);
188 for (int l = 0; l < 8; ++l) aux32[l] += scale * aux16[l];
189 q8 += 8; a += 8;
190
191 for (int l = 0; l < 8; ++l) aux16[l] = (int16_t)(q8[l] * a[l]);
192 for (int l = 0; l < 8; ++l) aux32[l] += scale * aux16[l];
193 q8 += 8; a += 8;
194
195 for (int l = 0; l < 8; ++l) aux16[l] = (int16_t)(q8[l] * a[l]);
196 for (int l = 0; l < 8; ++l) aux32[l] += scale * aux16[l];
197 q8 += 8; a += 8;
198 }
199
200 int32_t doti = 0;
201 for (int l = 0; l < 8; ++l) {
202 doti += aux32[l];
203 }
204
205 const float d = CK_FP16_TO_FP32(w->d) * x->d;
206 const float dmin = CK_FP16_TO_FP32(w->dmin) * x->d;
207 return d * (float)doti - dmin * (float)sumi;
208}

210/* FP32 fallback for oversized K (very rare for current models). */
211static void gemv_q5_k_ref_fp32(float *y, const void *W, const float *x, int M, int K)
212{
213 const block_q5_K *blocks = (const block_q5_K *)W;
214 const int blocks_per_row = K / QK_K;
215
216 for (int m = 0; m < M; m++) {
217 const float *x_row = x;
218 float sum = 0.0f;
219
220 for (int b = 0; b < blocks_per_row; b++) {
221 const block_q5_K *block = &blocks[m * blocks_per_row + b];
222 const float d = CK_FP16_TO_FP32(block->d);
223 const float dmin = CK_FP16_TO_FP32(block->dmin);
224 const uint8_t *qh = block->qh;
225 const uint8_t *qs = block->qs;
226 uint8_t sc_arr[8], m_arr[8];
227 unpack_q5_k_scales(block->scales, sc_arr, m_arr);
228
229 for (int sb = 0; sb < 8; sb++) {
230 const float d_sub = d * (float)sc_arr[sb];
231 const float m_sub = dmin * (float)m_arr[sb];
232 const int qs_offset = sb * 16;
233 const int qh_offset = sb * 4;
234
235 for (int i = 0; i < 32; i++) {
236 const uint8_t qs_val = (uint8_t)((qs[qs_offset + i / 2] >> (4 * (i % 2))) & 0xF);
237 const uint8_t qh_bit = (uint8_t)((qh[qh_offset + i / 8] >> (i % 8)) & 1);
238 const uint8_t q = (uint8_t)(qs_val | (qh_bit << 4));
239 sum += (d_sub * (float)q - m_sub) * x_row[b * QK_K + sb * 32 + i];
240 }
241 }
242 }
243
244 y[m] = sum;
245 }

248static void gemm_nt_q5_k_ref_fp32(const float *A,
249 const void *B,
250 const float *bias,
251 float *C,
252 int M, int N, int K)
253{
254 const block_q5_K *blocks = (const block_q5_K *)B;
255 const int blocks_per_col = K / QK_K;
256
257 for (int m = 0; m < M; m++) {
258 const float *a_row = &A[m * K];
259
260 for (int n = 0; n < N; n++) {
261 float sum = 0.0f;
262 const block_q5_K *w_row = &blocks[n * blocks_per_col];
263 for (int b = 0; b < blocks_per_col; b++) {
264 const block_q5_K *block = &w_row[b];
265 const float d = CK_FP16_TO_FP32(block->d);
266 const float dmin = CK_FP16_TO_FP32(block->dmin);
267 uint8_t sc_arr[8], m_arr[8];
268 unpack_q5_k_scales(block->scales, sc_arr, m_arr);
269
270 for (int sb = 0; sb < 8; sb++) {
271 const float d_sub = d * (float)sc_arr[sb];
272 const float m_sub = dmin * (float)m_arr[sb];
273 const int qs_offset = sb * 16;
274 const int qh_offset = sb * 4;
275
276 for (int i = 0; i < 32; i++) {
277 const uint8_t qs_val = (uint8_t)((block->qs[qs_offset + i / 2] >> (4 * (i % 2))) & 0xF);
278 const uint8_t qh_bit = (uint8_t)((block->qh[qh_offset + i / 8] >> (i % 8)) & 1);
279 const uint8_t q = (uint8_t)(qs_val | (qh_bit << 4));
280 sum += (d_sub * (float)q - m_sub) * a_row[b * QK_K + sb * 32 + i];
281 }
282 }
283 }
284
285 C[m * N + n] = sum + (bias ? bias[n] : 0.0f);
286 }
287 }

290/* ============================================================================
291 * Q5_K x Q8_K Kernels (explicit contract)
292 *
293 * WHY THESE EXIST:
294 * llama.cpp's Q5_K matmul contract is "Q5_K weights x Q8_K activations".
295 * The activation quantization is part of the numerical contract, not just
296 * an optimization. If we accidentally do FP32 activation dot here, we can
297 * get large parity drift at attn_proj/mlp_down while tests still pass if
298 * they compare against FP32-dequant references.
299 *
300 * These entry points make the contract explicit in code:
301 * - gemv_q5_k_q8_k(): decode-style matrix-vector (single token)
302 * - gemm_nt_q5_k_q8_k(): prefill-style matrix-matrix
303 * ============================================================================ */
304
305void gemv_q5_k_q8_k_ref(float *y,
306 const void *W,
307 const void *x_q8,
308 int M, int K)
309{
310 if (!y || !W || !x_q8 || M <= 0 || K <= 0) {
311 return;
312 }
313 if (K % QK_K != 0) {
314 return;
315 }
316
317 const block_q5_K *blocks = (const block_q5_K *)W;
318 const block_q8_K *x = (const block_q8_K *)x_q8;
319 const int blocks_per_row = K / QK_K;
320
321 for (int m = 0; m < M; ++m) {
322 const block_q5_K *w_row = &blocks[m * blocks_per_row];
323 float sum = 0.0f;
324 for (int b = 0; b < blocks_per_row; ++b) {
325 sum += dot_q5_k_q8_k_block(&w_row[b], &x[b]);
326 }
327 y[m] = sum;
328 }
329}

361/* ============================================================================
362 * FP32 adapter path (keeps existing call sites stable)
363 *
364 * Existing generated code and orchestration call gemv_q5_k/gemm_nt_q5_k with
365 * FP32 activations. These adapter functions quantize activations to Q8_K and
366 * then call the explicit Q5_K x Q8_K kernels above.
367 * ============================================================================ */
368
369void gemv_q5_k_ref(float *y, const void *W, const float *x, int M, int K)
370{
371 if (!y || !W || !x || M <= 0 || K <= 0) {
372 return;
373 }
374 if (K % QK_K != 0) {
375 gemv_q5_k_ref_fp32(y, W, x, M, K);
376 return;
377 }
378
379 const block_q5_K *blocks = (const block_q5_K *)W;
380 const int blocks_per_row = K / QK_K;
381 if (blocks_per_row > CK_Q5K_STACK_Q8_BLOCKS) {
382 gemv_q5_k_ref_fp32(y, W, x, M, K);
383 return;
384 }
385
386 block_q8_K x_q8[CK_Q5K_STACK_Q8_BLOCKS];
387 for (int b = 0; b < blocks_per_row; ++b) {
388 quantize_row_q8_k_scalar(x + b * QK_K, &x_q8[b]);
389 }
390 gemv_q5_k_q8_k_ref(y, blocks, x_q8, M, K);
391}

393/* ============================================================================
394 * GEMM NT Reference: C = A @ B^T + bias
395 * - A: FP32 activation matrix [M, K] (quantized internally to Q8_K per row)
396 * - B: Q5_K weight matrix [N, K] (stored transposed, accessed as [N, K])
397 * - bias: Optional FP32 bias [N]
398 * - C: FP32 output matrix [M, N]
399 * ============================================================================ */
400
401void gemm_nt_q5_k_ref(const float *A,
402 const void *B,
403 const float *bias,
404 float *C,
405 int M, int N, int K)
406{
407 if (!A || !B || !C || M <= 0 || N <= 0 || K <= 0) {
408 return;
409 }
410 if (K % QK_K != 0) {
411 gemm_nt_q5_k_ref_fp32(A, B, bias, C, M, N, K);
412 return;
413 }
414
415 const block_q5_K *blocks = (const block_q5_K *)B;
416 const int blocks_per_col = K / QK_K;
417 if (blocks_per_col > CK_Q5K_STACK_Q8_BLOCKS) {
418 gemm_nt_q5_k_ref_fp32(A, B, bias, C, M, N, K);
419 return;
420 }
421
422 for (int m = 0; m < M; ++m) {
423 const float *a_row = &A[m * K];
424 block_q8_K a_q8[CK_Q5K_STACK_Q8_BLOCKS];
425 for (int b = 0; b < blocks_per_col; ++b) {
426 quantize_row_q8_k_scalar(a_row + b * QK_K, &a_q8[b]);
427 }
428 gemm_nt_q5_k_q8_k_ref(a_q8, blocks, bias, &C[m * N], 1, N, K);
429 }
